139  DerivedDM transformed(const Transformation& T) const override {
140 
141  // The transformation formulas for two-electron operators and 2-DMs are similar, but not quite the same. Instead of using T, the transformation formula for the 2-DM uses T_inverse_transpose. See also (https://gqcg-res.github.io/knowdes/spinor-transformations.html).
142 
143  // Since we're only getting T as a matrix, we should convert it to an appropriate tensor to perform contractions.
144  // Although not a necessity for the einsum implementation, it makes it a lot easier to follow the formulas.
145  const GQCP::SquareMatrix<Scalar> T_related = T.matrix().transpose().inverse();
146  const GQCP::Tensor<Scalar, 2> T_related_tensor = Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>>(T_related.data(), T_related.rows(), T_related.cols());
147 
148  // We calculate the conjugate as a tensor as well.
149  const GQCP::Tensor<Scalar, 2> T_related_conjugate = T_related_tensor.conjugate();
150 
151  // We will have to do four single contractions
152  // d(T U V W) T^*(V R) -> a(T U R W)
153  // a(T U R W) T(W S) -> b(T U R S)
154  const auto temp_1 = this->template einsum<1>("TUVW,VR->TURW", T_related_conjugate).template einsum<1>("TURW,WS->TURS", T_related_tensor);
155 
156  // T(U Q) b(T U R S) -> c(T Q R S)
157  const auto temp_2 = T_related_tensor.template einsum<1>("UQ,TURS->TQRS", temp_1);
158 
159  // T^*(T P) c(T Q R S) -> d'(P Q R S)
160  const auto transformed = T_related_conjugate.template einsum<1>("TP,TQRS->PQRS", temp_2);
161 
162  return DerivedDM {transformed};
163  }
