274  static enable_if_t<std::is_same<Z1, double>::value && std::is_same<Z2, RestrictedSpinOrbitalTag>::value, SQHamiltonian<ScalarSQOneElectronOperator, ScalarSQTwoElectronOperator>> FromFCIDUMP(const std::string& fcidump_filename) {
275 
276  std::ifstream input_file_stream = validateAndOpen(fcidump_filename, "FCIDUMP");
277 
278 
279  // Do the actual parsing
280 
281  // Get the number of orbitals to check if it's a valid FCIDUMP file
282  std::string start_line; // first line contains orbitals and electron count
283  std::getline(input_file_stream, start_line);
284  std::stringstream linestream {start_line};
285 
286  size_t K = 0;
287  char iter;
288 
289  while (linestream >> iter) {
290  if (iter == '=') {
291  linestream >> K; // right here we have the number of orbitals
292  break; // we can finish reading the linestream after we found K
293  }
294  }
295 
296  if (K == 0) {
297  throw std::invalid_argument("SQHamiltonian::FromFCIDUMP(std::string): The .FCIDUMP-file is invalid: could not read a number of orbitals.");
298  }
299 
300 
301  SquareMatrix<double> h_core = SquareMatrix<double>::Zero(K);
302  SquareRankFourTensor<double> g {K};
303  g.setZero();
304 
305  // Skip 3 lines
306  for (size_t counter = 0; counter < 3; counter++) {
307  std::getline(input_file_stream, start_line);
308  }
309 
310 
311  // Start reading in the one- and two-electron integrals
312  double x;
313  size_t i, j, a, b;
314 
315  std::string line;
316  while (std::getline(input_file_stream, line)) {
317  std::istringstream iss {line};
318 
319  // Based on what the values of the indices are, we can read one-electron integrals, two-electron integrals and the internuclear repulsion energy
320  // See also (http://hande.readthedocs.io/en/latest/manual/integrals.html)
321  // I think the documentation is a bit unclear for the two-electron integrals, but we can rest assured that FCIDUMP files give the two-electron integrals in CHEMIST's notation.
322  iss >> x >> i >> a >> j >> b;
323 
324  // Single-particle eigenvalues (skipped)
325  if ((a == 0) && (j == 0) && (b == 0)) {
326  }
327 
328  // One-electron integrals (h_core)
329  else if ((j == 0) && (b == 0)) {
330  size_t p = i - 1;
331  size_t q = a - 1;
332  h_core(p, q) = x;
333 
334  // Apply the permutational symmetry for real orbitals
335  h_core(q, p) = x;
336  }
337 
338  // Two-electron integrals are given in CHEMIST'S NOTATION, so just copy them over
339  else if ((i > 0) && (a > 0) && (j > 0) && (b > 0)) {
340  size_t p = i - 1;
341  size_t q = a - 1;
342  size_t r = j - 1;
343  size_t s = b - 1;
344  g(p, q, r, s) = x;
345 
346  // Apply the permutational symmetries for real orbitals
347  g(p, q, s, r) = x;
348  g(q, p, r, s) = x;
349  g(q, p, s, r) = x;
350 
351  g(r, s, p, q) = x;
352  g(s, r, p, q) = x;
353  g(r, s, q, p) = x;
354  g(s, r, q, p) = x;
355  }
356  } // while loop
357 
358 
359  return SQHamiltonian(ScalarSQOneElectronOperator(h_core), ScalarSQTwoElectronOperator(g));
360  }

516  template <typename Z = SpinorTag>
517  enable_if_t<std::is_same<Z, GeneralSpinorTag>::value, ScalarSQOneElectronOperator> calculateInactiveFockian(const OrbitalSpace orbital_space) const {
518 
519  const auto& h_par = this->core().parameters();
520  const auto& g_par = this->twoElectron().parameters();
521 
522  // A KISS implementation of the calculation of the (general) inactive Fockian matrix
523  auto F_par = h_par; // one-electron part
524 
525  // Two-electron part
526  for (const auto& p : orbital_space.indices()) {
527  for (const auto& q : orbital_space.indices()) {
528 
529  for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
530  F_par(p, q) += g_par(p, q, i, i) - g_par(p, i, i, q);
531  }
532  }
533  } // F elements loop
534 
535  return ScalarSQOneElectronOperator {F_par};
536  }
