76  template <typename Solver>
77  QCStructure<LinearExpansion<ONVBasis>> optimize(Solver& solver, EigenproblemEnvironment& environment) const {
78 
79  // The CI method's responsibility is to try to optimize the parameters of its method, given a solver and associated environment.
80  solver.perform(environment);
81 
82 
83  // Extract the requested number of eigenpairs from the environment and place them into the LinearExpansion wave function model. Consequently, check if the LinearExpansion fulfills the objective.
84  const auto eigenpairs = environment.eigenpairs(this->number_of_states);
85 
86  std::vector<LinearExpansion<ONVBasis>> linear_expansions {};
87  linear_expansions.reserve(number_of_states);
88 
89  std::vector<double> energies {};
90  energies.reserve(this->number_of_states);
91 
92  for (const auto& eigenpair : eigenpairs) {
93  linear_expansions.emplace_back(onv_basis, eigenpair.eigenvector());
94 
95  // TODO: We can't check for a 'NormedHamiltonianEigenvectorObjective' to be fulfilled, because there is no uniform way to let an SQHamiltonian act on a LinearExpansion.
96 
97  energies.push_back(eigenpair.eigenvalue());
98  }
99 
100 
101  // Wrap all the requested number of states into a QCStructure.
102  // Since we have already created a list of LinearExpansions, we only have to create a list of the corresponding energies.
103  return QCStructure<LinearExpansion<ONVBasis>>(energies, linear_expansions);
104  }
