74  std::shared_ptr<BaseOneElectronIntegralBuffer<IntegralScalar, N>> calculate(const Shell& shell1, const Shell& shell2) override {
75 
76  // Prepare some variables.
77  const auto K = shell1.nucleus().position();
78  const auto L = shell2.nucleus().position();
79 
80  const auto& gaussian_exponents1 = shell1.gaussianExponents();
81  const auto& gaussian_exponents2 = shell2.gaussianExponents();
82 
83  const auto& contraction_coefficients1 = shell1.contractionCoefficients();
84  const auto& contraction_coefficients2 = shell2.contractionCoefficients();
85 
86 
87  // Loop over all basis functions that are contained in the shell. Since they are contracted GTOs, we will have to generate all the primitives.
88  const auto all_cartesian_exponents1 = shell1.generateCartesianExponents();
89  const auto all_cartesian_exponents2 = shell2.generateCartesianExponents();
90  std::array<std::vector<IntegralScalar>, N> integrals; // a "buffer" that stores the calculated integrals
91 
92  for (size_t i = 0; i < N; i++) { // loop over all components of the operator
93  this->primitive_engine.prepareStateForComponent(i);
94 
95  for (const auto cartesian_exponents1 : all_cartesian_exponents1) {
96  for (const auto cartesian_exponents2 : all_cartesian_exponents2) {
97 
98  // Calculate the contracted integral as a contraction over the contraction coefficients and the primitive integrals.
99  IntegralScalar integral = 0.0;
100  for (size_t c1 = 0; c1 < shell1.contractionSize(); c1++) {
101  const auto alpha = gaussian_exponents1[c1];
102  const CartesianGTO primitive1 {alpha, cartesian_exponents1, K};
103  const auto d1 = contraction_coefficients1[c1];
104 
105  for (size_t c2 = 0; c2 < shell2.contractionSize(); c2++) {
106  const auto beta = gaussian_exponents2[c2];
107  const CartesianGTO primitive2 {beta, cartesian_exponents2, L};
108  const auto d2 = contraction_coefficients2[c2];
109 
110  const auto primitive_integral = this->primitive_engine.calculate(primitive1, primitive2);
111  integral += d1 * d2 * primitive_integral;
112  }
113  }
114 
115  integrals[i].push_back(integral);
116  }
117  }
118  }
119 
120  return std::make_shared<OneElectronIntegralBuffer<IntegralScalar, N>>(shell1.numberOfBasisFunctions(), shell2.numberOfBasisFunctions(), integrals);
121  }
122 };
