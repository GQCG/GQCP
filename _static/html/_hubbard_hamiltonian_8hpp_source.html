69  template <typename Z = Scalar>
70  enable_if_t<std::is_same<Z, double>::value, ScalarRSQOneElectronOperator<double>> core() const {
71 
72  // Prepare some variables.
73  const auto K = this->numberOfLatticeSites();
74  const auto& H = this->hoppingMatrix();
75  auto h = ScalarRSQOneElectronOperator<double>::Zero(K);
76 
77  // The one-electron hopping terms can be found on the off-diagonal elements of the hopping matrix.
78  for (size_t p = 0; p < K; p++) {
79  for (size_t q = p; q < K; q++) {
80  if (p != q) {
81  h.parameters()(p, q) = H(p, q);
82  h.parameters()(q, p) = H(q, p);
83  }
84  }
85  }
86 
87  return h;
88  }
