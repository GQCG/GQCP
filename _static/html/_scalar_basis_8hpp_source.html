131  std::vector<size_t> basisFunctionIndices(const std::function<bool(const Shell&)>& selector) const {
132 
133  const auto shells = this->shellSet().asVector();
134 
135  // Find the indices of those basis functions for which the shell selector returns true.
136  std::vector<size_t> ao_indices;
137  size_t bf_index = 0;
138  for (size_t shell_index = 0; shell_index < shells.size(); shell_index++) {
139  const auto& shell = shells[shell_index];
140 
141  // If a shell has to be included, include all indices of the basis functions in it.
142  const auto number_of_bf_in_shell = shell.numberOfBasisFunctions();
143  if (selector(shell)) {
144  for (size_t i = 0; i < number_of_bf_in_shell; i++) {
145  ao_indices.push_back(bf_index);
146  bf_index++;
147  }
148  } else {
149  // Increase the current BF index to accommodate to the next shell.
150  bf_index += number_of_bf_in_shell;
151  }
152  }
153 
154  return ao_indices;
155  }
