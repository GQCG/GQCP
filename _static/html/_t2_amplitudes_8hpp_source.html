87  static T2Amplitudes<Scalar> Perturbative(const SquareMatrix<Scalar>& f, const SquareRankFourTensor<Scalar>& V_A, const OrbitalSpace& orbital_space) {
88 
89  // Zero-initialize a tensor representation for the (occupied-occupied-virtual-virtual) T2-amplitudes t_{ij}^{ab}.
90  auto t2 = orbital_space.initializeRepresentableObjectFor<double>(OccupationType::k_occupied, OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_virtual);
91 
92 
93  // Provide the perturbative T2-amplitudes, by setting the RHS amplitudes in Stanton1991, equation (2) to zero.
94  for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
95  for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
96  for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
97  for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
98  const auto denominator = f(i, i) + f(j, j) - f(a, a) - f(b, b);
99 
100  t2(i, j, a, b) = V_A(i, j, a, b) / denominator;
101  }
102  }
103  }
104  }
105 
106  return T2Amplitudes<Scalar>(t2, orbital_space);
107  }
