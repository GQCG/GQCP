104  Self transformed(const UTransformationComponent<Scalar>& T, const Spin sigma) const {
105 
106  // The transformation formulas for two-electron operators and 2-DMs are similar, but not quite the same. Instead of using T, the transformation formula for the 2-DM uses T_inverse_transpose. See also (https://gqcg-res.github.io/knowdes/spinor-transformations.html).
107 
108  // Since we're only getting T as a matrix, we should convert it to an appropriate tensor to perform contractions.
109  // Although not a necessity for the einsum implementation, it makes it a lot easier to follow the formulas.
110  const GQCP::SquareMatrix<Scalar> T_related = T.matrix().transpose().inverse();
111  const GQCP::Tensor<Scalar, 2> T_related_tensor = Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>>(T_related.data(), T_related.rows(), T_related.cols());
112 
113  // We calculate the conjugate as a tensor as well.
114  const GQCP::Tensor<Scalar, 2> T_related_conjugate = T_related_tensor.conjugate();
115 
116 
117  // Depending on the given spin-component, we should either transform the first two, or the second two axes.
118  switch (sigma) {
119  case Spin::alpha: {
120  const auto temp = T_related_tensor.template einsum<1>("UQ,TUVW->TQVW", *this);
121  const auto transformed = T_related_conjugate.template einsum<1>("TP,TQVW->PQVW", temp);
122  return Self {transformed};
123  break;
124  }
125 
126  case Spin::beta: {
127  const auto temp = this->template einsum<1>("PQVW, WS->PQVS", T_related_tensor);
128  const auto transformed = temp.template einsum<1>("PQVS, VR->PQRS", T_related_conjugate);
129  return Self {transformed};
130  break;
131  }
132  }
133  }
