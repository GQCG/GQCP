88  static T1Amplitudes<Scalar> Perturbative(const SquareMatrix<Scalar>& f, const OrbitalSpace& orbital_space) {
89 
90  // Zero-initialize a matrix representation for the (occupied-virtual) T1-amplitudes t_i^a.
91  auto t1 = orbital_space.initializeRepresentableObjectFor<Scalar>(OccupationType::k_occupied, OccupationType::k_virtual);
92 
93  // Provide the perturbative T1-amplitudes, by setting the RHS amplitudes in Stanton1991, equation (1) to zero.
94  for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
95  for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
96  const auto denominator = f(i, i) - f(a, a);
97 
98  t1(i, a) = f(i, a) / denominator;
99  }
100  }
101 
102  return T1Amplitudes<Scalar>(t1, orbital_space);
103  }
