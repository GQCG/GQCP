53 template <typename Solver>
54 QCStructure<GQCP::QCModel::CCSD<Scalar>, Scalar> optimize(Solver& solver, CCSDEnvironment<Scalar>& environment) const {
55
56 // The CCSD method's responsibility is to try to optimize the parameters of its method, given a solver and associated environment.
57 solver.perform(environment);
58
59 // To make a QCStructure, we need the electronic (correlation) energy and the T1- and T2-amplitudes.
60 // Furthermore, the solvers only find the ground state wave function parameters, so the QCStructure only needs to contain the parameters for one state.
61 const auto& T1 = environment.t1_amplitudes.back();
62 const auto& T2 = environment.t2_amplitudes.back();
63
64 const auto E_electronic_correlation = environment.correlation_energies.back();
65 const QCModel::CCSD<Scalar> ccsd_parameters {T1, T2};
66
67 return QCStructure<GQCP::QCModel::CCSD<Scalar>, Scalar>({E_electronic_correlation}, {ccsd_parameters});
68 }
