69  template <typename Z = T>
70  static enable_if_t<std::is_same<Z, double>::value, Field<double>> ReadCubeFile(const std::string& filename) {
71 
72  // Prepare the input file.
73  std::ifstream input_file_stream = validateAndOpen(filename, "cube");
74 
75 
76  // Skip the first two comment lines.
77  std::string line;
78  std::getline(input_file_stream, line);
79  std::getline(input_file_stream, line);
80 
81 
82  // Figure out the number of atoms, in order to ignore that information later on.
83  std::getline(input_file_stream, line);
84 
85  // Split the line on any whitespace or tabs.
86  std::vector<std::string> splitted_line; // create a container for the line to be split in
87 
88  boost::trim_if(line, boost::is_any_of(" \t"));
89  boost::split(splitted_line, line, boost::is_any_of(" \t"), boost::token_compress_on);
90 
91  const auto number_of_atoms = std::stoi(splitted_line[0]);
92 
93 
94  // The next three lines contain the number of steps.
95  size_t grid_size {1}; // will contain the total grid size after parsing
96  for (size_t i = 0; i < 3; i++) {
97  std::getline(input_file_stream, line);
98 
99  // Split the line on any whitespace or tabs.
100  boost::trim_if(line, boost::is_any_of(" \t"));
101  boost::split(splitted_line, line, boost::is_any_of(" \t"), boost::token_compress_on);
102 
103  // The first column contains the number of steps.
104  grid_size *= static_cast<size_t>(std::stoll(splitted_line[0]));
105  }
106 
107 
108  // Skip the lines containing atomic information.
109  for (size_t i = 0; i < number_of_atoms; i++) {
110  std::getline(input_file_stream, line);
111  }
112 
113 
114  // Finally, read in all field values.
115  std::vector<double> field_values;
116  field_values.reserve(grid_size); // allocate memory for all the grid points
117 
118  while (std::getline(input_file_stream, line)) {
119 
120  // Split the line on any whitespace or tabs.
121  boost::trim_if(line, boost::is_any_of(" \t"));
122  boost::split(splitted_line, line, boost::is_any_of(" \t"), boost::token_compress_on);
123 
124  // Add the field values from this line to the total array.
125  for (const auto& value : splitted_line) {
126  field_values.push_back(std::stod(value)); // no worries about push_back, since we have reserved enough memory up-front
127  }
128  }
129 
130  input_file_stream.close();
131 
132  return Field<double>(field_values);
133  }

156  template <int N>
157  static Field<Vector<double, N>> ReadGridFile(const std::string& filename) {
158 
159  // Find the extension of the given path (https://stackoverflow.com/a/51992).
160  std::string filename_extension; // will contain the extension of the given filename
161  std::string::size_type idx = filename.rfind('.');
162 
163  if (idx != std::string::npos) {
164  filename_extension = filename.substr(idx + 1);
165  } else {
166  throw std::invalid_argument("Field::ReadGridFile(const std::string&): I did not find an extension in your given file name.");
167  }
168 
169 
170  // Check if the user supplied an .rgrid- or .igrid-file.
171  if ((filename_extension != "rgrid") && (filename_extension != "igrid")) {
172  throw std::invalid_argument("Field::ReadGridFile(const std::string&): The given file is not an .igrid- or .rgrid-file.");
173  }
174 
175 
176  // If the filename isn't properly converted into an input file stream, we assume the user supplied a wrong file
177  std::ifstream input_file_stream {filename};
178  if (!input_file_stream.good()) {
179  throw std::invalid_argument("validateAndOpen(const std::string&, const std::string&): The provided file name is illegible. Maybe you specified a wrong path?");
180  }
181 
182 
183  // Do the actual parsing.
184  std::vector<Vector<double, N>> field_values;
185 
186  std::string line;
187  while (std::getline(input_file_stream, line)) {
188  std::vector<std::string> splitted_line; // create a container for the line to be split in
189 
190  // Split the line on any whitespace or tabs.
191  boost::trim_if(line, boost::is_any_of(" \t"));
192  boost::split(splitted_line, line, boost::is_any_of(" \t"), boost::token_compress_on);
193 
194  // We can skip the first 4 columns (index and grid point coordinates).
195 
196  // Read the field value by converting the string(s) to a double and using Eigen::Map to instantiate the correct Vector.
197  std::vector<double> field_value_vector;
198  for (size_t index = 4; index < 4 + N; index++) {
199  field_value_vector.push_back(std::stod(splitted_line[index]));
200  }
201  Eigen::Map<Eigen::VectorXd> field_value {field_value_vector.data(), N};
202  field_values.push_back(Vector<double, N>(field_value));
203 
204 
205  // We can skip the weight column as well.
206  }
207  input_file_stream.close();
208 
209 
210  // Convert the std::vector of weights to an Array.
211  return Field<Vector<double, N>>(field_values);
212  }
