76 return "Calculate the accelerated Fock matrix, and perform a diagonalization step on it.";

85 void execute(Environment& environment) override {
86
87 if (environment.error_vectors.size() < this->minimum_subspace_dimension) {
88
89 // No acceleration is possible, so calculate the regular Fock matrix and diagonalize it.
90 GHFFockMatrixCalculation<Scalar>().execute(environment);
91 GHFFockMatrixDiagonalization<Scalar>().execute(environment);
92 return;
93 }
94
95 // Convert the deques in the environment to vectors that can be accepted by the DIIS accelerator. The total number of elements we can use in DIIS is either the maximum subspace dimension or the number of available error matrices.
96 const auto n = std::min(this->maximum_subspace_dimension, environment.error_vectors.size());
97 const std::vector<VectorX<Scalar>> error_vectors {environment.error_vectors.end() - n, environment.error_vectors.end()}; // The n-th last error vectors.
98 const std::vector<ScalarGSQOneElectronOperator<Scalar>> fock_matrices {environment.fock_matrices.end() - n, environment.fock_matrices.end()}; // The n-th last Fock matrices.
99
100 // Calculate the accelerated Fock matrix and do a diagonalization step on it.
101 const auto F_accelerated = this->diis.accelerate(fock_matrices, error_vectors);
102
103 environment.fock_matrices.push_back(F_accelerated); // The diagonalization step can only read from the environment.
104 GHFFockMatrixDiagonalization<Scalar>().execute(environment);
105 environment.fock_matrices.pop_back(); // The accelerated/extrapolated Fock matrix should not be used in further extrapolation steps, as it is not created from a density matrix.
106 }
107};
