96  static GTransformation<Scalar> FromUnrestricted(const UTransformation<Scalar>& u_transformation) {
97 
98  // The goal in this named constructor is to build up the general coefficient matrix (2K x 2K) from the unrestricted (K_sigma x K_sigma) ones.
99  const auto K_alpha = u_transformation.alpha().numberOfOrbitals();
100  const auto K_beta = u_transformation.beta().numberOfOrbitals();
101  const auto M = K_alpha + K_beta;
102 
103  SquareMatrix<Scalar> T_general = SquareMatrix<Scalar>::Zero(M);
104 
105  // alpha | 0
106  // 0 | beta
107  T_general.topLeftCorner(K_alpha, K_alpha) = u_transformation.alpha().matrix();
108  T_general.bottomRightCorner(K_beta, K_beta) = u_transformation.beta().matrix();
109 
110  return GTransformation<Scalar> {T_general, K_alpha, K_beta};
111  }
