112  static CCSDEnvironment<Scalar> PerturbativeCCSD(const GSQHamiltonian<Scalar>& sq_hamiltonian, const OrbitalSpace& orbital_space) {
113 
114  // For the CCSD environment equation, we need the inactive Fock matrix and the anti-symmetrized two-electron integrals in physicist's notation.
115  const auto f = sq_hamiltonian.calculateInactiveFockian(orbital_space).parameters();
116 
117  const auto& g_chemists = sq_hamiltonian.twoElectron();
118  const auto V_A = g_chemists.convertedToPhysicistsNotation().antisymmetrized().parameters();
119 
120 
121  const auto t1_amplitudes = T1Amplitudes<Scalar>::Perturbative(f, orbital_space);
122  const auto t2_amplitudes = T2Amplitudes<Scalar>::Perturbative(f, V_A, orbital_space);
123 
124  return CCSDEnvironment<Scalar>(t1_amplitudes, t2_amplitudes, f, V_A);
125  }
