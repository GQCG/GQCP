111  static CCSDEnvironment<Scalar> PerturbativeCCSD(const GSQHamiltonian<Scalar>& sq_hamiltonian, const OrbitalSpace& orbital_space) {
112 
113  // For the CCSD environment equation, we need the inactive Fock matrix and the anti-symmetrized two-electron integrals in physicist's notation.
114  const auto f = sq_hamiltonian.calculateInactiveFockian(orbital_space).parameters();
115 
116  const auto& g_chemists = sq_hamiltonian.twoElectron();
117  const auto V_A = g_chemists.convertedToPhysicistsNotation().antisymmetrized().parameters();
118 
119 
120  const auto t1_amplitudes = T1Amplitudes<Scalar>::Perturbative(f, orbital_space);
121  const auto t2_amplitudes = T2Amplitudes<Scalar>::Perturbative(f, V_A, orbital_space);
122 
123  return CCSDEnvironment<Scalar>(t1_amplitudes, t2_amplitudes, f, V_A);
124  }
