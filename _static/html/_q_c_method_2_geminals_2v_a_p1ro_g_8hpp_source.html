72  template <typename NonLinearSolver, typename LinearSolver>
73  QCStructure<GQCP::QCModel::vAP1roG> optimize(NonLinearSolver& non_linear_solver, NonLinearEquationEnvironment<double>& non_linear_environment, LinearSolver& linear_solver) const {
74 
75  // The vAP1roG method's responsibility is to try to:
76  // 1) optimize the geminal coefficients
77  // 2) find the associated Lagrange multipliers
78 
79  // 1. Optimize the geminal coefficients.
80  non_linear_solver.perform(non_linear_environment);
81  const auto G_optimal = AP1roGGeminalCoefficients::FromColumnMajor(non_linear_environment.variables.back(), this->N_P, this->K);
82 
83 
84  // 2. Find the associated Lagrange multipliers.
85  const auto A = QCModel::vAP1roG::calculateMultiplierResponseForceConstant(sq_hamiltonian, G_optimal); // k_lambda
86  const auto b = QCModel::vAP1roG::calculateMultiplierResponseForce(sq_hamiltonian, N_P).asVector(); // -F_lambda; column-major
87  LinearEquationEnvironment<double> linear_environment {A, b};
88 
89  linear_solver.perform(linear_environment);
90  const auto& lambda_optimal_vector = linear_environment.x; // since b was column-major, so is this vector
91 
92  const size_t rows = this->N_P; // the number of rows in the multiplier matrix
93  const size_t cols = this->K - this->N_P; // the number of columns in the multiplier matrix
94  const auto orbital_space = OrbitalSpace::Implicit({{OccupationType::k_occupied, this->N_P}, {OccupationType::k_virtual, this->K - this->N_P}});
95 
96  const MatrixX<double> lambda_optimal_matrix = MatrixX<double>::FromColumnMajorVector(lambda_optimal_vector, rows, cols); // the actual Lagrange multipliers, reshaped into a matrix
97  const auto lambda_optimal = orbital_space.createRepresentableObjectFor<double>(OccupationType::k_occupied, OccupationType::k_virtual, lambda_optimal_matrix);
98 
99 
100  // To make a QCStructure, we need the electronic energy, geminal coefficients and Lagrange multipliers.
101  // Furthermore, the solvers only find the ground state wave function parameters, so the QCStructure only needs to contain the parameters for one state.
102  const GQCP::QCModel::vAP1roG vap1rog_parameters {G_optimal, lambda_optimal};
103  const auto E_electronic = vap1rog_parameters.calculateEnergy(sq_hamiltonian);
104 
105  return QCStructure<GQCP::QCModel::vAP1roG>({E_electronic}, {vap1rog_parameters});
106  }
