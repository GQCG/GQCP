139  static Scalar calculateElectronicEnergy(const SpinResolved1DM<Scalar>& P, const ScalarUSQOneElectronOperator<Scalar>& H_core, const ScalarUSQOneElectronOperator<Scalar>& F) {
140 
141  // First, calculate the sum of H_core and F (this saves a contraction).
142  const auto Z = H_core + F;
143 
144  // Convert the matrix Z to a GQCP::Tensor<double, 2> Z_tensor.
145  // Einsum is only implemented for a tensor + a matrix, not for 2 matrices.
146  Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>> Z_alpha_t {Z.alpha().parameters().data(), Z.alpha().parameters().rows(), Z.alpha().parameters().cols()};
147  Tensor<Scalar, 2> Z_alpha_tensor = Tensor<Scalar, 2>(Z_alpha_t);
148 
149  Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>> Z_beta_t {Z.beta().parameters().data(), Z.beta().parameters().rows(), Z.beta().parameters().cols()};
150  Tensor<Scalar, 2> Z_beta_tensor = Tensor<Scalar, 2>(Z_beta_t);
151 
152  // To calculate the electronic energy, we must perform a double contraction (with prefactor 0.5):
153  // 0.5 P_sigma(mu nu) P_sigma(mu nu).
154  Tensor<Scalar, 0> contraction_alpha = 0.5 * Z_alpha_tensor.template einsum<2>("ij,ji->", P.alpha().matrix());
155 
156  Tensor<Scalar, 0> contraction_beta = 0.5 * Z_beta_tensor.template einsum<2>("ij,ji->", P.beta().matrix());
157 
158  // As the double contraction of two rank-2 tensors is a scalar (a tensor of rank 0), we should access the value as (0).
159  return contraction_alpha(0) + contraction_beta(0);
160  }

218  static SpinResolved2DM<Scalar> calculateOrthonormalBasis2DM(const size_t K, const size_t N_a, const size_t N_b) {
219 
220  // Create the orbital space to determine the loops.
221  const auto orbital_space = UHF<Scalar>::orbitalSpace(K, K, N_a, N_b);
222 
223 
224  // Use KISS formulas to implement the spin components of the UHF 2-DM.
225  SquareRankFourTensor<Scalar> d_aaaa = SquareRankFourTensor<Scalar>::Zero(K);
226  for (const auto& i : orbital_space.alpha().indices(OccupationType::k_occupied)) {
227  for (const auto& j : orbital_space.alpha().indices(OccupationType::k_occupied)) {
228  for (const auto& k : orbital_space.alpha().indices(OccupationType::k_occupied)) {
229  for (const auto& l : orbital_space.alpha().indices(OccupationType::k_occupied)) {
230  if ((i == j) && (k == l)) {
231  d_aaaa(i, j, k, l) += 1.0;
232  }
233 
234  if ((i == l) && (j == k)) {
235  d_aaaa(i, j, k, l) -= 1.0;
236  }
237  }
238  }
239  }
240  }
241 
242 
243  SquareRankFourTensor<Scalar> d_aabb = SquareRankFourTensor<Scalar>::Zero(K);
244  for (const auto& i : orbital_space.alpha().indices(OccupationType::k_occupied)) {
245  for (const auto& j : orbital_space.alpha().indices(OccupationType::k_occupied)) {
246  for (const auto& k : orbital_space.beta().indices(OccupationType::k_occupied)) {
247  for (const auto& l : orbital_space.beta().indices(OccupationType::k_occupied)) {
248  if ((i == j) && (k == l)) {
249  d_aabb(i, j, k, l) += 1.0;
250  }
251  }
252  }
253  }
254  }
255 
256 
257  SquareRankFourTensor<Scalar> d_bbaa = SquareRankFourTensor<Scalar>::Zero(K);
258  for (const auto& i : orbital_space.beta().indices(OccupationType::k_occupied)) {
259  for (const auto& j : orbital_space.beta().indices(OccupationType::k_occupied)) {
260  for (const auto& k : orbital_space.alpha().indices(OccupationType::k_occupied)) {
261  for (const auto& l : orbital_space.alpha().indices(OccupationType::k_occupied)) {
262  if ((i == j) && (k == l)) {
263  d_aabb(i, j, k, l) += 1.0;
264  }
265  }
266  }
267  }
268  }
269 
270 
271  SquareRankFourTensor<Scalar> d_bbbb = SquareRankFourTensor<Scalar>::Zero(K);
272  for (const auto& i : orbital_space.beta().indices(OccupationType::k_occupied)) {
273  for (const auto& j : orbital_space.beta().indices(OccupationType::k_occupied)) {
274  for (const auto& k : orbital_space.beta().indices(OccupationType::k_occupied)) {
275  for (const auto& l : orbital_space.beta().indices(OccupationType::k_occupied)) {
276  if ((i == j) && (k == l)) {
277  d_aaaa(i, j, k, l) += 1.0;
278  }
279 
280  if ((i == l) && (j == k)) {
281  d_aaaa(i, j, k, l) -= 1.0;
282  }
283  }
284  }
285  }
286  }
287 
288  return SpinResolved2DM<Scalar> {PureSpinResolved2DMComponent<Scalar>(d_aaaa), MixedSpinResolved2DMComponent<Scalar>(d_aabb), MixedSpinResolved2DMComponent<Scalar>(d_bbaa), PureSpinResolved2DMComponent<Scalar>(d_bbbb)};
289  }

320  static ScalarUSQOneElectronOperator<Scalar> calculateScalarBasisDirectMatrix(const SpinResolved1DM<Scalar>& P, const USQHamiltonian<Scalar>& sq_hamiltonian) {
321 
322  // Get the two-electron parameters.
323  const auto& g_a = sq_hamiltonian.twoElectron().alphaAlpha().parameters();
324  const auto& g_b = sq_hamiltonian.twoElectron().betaBeta().parameters();
325 
326  // Specify the contraction pairs for the direct contractions:
327  // (mu nu|rho lambda) P(rho lambda).
328  const auto J_alpha = g_a.template einsum<2>("ijkl,kl->ij", P.alpha().matrix()).asMatrix();
329  const auto J_beta = g_b.template einsum<2>("ijkl,kl->ij", P.beta().matrix()).asMatrix();
330 
331  // Calculate the total J tensor.
332  const auto J = J_alpha + J_beta;
333 
334  return ScalarUSQOneElectronOperator<Scalar> {J, J};

369  static ScalarUSQOneElectronOperator<Scalar> calculateScalarBasisFockMatrix(const SpinResolved1DM<Scalar>& P, const USQHamiltonian<Scalar>& sq_hamiltonian) {
370 
371  // F_sigma = H_core + (J_alpha + J_beta) - K_sigma.
372  // H_core is always the same.
373  const auto& H_core = sq_hamiltonian.core();
374 
375  // Get the alpha and beta parameters of the coulomb and exchange matrices.
376  const auto J_a = UHF<Scalar>::calculateScalarBasisDirectMatrix(P, sq_hamiltonian).alpha().parameters();
377  const auto J_b = UHF<Scalar>::calculateScalarBasisDirectMatrix(P, sq_hamiltonian).beta().parameters();
378 
379  const auto K_a = UHF<Scalar>::calculateScalarBasisExchangeMatrix(P, sq_hamiltonian).alpha().parameters();
380  const auto K_b = UHF<Scalar>::calculateScalarBasisExchangeMatrix(P, sq_hamiltonian).beta().parameters();
381 
382 
383  // Generate the alpha and beta Fock matrix and put them in a USQOneElectronOperator.
384  const auto F_a = H_core.alpha().parameters() + J_a - K_a;
385  const auto F_b = H_core.beta().parameters() + J_b - K_b;
386 
387  return ScalarUSQOneElectronOperator<Scalar> {F_a, F_b};
388  }

473  GQCP::Matrix<Scalar> calculateMixedSpinConservedAComponent(const USQHamiltonian<Scalar>& usq_hamiltonian, const Spin sigma, const Spin sigma_bar) const {
474 
475  // Sigma and sigma_bar need to be different. If they are the same, the method throws an error.
476  if (sigma == sigma_bar) {
477  throw std::invalid_argument("QCModel::UHF<Scalar>.calculateMixedSpinConservedAComponent(const USQHamiltonian<Scalar>& usq_hamiltonian, const Spin sigma, const Spin sigma_bar): The spin 'sigma' and spin 'sigma_bar' arguments are not allowed to be the same.");
478  }
479 
480  // Depending on whether we are making the aabb or bbaa A'-component, we need a different component of the orbital_space.
481  const auto orbital_space = this->orbitalSpace();
482  const auto& orbital_space_sigma = orbital_space.component(sigma);
483  const auto& orbital_space_sigma_bar = orbital_space.component(sigma_bar);
484 
485  // Determine the number of occupied and virtual orbitals for both spin components.
486  const auto& n_occ_sigma = orbital_space_sigma.numberOfOrbitals(OccupationType::k_occupied);
487  const auto& n_virt_sigma = orbital_space_sigma.numberOfOrbitals(OccupationType::k_virtual);
488 
489  const auto& n_occ_sigma_bar = orbital_space_sigma_bar.numberOfOrbitals(OccupationType::k_occupied);
490  const auto& n_virt_sigma_bar = orbital_space_sigma_bar.numberOfOrbitals(OccupationType::k_virtual);
491 
492  // The mixed alpha-beta two electron integrals are extracted from the Hamiltonian.
493  const auto& g_aabb = usq_hamiltonian.twoElectron().alphaBeta().parameters();
494 
495  // The next step is to create the needed tensor slice.
496  // Zero-initialize an occupied-virtual-occupied-virtual object of mixed spins.
497  auto A_iajb_slice = orbital_space.template initializeMixedRepresentableObjectFor<Scalar>(OccupationType::k_occupied, sigma, OccupationType::k_virtual, sigma,
498  OccupationType::k_occupied, sigma_bar, OccupationType::k_virtual, sigma_bar);
499  for (const auto& i : orbital_space_sigma.indices(OccupationType::k_occupied)) {
500  for (const auto& a : orbital_space_sigma.indices(OccupationType::k_virtual)) {
501  for (const auto& j : orbital_space_sigma_bar.indices(OccupationType::k_occupied)) {
502  for (const auto& b : orbital_space_sigma_bar.indices(OccupationType::k_virtual)) {
503 
504  // Depending on which spin component is alpha and which is beta, the indices need to be transposed correctly.
505  if (sigma == Spin::alpha && sigma_bar == Spin::beta) {
506  A_iajb_slice(i, a, j, b) = g_aabb(a, i, j, b);
507  } else {
508  A_iajb_slice(i, a, j, b) = g_aabb(j, b, a, i);
509  }
510  }
511  }
512  }
513  }
514 
515  // Turn the ImplicitRankFourTensorSlice in an actual Tensor.
516  auto A_iajb = A_iajb_slice.asTensor();
517 
518  // Finally, reshape the tensor to a matrix.
519  const auto mixed_spin_conserved_A_component = A_iajb.reshape(n_occ_sigma * n_virt_sigma, n_occ_sigma_bar * n_virt_sigma_bar);
520 
521  return mixed_spin_conserved_A_component;
522  }

606  GQCP::Matrix<Scalar> calculatePureSpinConservedAComponent(const USQHamiltonian<Scalar>& usq_hamiltonian, const Spin sigma) const {
607 
608  // Depending on whether we are making the alpha or beta A'-component, we need a different component of the orbital_space.
609  const auto orbital_space_sigma = this->orbitalSpace().component(sigma);
610 
611  // Determine the number of occupied and virtual orbitals.
612  const auto& n_occ = orbital_space_sigma.numberOfOrbitals(OccupationType::k_occupied);
613  const auto& n_virt = orbital_space_sigma.numberOfOrbitals(OccupationType::k_virtual);
614 
615  // The pure alpha-alpha two electron integrals are extracted from the Hamiltonian.
616  // We need the anti-symmetrized tensor: (AI||JB) = (AI|JB) - (AB|JI). This is obtained by the `.antisymmetrized()` method.
617  const auto g_aaaa = usq_hamiltonian.twoElectron().alphaAlpha().antisymmetrized().parameters();
618 
619  // The elements F_BA and F_IJ are the eigenvalues of the one-electron Fock operator.
620  // The excitationEnergies API can be used to find these values.
621  const auto F_values = this->excitationEnergies().component(sigma);
622 
623  // The next step is to create the needed tensor slice.
624  // Zero-initialize an occupied-virtual-occupied-virtual object.
625  auto A_iajb_slice = orbital_space_sigma.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_occupied, OccupationType::k_virtual);

800  GQCP::Matrix<Scalar> calculateSpinUnconservedAComponent(const USQHamiltonian<Scalar>& usq_hamiltonian, const Spin sigma, const Spin sigma_bar) const {
801 
802  // Sigma and sigma_bar need to be different. If they are the same, the method throws an error.
803  if (sigma == sigma_bar) {
804  throw std::invalid_argument("QCModel::UHF<Scalar>.calculateMixedSpinConservedAComponent(const USQHamiltonian<Scalar>& usq_hamiltonian, const Spin sigma, const Spin sigma_bar): The spin 'sigma' and spin 'sigma_bar' arguments are not allowed to be the same.");
805  }
806 
807  // Depending on whether we're making the abab or baba A''-component, we need a different component of the orbital_space.
808  const auto orbital_space = this->orbitalSpace();
809  const auto& orbital_space_sigma = orbital_space.component(sigma);
810  const auto& orbital_space_sigma_bar = orbital_space.component(sigma_bar);
811 
812  // Determine the number of occupied and virtual orbitals.
813  const auto& n_occ_sigma = orbital_space_sigma.numberOfOrbitals(OccupationType::k_occupied);
814  const auto& n_virt_sigma = orbital_space_sigma.numberOfOrbitals(OccupationType::k_virtual);

817  const auto& n_virt_sigma_bar = orbital_space_sigma_bar.numberOfOrbitals(OccupationType::k_virtual);
818 
820  const auto& g_aabb = usq_hamiltonian.twoElectron().alphaBeta().parameters();
821 
822  // The next step is to create the needed tensor slice.
823  // Zero-initialize an occupied-virtual-occupied-virtual object of mixed spins.
824  auto A_iajb_slice = orbital_space.template initializeMixedRepresentableObjectFor<Scalar>(OccupationType::k_occupied, sigma_bar, OccupationType::k_virtual, sigma,
825  OccupationType::k_occupied, sigma_bar, OccupationType::k_virtual, sigma);
826  for (const auto& i : orbital_space_sigma_bar.indices(OccupationType::k_occupied)) {
827  for (const auto& a : orbital_space_sigma.indices(OccupationType::k_virtual)) {
828  for (const auto& j : orbital_space_sigma_bar.indices(OccupationType::k_occupied)) {
829  for (const auto& b : orbital_space_sigma.indices(OccupationType::k_virtual)) {
830 
831  // Depending on which spin component is alpha and which is beta, the indices need to be transposed correctly.
832  if (sigma == Spin::alpha && sigma_bar == Spin::beta) {
833  A_iajb_slice(i, a, j, b) = -g_aabb(a, b, j, i);
834  } else {
835  A_iajb_slice(i, a, j, b) = -g_aabb(j, i, a, b);
836  }
837  }
838  }
839  }

845  // We have to construct a mixed matrix in this particular case.
846  GQCP::MatrixX<Scalar> F_values {n_virt_sigma, n_occ_sigma_bar};
847  const auto virtual_energies = this->virtualOrbitalEnergies().component(sigma);
848  const auto occupied_energies = this->occupiedOrbitalEnergies().component(sigma_bar);
849 
850  for (int a = 0; a < n_virt_sigma; a++) {
851  for (int i = 0; i < n_occ_sigma_bar; i++) {
852  F_values(a, i) = virtual_energies[a] - occupied_energies[i];
853  }
854  }
855 
856  // Add the previously calculated F values on the correct positions.
857  for (int a = 0; a < n_virt_sigma; a++) {
858  for (int i = 0; i < n_occ_sigma_bar; i++) {
859  A_iajb(i, a, i, a) += F_values(a, i);
860  }
861  }
862 
863  // Finally, reshape the tensor to a matrix.
864  const auto spin_unconserved_A_component = A_iajb.reshape(n_occ_sigma_bar * n_virt_sigma, n_occ_sigma_bar * n_virt_sigma);
865 
866  return spin_unconserved_A_component;
