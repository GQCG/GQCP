139 static Scalar calculateElectronicEnergy(const SpinResolved1DM<Scalar>& P, const ScalarUSQOneElectronOperator<Scalar>& H_core, const ScalarUSQOneElectronOperator<Scalar>& F) {
140
141 // First, calculate the sum of H_core and F (this saves a contraction).
142 const auto Z = H_core + F;
143
144 // Convert the matrix Z to a GQCP::Tensor<double, 2> Z_tensor.
145 // Einsum is only implemented for a tensor + a matrix, not for 2 matrices.
146 Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>> Z_alpha_t {Z.alpha().parameters().data(), Z.alpha().parameters().rows(), Z.alpha().parameters().cols()};
147 Tensor<Scalar, 2> Z_alpha_tensor = Tensor<Scalar, 2>(Z_alpha_t);
148
149 Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>> Z_beta_t {Z.beta().parameters().data(), Z.beta().parameters().rows(), Z.beta().parameters().cols()};
150 Tensor<Scalar, 2> Z_beta_tensor = Tensor<Scalar, 2>(Z_beta_t);
151
152 // To calculate the electronic energy, we must perform a double contraction (with prefactor 0.5):
153 // 0.5 Z_sigma(mu nu) P_sigma(mu nu).
154 Tensor<Scalar, 0> contraction_alpha = 0.5 * Z_alpha_tensor.template einsum<2>("ij,ij->", P.alpha().matrix());
155 Tensor<Scalar, 0> contraction_beta = 0.5 * Z_beta_tensor.template einsum<2>("ij,ij->", P.beta().matrix());
156
157 // As the double contraction of two rank-2 tensors is a scalar (a tensor of rank 0), we should access the value as (0).
158 return contraction_alpha(0) + contraction_beta(0);
159 }

169 static SpinResolved<SquareMatrix<Scalar>> calculateError(const ScalarUSQOneElectronOperator<Scalar>& F, const SpinResolved1DM<Scalar>& D, const ScalarUSQOneElectronOperator<Scalar>& S) {
170
171 // Calculate the alpha and beta error vectors separately.
172 const auto error_vector_alpha = F.alpha().parameters() * D.alpha().matrix() * S.alpha().parameters() - S.alpha().parameters() * D.alpha().matrix() * F.alpha().parameters();
173 const auto error_vector_beta = F.beta().parameters() * D.beta().matrix() * S.beta().parameters() - S.beta().parameters() * D.beta().matrix() * F.beta().parameters();
174
175 return SpinResolved<SquareMatrix<Scalar>> {error_vector_alpha, error_vector_beta};

189 static SpinResolved1DM<Scalar> calculateOrthonormalBasis1DM(const size_t K_a, const size_t K_b, const size_t N_a, const size_t N_b) {
190
191 // The (alpha) 1-DM for UHF looks like (for K_alpha=5, N_alpha=3):
192 // 1 0 0 0 0
193 // 0 1 0 0 0
194 // 0 0 1 0 0
195 // 0 0 0 0 0
196 // 0 0 0 0 0
197
198 SquareMatrix<Scalar> D_MO_a = SquareMatrix<Scalar>::Zero(K_a);
199 D_MO_a.topLeftCorner(N_a, N_a) = SquareMatrix<Scalar>::Identity(N_a);
200
201 SquareMatrix<Scalar> D_MO_b = SquareMatrix<Scalar>::Zero(K_b);
202 D_MO_b.topLeftCorner(N_b, N_b) = SquareMatrix<Scalar>::Identity(N_b);
203
204 return SpinResolved1DM<Scalar> {SpinResolved1DMComponent<Scalar> {D_MO_a}, SpinResolved1DMComponent<Scalar> {D_MO_b}};

217 static SpinResolved2DM<Scalar> calculateOrthonormalBasis2DM(const size_t K, const size_t N_a, const size_t N_b) {
218
219 // Create the orbital space to determine the loops.
220 const auto orbital_space = UHF<Scalar>::orbitalSpace(K, K, N_a, N_b);
221
222
223 // Use KISS formulas to implement the spin components of the UHF 2-DM.
224 SquareRankFourTensor<Scalar> d_aaaa = SquareRankFourTensor<Scalar>::Zero(K);
225 for (const auto& i : orbital_space.alpha().indices(OccupationType::k_occupied)) {
226 for (const auto& j : orbital_space.alpha().indices(OccupationType::k_occupied)) {
227 for (const auto& k : orbital_space.alpha().indices(OccupationType::k_occupied)) {
228 for (const auto& l : orbital_space.alpha().indices(OccupationType::k_occupied)) {
229 if ((i == j) && (k == l)) {
230 d_aaaa(i, j, k, l) += 1.0;
231 }
232
233 if ((i == l) && (j == k)) {
234 d_aaaa(i, j, k, l) -= 1.0;
235 }
236 }
237 }
238 }
239 }
240
241
242 SquareRankFourTensor<Scalar> d_aabb = SquareRankFourTensor<Scalar>::Zero(K);
243 for (const auto& i : orbital_space.alpha().indices(OccupationType::k_occupied)) {
244 for (const auto& j : orbital_space.alpha().indices(OccupationType::k_occupied)) {
245 for (const auto& k : orbital_space.beta().indices(OccupationType::k_occupied)) {
246 for (const auto& l : orbital_space.beta().indices(OccupationType::k_occupied)) {
247 if ((i == j) && (k == l)) {
248 d_aabb(i, j, k, l) += 1.0;
249 }
250 }
251 }
252 }
253 }
254
255
256 SquareRankFourTensor<Scalar> d_bbaa = SquareRankFourTensor<Scalar>::Zero(K);
257 for (const auto& i : orbital_space.beta().indices(OccupationType::k_occupied)) {
258 for (const auto& j : orbital_space.beta().indices(OccupationType::k_occupied)) {
259 for (const auto& k : orbital_space.alpha().indices(OccupationType::k_occupied)) {
260 for (const auto& l : orbital_space.alpha().indices(OccupationType::k_occupied)) {
261 if ((i == j) && (k == l)) {
262 d_aabb(i, j, k, l) += 1.0;
263 }
264 }
265 }
266 }
267 }
268
269
270 SquareRankFourTensor<Scalar> d_bbbb = SquareRankFourTensor<Scalar>::Zero(K);
271 for (const auto& i : orbital_space.beta().indices(OccupationType::k_occupied)) {
272 for (const auto& j : orbital_space.beta().indices(OccupationType::k_occupied)) {
273 for (const auto& k : orbital_space.beta().indices(OccupationType::k_occupied)) {
274 for (const auto& l : orbital_space.beta().indices(OccupationType::k_occupied)) {
275 if ((i == j) && (k == l)) {
276 d_aaaa(i, j, k, l) += 1.0;
277 }
278
279 if ((i == l) && (j == k)) {
280 d_aaaa(i, j, k, l) -= 1.0;
281 }
282 }
283 }
284 }
285 }
286
287 return SpinResolved2DM<Scalar> {PureSpinResolved2DMComponent<Scalar>(d_aaaa), MixedSpinResolved2DMComponent<Scalar>(d_aabb), MixedSpinResolved2DMComponent<Scalar>(d_bbaa), PureSpinResolved2DMComponent<Scalar>(d_bbbb)};
288 }

338 static ScalarUSQOneElectronOperator<Scalar> calculateScalarBasisDirectMatrix(const SpinResolved1DM<Scalar>& P, const USQHamiltonian<Scalar>& sq_hamiltonian) {
339
340 // Get the two-electron parameters.
341 const auto& g_a = sq_hamiltonian.twoElectron().alphaAlpha().parameters();
342 const auto& g_b = sq_hamiltonian.twoElectron().betaBeta().parameters();
343
344 // Specify the contraction pairs for the direct contractions:
345 // (mu nu|rho lambda) P(rho lambda).
346 const auto J_alpha = g_a.template einsum<2>("ijkl,kl->ij", P.alpha().matrix()).asMatrix();
347 const auto J_beta = g_b.template einsum<2>("ijkl,kl->ij", P.beta().matrix()).asMatrix();
348
349 // Calculate the total J tensor.
350 const auto J = J_alpha + J_beta;
351
352 return ScalarUSQOneElectronOperator<Scalar> {J, J};
353 }

364 static ScalarUSQOneElectronOperator<Scalar> calculateScalarBasisExchangeMatrix(const SpinResolved1DM<Scalar>& P, const USQHamiltonian<Scalar>& sq_hamiltonian) {
365
366 // Get the two-electron parameters.
367 const auto& g_a = sq_hamiltonian.twoElectron().alphaAlpha().parameters();
368 const auto& g_b = sq_hamiltonian.twoElectron().betaBeta().parameters();
369
370 // Specify the contraction pairs for the exchange contraction:
371 // (mu rho|lambda nu) P(lambda rho).
372 const auto K_alpha = g_a.template einsum<2>("ijkl,kj->il", P.alpha().matrix()).asMatrix();
373 const auto K_beta = g_b.template einsum<2>("ijkl,kj->il", P.beta().matrix()).asMatrix();
374
375 return ScalarUSQOneElectronOperator<Scalar> {K_alpha, K_beta};

387 static ScalarUSQOneElectronOperator<Scalar> calculateScalarBasisFockMatrix(const SpinResolved1DM<Scalar>& P, const USQHamiltonian<Scalar>& sq_hamiltonian) {
388
389 // F_sigma = H_core + (J_alpha + J_beta) - K_sigma.
390 // H_core is always the same.
391 const auto& H_core = sq_hamiltonian.core();
392
393 // Get the alpha and beta parameters of the coulomb and exchange matrices.
394 const auto J_a = UHF<Scalar>::calculateScalarBasisDirectMatrix(P, sq_hamiltonian).alpha().parameters();
395 const auto J_b = UHF<Scalar>::calculateScalarBasisDirectMatrix(P, sq_hamiltonian).beta().parameters();
396
397 const auto K_a = UHF<Scalar>::calculateScalarBasisExchangeMatrix(P, sq_hamiltonian).alpha().parameters();
398 const auto K_b = UHF<Scalar>::calculateScalarBasisExchangeMatrix(P, sq_hamiltonian).beta().parameters();
399
400
401 // Generate the alpha and beta Fock matrix and put them in a USQOneElectronOperator.
402 const auto F_a = H_core.alpha().parameters() + J_a - K_a;
403 const auto F_b = H_core.beta().parameters() + J_b - K_b;
404
405 return ScalarUSQOneElectronOperator<Scalar> {F_a, F_b};
406 }

507 GQCP::Matrix<Scalar> calculateMixedSpinConservedAComponent(const USQHamiltonian<Scalar>& usq_hamiltonian, const Spin sigma, const Spin sigma_bar) const {
508
509 // Sigma and sigma_bar need to be different. If they are the same, the method throws an error.
510 if (sigma == sigma_bar) {
511 throw std::invalid_argument("QCModel::UHF<Scalar>.calculateMixedSpinConservedAComponent(const USQHamiltonian<Scalar>& usq_hamiltonian, const Spin sigma, const Spin sigma_bar): The spin 'sigma' and spin 'sigma_bar' arguments are not allowed to be the same.");
512 }
513
514 // Depending on whether we are making the aabb or bbaa A'-component, we need a different component of the orbital_space.
515 const auto orbital_space = this->orbitalSpace();
516 const auto& orbital_space_sigma = orbital_space.component(sigma);
517 const auto& orbital_space_sigma_bar = orbital_space.component(sigma_bar);
518
519 // Determine the number of occupied and virtual orbitals for both spin components.
520 const auto& n_occ_sigma = orbital_space_sigma.numberOfOrbitals(OccupationType::k_occupied);
521 const auto& n_virt_sigma = orbital_space_sigma.numberOfOrbitals(OccupationType::k_virtual);
522
523 const auto& n_occ_sigma_bar = orbital_space_sigma_bar.numberOfOrbitals(OccupationType::k_occupied);
524 const auto& n_virt_sigma_bar = orbital_space_sigma_bar.numberOfOrbitals(OccupationType::k_virtual);
525
526 // The mixed alpha-beta two electron integrals are extracted from the Hamiltonian.
527 const auto& g_aabb = usq_hamiltonian.twoElectron().alphaBeta().parameters();
528
529 // The next step is to create the needed tensor slice.
530 // Zero-initialize an occupied-virtual-occupied-virtual object of mixed spins.
531 auto A_iajb_slice = orbital_space.template initializeMixedRepresentableObjectFor<Scalar>(OccupationType::k_occupied, sigma, OccupationType::k_virtual, sigma,
532 OccupationType::k_occupied, sigma_bar, OccupationType::k_virtual, sigma_bar);
533 for (const auto& i : orbital_space_sigma.indices(OccupationType::k_occupied)) {
534 for (const auto& a : orbital_space_sigma.indices(OccupationType::k_virtual)) {
535 for (const auto& j : orbital_space_sigma_bar.indices(OccupationType::k_occupied)) {
536 for (const auto& b : orbital_space_sigma_bar.indices(OccupationType::k_virtual)) {
537
538 // Depending on which spin component is alpha and which is beta, the indices need to be transposed correctly.
539 if (sigma == Spin::alpha && sigma_bar == Spin::beta) {
540 A_iajb_slice(i, a, j, b) = g_aabb(a, i, j, b);
541 } else {
542 A_iajb_slice(i, a, j, b) = g_aabb(j, b, a, i);
543 }
544 }
545 }
546 }
547 }
548
549 // Turn the ImplicitRankFourTensorSlice in an actual Tensor.
550 auto A_iajb = A_iajb_slice.asTensor();
551
552 // Finally, reshape the tensor to a matrix.
553 const auto mixed_spin_conserved_A_component = A_iajb.reshape(n_occ_sigma * n_virt_sigma, n_occ_sigma_bar * n_virt_sigma_bar);
554
555 return mixed_spin_conserved_A_component;
556 }

640 GQCP::Matrix<Scalar> calculatePureSpinConservedAComponent(const USQHamiltonian<Scalar>& usq_hamiltonian, const Spin sigma) const {
641
642 // Depending on whether we are making the alpha or beta A'-component, we need a different component of the orbital_space.
643 const auto orbital_space_sigma = this->orbitalSpace().component(sigma);
644
645 // Determine the number of occupied and virtual orbitals.
646 const auto& n_occ = orbital_space_sigma.numberOfOrbitals(OccupationType::k_occupied);
647 const auto& n_virt = orbital_space_sigma.numberOfOrbitals(OccupationType::k_virtual);
648
649 // The pure alpha-alpha two electron integrals are extracted from the Hamiltonian.
650 // We need the anti-symmetrized tensor: (AI||JB) = (AI|JB) - (AB|JI). This is obtained by the `.antisymmetrized()` method.
651 const auto g_aaaa = usq_hamiltonian.twoElectron().alphaAlpha().antisymmetrized().parameters();
652
653 // The elements F_BA and F_IJ are the eigenvalues of the one-electron Fock operator.
654 // The excitationEnergies API can be used to find these values.
655 const auto F_values = this->excitationEnergies().component(sigma);
656
657 // The next step is to create the needed tensor slice.
658 // Zero-initialize an occupied-virtual-occupied-virtual object.
659 auto A_iajb_slice = orbital_space_sigma.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_occupied, OccupationType::k_virtual);
660 for (const auto& i : orbital_space_sigma.indices(OccupationType::k_occupied)) {
661 for (const auto& a : orbital_space_sigma.indices(OccupationType::k_virtual)) {

834 GQCP::Matrix<Scalar> calculateSpinUnconservedAComponent(const USQHamiltonian<Scalar>& usq_hamiltonian, const Spin sigma, const Spin sigma_bar) const {
835
836 // Sigma and sigma_bar need to be different. If they are the same, the method throws an error.
837 if (sigma == sigma_bar) {
838 throw std::invalid_argument("QCModel::UHF<Scalar>.calculateMixedSpinConservedAComponent(const USQHamiltonian<Scalar>& usq_hamiltonian, const Spin sigma, const Spin sigma_bar): The spin 'sigma' and spin 'sigma_bar' arguments are not allowed to be the same.");
839 }
840
841 // Depending on whether we're making the abab or baba A''-component, we need a different component of the orbital_space.
842 const auto orbital_space = this->orbitalSpace();
843 const auto& orbital_space_sigma = orbital_space.component(sigma);
844 const auto& orbital_space_sigma_bar = orbital_space.component(sigma_bar);
845
846 // Determine the number of occupied and virtual orbitals.
847 const auto& n_occ_sigma = orbital_space_sigma.numberOfOrbitals(OccupationType::k_occupied);
848 const auto& n_virt_sigma = orbital_space_sigma.numberOfOrbitals(OccupationType::k_virtual);
849
850 const auto& n_occ_sigma_bar = orbital_space_sigma_bar.numberOfOrbitals(OccupationType::k_occupied);
851 const auto& n_virt_sigma_bar = orbital_space_sigma_bar.numberOfOrbitals(OccupationType::k_virtual);
852
854 const auto& g_aabb = usq_hamiltonian.twoElectron().alphaBeta().parameters();
855
856 // The next step is to create the needed tensor slice.
857 // Zero-initialize an occupied-virtual-occupied-virtual object of mixed spins.
858 auto A_iajb_slice = orbital_space.template initializeMixedRepresentableObjectFor<Scalar>(OccupationType::k_occupied, sigma_bar, OccupationType::k_virtual, sigma,
859 OccupationType::k_occupied, sigma_bar, OccupationType::k_virtual, sigma);
860 for (const auto& i : orbital_space_sigma_bar.indices(OccupationType::k_occupied)) {
861 for (const auto& a : orbital_space_sigma.indices(OccupationType::k_virtual)) {
862 for (const auto& j : orbital_space_sigma_bar.indices(OccupationType::k_occupied)) {
863 for (const auto& b : orbital_space_sigma.indices(OccupationType::k_virtual)) {
864
865 // Depending on which spin component is alpha and which is beta, the indices need to be transposed correctly.
866 if (sigma == Spin::alpha && sigma_bar == Spin::beta) {
867 A_iajb_slice(i, a, j, b) = -g_aabb(a, b, j, i);
868 } else {
869 A_iajb_slice(i, a, j, b) = -g_aabb(j, i, a, b);
870 }
871 }
872 }
873 }
874 }
875 // Turn the ImplicitRankFourTensorSlice in an actual Tensor.
876 auto A_iajb = A_iajb_slice.asTensor();
877
878 // The elements F_BA and F_IJ are the eigenvalues of the one-electron Fock operator.
879 // We have to construct a mixed matrix in this particular case.
880 GQCP::MatrixX<Scalar> F_values {n_virt_sigma, n_occ_sigma_bar};
881 const auto virtual_energies = this->virtualOrbitalEnergies().component(sigma);
882 const auto occupied_energies = this->occupiedOrbitalEnergies().component(sigma_bar);
883
884 for (int a = 0; a < n_virt_sigma; a++) {
885 for (int i = 0; i < n_occ_sigma_bar; i++) {
886 F_values(a, i) = virtual_energies[a] - occupied_energies[i];
887 }
888 }
889
890 // Add the previously calculated F values on the correct positions.
891 for (int a = 0; a < n_virt_sigma; a++) {
892 for (int i = 0; i < n_occ_sigma_bar; i++) {
893 A_iajb(i, a, i, a) += F_values(a, i);
894 }
895 }
896
897 // Finally, reshape the tensor to a matrix.
898 const auto spin_unconserved_A_component = A_iajb.reshape(n_occ_sigma_bar * n_virt_sigma, n_occ_sigma_bar * n_virt_sigma);
899
900 return spin_unconserved_A_component;
901 }
