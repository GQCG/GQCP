104  static RHFSCFEnvironment<Scalar> WithCoreGuess(const size_t N, const RSQHamiltonian<Scalar>& sq_hamiltonian, const ScalarRSQOneElectronOperator<Scalar>& S) {
105 
106  const auto& H_core = sq_hamiltonian.core().parameters(); // In AO basis.

125  template <typename Z = Scalar>
126  static enable_if_t<std::is_same<Z, complex>::value, RHFSCFEnvironment<complex>> WithComplexlyTransformedCoreGuess(const size_t N, const RSQHamiltonian<Scalar>& sq_hamiltonian, const ScalarRSQOneElectronOperator<Scalar>& S) {
127 
128  // Set up the lambda function used to transform the coefficient matrix.
129  const auto transformation_function = [](SquareMatrix<complex> C_initial) {
130  // Define the complex constant used to transform the initial coefficient matrix.
131  const complex x {0, 0.1};
132 
133  // Add/subtract the small complex value from the off-diagonal elements of the initial coefficient matrix.
134  for (size_t i = 0; i < C_initial.cols(); i++) {
135  C_initial(0, i) += x;
136  }
137  for (size_t j = 0; j < C_initial.rows(); j++) {
138  C_initial(j, 0) -= x;
139  }
140 
141  // Return the updated coefficient matrix.
142  return C_initial;
143  };
144 
145  return RHFSCFEnvironment<complex>::WithTransformedCoreGuess(N, sq_hamiltonian, S, transformation_function);
146  }
