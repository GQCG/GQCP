136  void prepareDMCalculation(const RSQHamiltonian<double>& sq_hamiltonian) override {
137 
138  // (Re)create the eigenproblem environment in the current orbital basis.
139  if (this->eigenproblem_environment.A.cols() != 0) { // If the optimization environment is 'dense'.
140  this->eigenproblem_environment = CIEnvironment::Dense(sq_hamiltonian, this->onv_basis);
141  } else {
142 
143  // Recreate the iterative eigenproblem environment with the previous guesses.
144  if (this->number_of_iterations != 0) { // Not needed when we haven't done an OO iteration.
145  const auto dim = this->onv_basis.dimension();
146  MatrixX<double> V = MatrixX<double>::Zero(dim, this->number_of_requested_eigenpairs);
147 
148  const auto m_eigenpairs = this->eigenproblem_environment.eigenpairs(this->number_of_requested_eigenpairs);
149  for (size_t i = 0; i < this->number_of_requested_eigenpairs; i++) {
150  V.col(i) = m_eigenpairs[i].eigenvector();
151  }
152 
153  this->eigenproblem_environment = CIEnvironment::Iterative(sq_hamiltonian, this->onv_basis, V);
154  }
155  }
156 
157  // Set the ground state expansion and the possibly requested excited states.
158  this->ground_state_expansion = QCMethod::CI<double, ONVBasis>(this->onv_basis, this->number_of_requested_eigenpairs).optimize(this->eigenproblem_solver, this->eigenproblem_environment).groundStateParameters();
159  this->m_eigenpairs = eigenproblem_environment.eigenpairs(this->number_of_requested_eigenpairs);
160  }
