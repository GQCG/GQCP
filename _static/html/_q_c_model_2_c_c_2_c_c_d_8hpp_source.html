72 static Scalar calculateCorrelationEnergy(const SquareMatrix<Scalar>& f, const SquareRankFourTensor<Scalar>& V_A, const T2Amplitudes<Scalar>& t2) {
73
74 const auto orbital_space = t2.orbitalSpace();
75
76 // A KISS implementation of the CCD energy correction.
77 // The implementation is in line with Crawford2000 "Chapter 2: An Introduction to Coupled Cluster Theory for Computational Chemists", eq. [134].
78 Scalar E {}; // Default initialization to 0.
79
80 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
81 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
82 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
83 for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
84 E += 0.25 * V_A(i, j, a, b) * t2(i, j, a, b);
85 }
86 }
87 }
88 }
89
90 return E;
91 }

113 static Scalar calculateT2AmplitudeEquation(const size_t i, const size_t j, const size_t a, const size_t b, const SquareMatrix<Scalar>& f, const SquareRankFourTensor<Scalar>& V_A, const T2Amplitudes<Scalar>& t2, const ImplicitMatrixSlice<Scalar>& F1, const ImplicitMatrixSlice<Scalar>& F2, const ImplicitRankFourTensorSlice<Scalar>& W1, const ImplicitRankFourTensorSlice<Scalar>& W2, const ImplicitRankFourTensorSlice<Scalar>& W3) {
114
115 const auto orbital_space = t2.orbitalSpace();
116
117 // We will use equation (2) in Stanton1991 by putting the left-hand term (with the energy denominator) to the right.
118 Scalar result {0.0}; // zero-initialize the scalar value for the result
119
120 // Calculate the contribution from the left-hand side.
121 const auto D_ijab = f(i, i) + f(j, j) - f(a, a) - f(b, b);
122 result -= t2(i, j, a, b) * D_ijab;
123
124 // Calculate the contribution from the first term.
125 result += V_A(i, j, a, b);
126
127 // Calculate the contribution from the second term.
128 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
129 result += t2(i, j, a, e) * F1(b, e);
130 result -= t2(i, j, b, e) * F1(a, e); // P(ab) applied
131 }
132
133 // Calculate the contribution from the third term.
134 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
135 result -= t2(i, m, a, b) * F2(m, j);
136 result += t2(j, m, a, b) * F2(m, i); // P(ij) applied
137 }
138
139 // Calculate the contribution from the fourth term.
140 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
141 for (const auto& n : orbital_space.indices(OccupationType::k_occupied)) {
142 result += 0.5 * t2(m, n, a, b) * W1(m, n, i, j);
143 }
144 }
145
146 // Calculate the contribution from the fifth term.
147 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
148 for (const auto& f : orbital_space.indices(OccupationType::k_virtual)) {
149 result += 0.5 * t2(i, j, e, f) * W2(a, b, e, f);
150 }
151 }
152
153 // Calculate the contribution from the sixth term.
154 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
155 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
156 result += t2(i, m, a, e) * W3(m, b, e, j);
157 result -= t2(j, m, a, e) * W3(m, b, e, i); // P(ij) applied
158 result -= t2(i, m, b, e) * W3(m, a, e, j); // P(ab) applied
159 result += t2(j, m, b, e) * W3(m, a, e, i); // P(ij) P(ab) applied
160 }
161 }
162
163 return result;
164 }

176 static ImplicitMatrixSlice<Scalar> calculateF1(const SquareMatrix<Scalar>& f, const SquareRankFourTensor<Scalar>& V_A, const T2Amplitudes<Scalar>& t2) {
177
178 const auto& orbital_space = t2.orbitalSpace();
179
180 // Implement the formula for the F1-intermediate: equation (3) in Stanton1993.
181 auto F1 = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_virtual, OccupationType::k_virtual); // zero-initialize a virtual-virtual object
182 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
183 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
184 Scalar value {0.0}; // zero-initialize the scalar value to be added
185
186 // Calculate the contribution from the first term.
187 if (a != e) { // (1 - delta_ae)
188 value += f(a, e);
189 }
190
191 // Calculate the contribution from the fourth term.
192 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
193 for (const auto& n : orbital_space.indices(OccupationType::k_occupied)) {
194 for (const auto& f : orbital_space.indices(OccupationType::k_virtual)) {
195 value -= 0.5 * t2(m, n, a, f) * V_A(m, n, e, f);
196 }
197 }
198 }
199
200 F1(a, e) = value;
201 }
202 }
203
204 return F1;
205 }
