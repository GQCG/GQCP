61

65  const auto& t1 = environment.t1_amplitudes.back();
66  const auto& t2 = environment.t2_amplitudes.back();
67 
68  // First, calculate the intermediate tau2 objects (since they depend on the the T1- and T2-amplitudes).
69  environment.tau2 = QCModel::CCSD<Scalar>::calculateTau2(t1, t2);
70  environment.tau2_tilde = QCModel::CCSD<Scalar>::calculateTau2Tilde(t1, t2);
71

74 
75  // Calculate the other CCSD intermediates and push them to the environment.

78  environment.F3 = QCModel::CCSD<Scalar>::calculateF3(f, V_A, t1);
79 
80  environment.W1 = QCModel::CCSD<Scalar>::calculateW1(V_A, t1, tau2);
