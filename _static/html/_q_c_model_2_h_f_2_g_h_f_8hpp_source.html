102  static Scalar calculateElectronicEnergy(const G1DM<Scalar>& P, const ScalarGSQOneElectronOperator<Scalar>& H_core, const ScalarGSQOneElectronOperator<Scalar>& F) {
103 
104  // First, calculate the sum of H_core and F (this saves a contraction).
105  const auto Z = H_core + F;
106 
107  // Convert the matrix Z to a Tensor.
108  // Einsum is only implemented for a tensor + a matrix, not for 2 matrices.
109  Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>> Z_t {Z.parameters().data(), P.rows(), P.cols()};
110  Tensor<Scalar, 2> Z_tensor = Tensor<Scalar, 2>(Z_t);
111 
112  // To calculate the electronic energy, we must perform a double contraction (with prefactor 0.5):
113  // 0.5 D(mu nu) Z(mu nu).
114  // See knowdes: https://gqcg-res.github.io/knowdes/general-hartree-fock-theory.html
115 
116  Tensor<Scalar, 0> contraction = 0.5 * Z_tensor.template einsum<2>("ij, ij ->", P);
117 
118  // As the double contraction of two matrices is a scalar (a tensor of rank 0), we should access the value as (0).
119  return contraction(0);
120  }

149  static ScalarGSQOneElectronOperator<Scalar> calculateScalarBasisDirectMatrix(const G1DM<Scalar>& P, const GSQHamiltonian<Scalar>& sq_hamiltonian) {
150 
151  // To perform the contraction, we will first have to convert the density matrix into a Eigen::Tensor (since contractions are only implemented for tensors).
152  // Since the two-electron integrals are spin-blocked (due to the nature of quantizing in a GSpinorBasis), the contractions must happen with a density matrix of the same dimension (M: the number of spinors). Therefore, we will construct a zero density matrix in which we only fill in one of the spin-blocks.
153  const auto M = P.dimension(); // The total number of basis functions.
154  G1DM<Scalar> P_aa = G1DM<Scalar>::Zero(M);
155  P_aa.topLeftCorner(M / 2, M / 2) = P.topLeftCorner(M / 2, M / 2);
156 
157  G1DM<Scalar> P_bb = G1DM<Scalar>::Zero(M);
158  P_bb.bottomRightCorner(M / 2, M / 2) = P.bottomRightCorner(M / 2, M / 2);
159 
160  // Specify the contraction pairs for the direct contractions:
161  // P(rho lambda) (mu nu|rho lambda).
162  // See knowdes: https://gqcg-res.github.io/knowdes/derivation-of-the-ghf-scf-equations-through-lagrange-multipliers.html
163  const auto& g = sq_hamiltonian.twoElectron().parameters();
164  const auto Ja = g.template einsum<2>("ijkl,kl->ij", P_aa).asMatrix();
165  const auto Jb = g.template einsum<2>("ijkl,kl->ij", P_bb).asMatrix();
166 
167 
168  return ScalarGSQOneElectronOperator<Scalar>(Ja + Jb);
169  }

180  static ScalarGSQOneElectronOperator<Scalar> calculateScalarBasisExchangeMatrix(const G1DM<Scalar>& P, const GSQHamiltonian<Scalar>& sq_hamiltonian) {
181 
182  // To perform the contraction, we will first have to convert the density matrix into a Eigen::Tensor (since contractions are only implemented for tensors).
183  // Since the two-electron integrals are spin-blocked (due to the nature of quantizing in a GSpinorBasis), the contractions must happen with a density matrix of the same dimension (M: the number of spinors). Therefore, we will construct a zero density matrix in which we only fill in one of the spin-blocks.
184  const auto M = P.dimension(); // The total number of basis functions.
185 
186  G1DM<Scalar> P_aa = G1DM<Scalar>::Zero(M);
187  P_aa.topLeftCorner(M / 2, M / 2) = P.topLeftCorner(M / 2, M / 2);
188 
189  G1DM<Scalar> P_ab = G1DM<Scalar>::Zero(M);
190  P_ab.topRightCorner(M / 2, M / 2) = P.topRightCorner(M / 2, M / 2);
191 
192  G1DM<Scalar> P_ba = G1DM<Scalar>::Zero(M);
193  P_ba.bottomLeftCorner(M / 2, M / 2) = P.bottomLeftCorner(M / 2, M / 2);
194 
195  G1DM<Scalar> P_bb = G1DM<Scalar>::Zero(M);
196  P_bb.bottomRightCorner(M / 2, M / 2) = P.bottomRightCorner(M / 2, M / 2);
197 
198 
199  // Specify the contraction pairs for the exchange contractions:
200  // P(lambda rho) (mu rho|lambda nu).
201  const auto& g = sq_hamiltonian.twoElectron().parameters();
202  const auto K_aa = g.template einsum<2>("ijkl,kj->il", P_aa).asMatrix();
203  const auto K_ab = g.template einsum<2>("ijkl,kj->il", P_ba).asMatrix();
204  const auto K_ba = g.template einsum<2>("ijkl,kj->il", P_ab).asMatrix();
205  const auto K_bb = g.template einsum<2>("ijkl,kj->il", P_bb).asMatrix();
206 
207  // Each of the spin-blocks are calculated separately (while the other blocks are zero), so the total exchange matrix can be calculated as the sum of each part.
208  return ScalarGSQOneElectronOperator<Scalar>(K_aa + K_ab + K_ba + K_bb);
209  }

220  static ScalarGSQOneElectronOperator<Scalar> calculateScalarBasisFockMatrix(const G1DM<Scalar>& P, const GSQHamiltonian<Scalar>& sq_hamiltonian) {
221

238  static G1DM<Scalar> calculateOrthonormalBasis1DM(const size_t M, const size_t N) {
239 
240  // The 1-DM for GHF looks like (for M=5, N=3):
241  // 1 0 0 0 0
242  // 0 1 0 0 0
243  // 0 0 1 0 0
244  // 0 0 0 0 0
245  // 0 0 0 0 0
246 
247  G1DM<Scalar> D = G1DM<Scalar>::Zero(M);
248  D.topLeftCorner(N, N) = SquareMatrix<Scalar>::Identity(N);
249 
250  return D;
251  }

262  static G2DM<Scalar> calculateOrthonormalBasis2DM(const size_t M, const size_t N) {
263 
264  // Create the orbital space to determine the loops.
265  const auto orbital_space = GHF<Scalar>::orbitalSpace(M, N);
266 
267 
268  // Implement a KISS formula for the GHF 2-DM.
269  G2DM<Scalar> d = G2DM<Scalar>::Zero(M);
270 
271  for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
272  for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
273  for (const auto& k : orbital_space.indices(OccupationType::k_occupied)) {
274  for (const auto& l : orbital_space.indices(OccupationType::k_occupied)) {
275  if ((i == j) && (k == l)) {
276  d(i, j, k, l) += 1.0;
277  }
278 
279  if ((i == l) && (j == k)) {
280  d(i, j, k, l) -= 1.0;
281  }
282  }
283  }
284  }
285  }
286 
287  return d;
288  }

313  Vector<complex, 3> calculateExpectationValueOf(const ElectronicSpinOperator& spin_op, const ScalarGSQOneElectronOperator<Scalar>& S) const {
314 
315  // Prepare some variables.
316  const auto M = this->numberOfSpinors();
317  const auto C_alpha = this->expansion().alpha();
318  const auto C_beta = this->expansion().beta();
319  const SquareMatrix<complex> S_AO = S.parameters().topLeftCorner(M / 2, M / 2); // Assume equal for alpha and beta.
320 
321  // Calculate overlaps between the alpha- and beta-spinors.
322  const MatrixX<complex> overlap_aa = C_alpha.adjoint() * S_AO * C_alpha;
323  const MatrixX<complex> overlap_ab = C_alpha.adjoint() * S_AO * C_beta;
324  const MatrixX<complex> overlap_ba = overlap_ab.adjoint();
325  const MatrixX<complex> overlap_bb = C_beta.adjoint() * S_AO * C_beta;
326 
327  // Create the orbital space for the GHF wavefunction model.
328  const GQCP::OrbitalSpace orbital_space = this->orbitalSpace(this->numberOfSpinors(), this->numberOfElectrons());
329 
330  // A KISS implementation of the expectation value of S, from knowdes. (https://gqcg-res.github.io/knowdes/spin-expectation-values-for-ghf.html)
331  complex s_x {0.0};
332  for (const auto& I : orbital_space.indices(OccupationType::k_occupied)) { // Loop over the occupied spinors.
333  s_x += overlap_ab(I, I).real();
334  }
335 
336  complex s_y {0.0};
337  for (const auto& I : orbital_space.indices(OccupationType::k_occupied)) { // Loop over the occupied spinors.
338  s_y += overlap_ba(I, I).imag();
339  }
340 
341  complex s_z {0.0};
342  for (const auto& I : orbital_space.indices(OccupationType::k_occupied)) { // Loop over the occupied spinors.
343  s_z += 0.5 * (overlap_aa(I, I) - overlap_bb(I, I));
344  }
345 
346  Vector<complex, 3> s_expectation_value = Vector<complex, 3>::Zero();
347  s_expectation_value << s_x, s_y, s_z;
348 
349  return s_expectation_value;
350  }

358  GQCP::MatrixX<Scalar> excitationEnergies() const {
359 
360  // Create the orbital space to determine the loops.
361  const auto orbital_space = this->orbitalSpace();
362 
363  // Determine the number of occupied and virtual orbitals.
364  const auto& n_occ = orbital_space.numberOfOrbitals(OccupationType::k_occupied);
365  const auto& n_virt = orbital_space.numberOfOrbitals(OccupationType::k_virtual);
366 
367  // Calculate the occupied and virtual orbital energies.
368  const auto occupied_energies = this->occupiedOrbitalEnergies();
369  const auto virtual_energies = this->virtualOrbitalEnergies();
370 
371  // Create the F matrix.
372  GQCP::MatrixX<Scalar> F_values {n_virt, n_occ};
373  for (int a = 0; a < n_virt; a++) {
374  for (int i = 0; i < n_occ; i++) {
375  F_values(a, i) = virtual_energies[a] - occupied_energies[i];
376  }
377  }
378  return F_values;
379  }

414  GQCP::MatrixX<Scalar> calculatePartialStabilityMatrixA(const GSQHamiltonian<Scalar>& gsq_hamiltonian) const {
415 
416  // Create the orbital space to determine the loops.
417  const auto orbital_space = this->orbitalSpace();
418 
419  // Determine the number of occupied and virtual orbitals.
420  const auto& n_occ = orbital_space.numberOfOrbitals(OccupationType::k_occupied);
421  const auto& n_virt = orbital_space.numberOfOrbitals(OccupationType::k_virtual);
422 
423  // We need the anti-symmetrized tensor: (AI||JB) = (AI|JB) - (AB|JI). This is obtained by the `.antisymmetrized()` method.
424  const auto g = gsq_hamiltonian.twoElectron().antisymmetrized().parameters();
425 
426  // The elements F_BA and F_IJ are the eigenvalues of the one-electron Fock operator.
427  // The excitationEnergies API can be used to find these values
428  const auto F_values = this->excitationEnergies();
429 
430  // The next step is to create the needed tensor slice.
431  // Zero-initialize an occupied-virtual-occupied-virtual object.
432  auto A_iajb_slice = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_occupied, OccupationType::k_virtual);
433  for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
434  for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
435  for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
436  for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
437  A_iajb_slice(i, a, j, b) = g(a, i, j, b);
438  }
439  }
440  }
441  }
442 
443  // Turn the ImplicitRankFourTensorSlice in an actual Tensor.
444  auto A_iajb = A_iajb_slice.asTensor();
445 
446  // Add the previously calculated F values on the correct positions.
447  for (int a = 0; a < n_virt; a++) {
448  for (int i = 0; i < n_occ; i++) {
449  A_iajb(i, a, i, a) += F_values(a, i);
450  }
451  }
452 
453  // Finally, reshape the tensor to a matrix.
454  const auto A_matrix = A_iajb.reshape(n_occ * n_virt, n_occ * n_virt);
455 
456  return A_matrix;
457  }
