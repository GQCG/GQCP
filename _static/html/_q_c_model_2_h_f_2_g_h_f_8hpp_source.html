103 static Scalar calculateElectronicEnergy(const G1DM<Scalar>& P, const ScalarGSQOneElectronOperator<Scalar>& H_core, const ScalarGSQOneElectronOperator<Scalar>& F) {
104
105 // First, calculate the sum of H_core and F (this saves a contraction).
106 const auto Z = H_core + F;
107
108 // Convert the matrix Z to a Tensor.
109 // Einsum is only implemented for a tensor + a matrix, not for 2 matrices.
110 Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>> Z_t {Z.parameters().data(), P.matrix().rows(), P.matrix().cols()};
111 Tensor<Scalar, 2> Z_tensor = Tensor<Scalar, 2>(Z_t);
112
113 // To calculate the electronic energy, we must perform a double contraction (with prefactor 0.5):
114 // 0.5 D(mu nu) Z(mu nu).
115 // See knowdes: https://gqcg-res.github.io/knowdes/general-hartree-fock-theory.html
116
117 Tensor<Scalar, 0> contraction = 0.5 * Z_tensor.template einsum<2>("ij, ij ->", P.matrix());
118
119 // As the double contraction of two matrices is a scalar (a tensor of rank 0), we should access the value as (0).
120 return contraction(0);
121 }

130 static G1DM<Scalar> calculateScalarBasis1DM(const GTransformation<Scalar>& C, const size_t N) {
131
132 const size_t M = C.dimension();
133 const auto P_orthonormal = GHF<Scalar>::calculateOrthonormalBasis1DM(M, N);
134
135 // Transform the 1-DM in an orthonormal basis to the underlying scalar basis.
136 return G1DM<Scalar>(C.matrix().conjugate() * P_orthonormal.matrix() * C.matrix().transpose());
137 }

166 static ScalarGSQOneElectronOperator<Scalar> calculateScalarBasisDirectMatrix(const G1DM<Scalar>& P, const GSQHamiltonian<Scalar>& sq_hamiltonian) {
167
168 // To perform the contraction, we will first have to convert the density matrix into a Eigen::Tensor (since contractions are only implemented for tensors).
169 // Since the two-electron integrals are spin-blocked (due to the nature of quantizing in a GSpinorBasis), the contractions must happen with a density matrix of the same dimension (M: the number of spinors). Therefore, we will construct a zero density matrix in which we only fill in one of the spin-blocks.
170 const auto M = P.numberOfOrbitals(); // The total number of basis functions.
171 SquareMatrix<Scalar> P_aa = SquareMatrix<Scalar>::Zero(M);
172 P_aa.topLeftCorner(M / 2, M / 2) = P.matrix().topLeftCorner(M / 2, M / 2);
173
174 SquareMatrix<Scalar> P_bb = SquareMatrix<Scalar>::Zero(M);
175 P_bb.bottomRightCorner(M / 2, M / 2) = P.matrix().bottomRightCorner(M / 2, M / 2);
176
177 // Specify the contraction pairs for the direct contractions:
178 // P(rho lambda) (mu nu|rho lambda).
179 // See knowdes: https://gqcg-res.github.io/knowdes/derivation-of-the-ghf-scf-equations-through-lagrange-multipliers.html
180 const auto& g = sq_hamiltonian.twoElectron().parameters();
181 const auto Ja = g.template einsum<2>("ijkl,kl->ij", P_aa).asMatrix();
182 const auto Jb = g.template einsum<2>("ijkl,kl->ij", P_bb).asMatrix();
183
184
185 return ScalarGSQOneElectronOperator<Scalar>(Ja + Jb);
186 }

197 static ScalarGSQOneElectronOperator<Scalar> calculateScalarBasisExchangeMatrix(const G1DM<Scalar>& P, const GSQHamiltonian<Scalar>& sq_hamiltonian) {
198
199 // To perform the contraction, we will first have to convert the density matrix into a Eigen::Tensor (since contractions are only implemented for tensors).
200 // Since the two-electron integrals are spin-blocked (due to the nature of quantizing in a GSpinorBasis), the contractions must happen with a density matrix of the same dimension (M: the number of spinors). Therefore, we will construct a zero density matrix in which we only fill in one of the spin-blocks.
201 const auto M = P.numberOfOrbitals(); // The total number of basis functions.
202
203 SquareMatrix<Scalar> P_aa = SquareMatrix<Scalar>::Zero(M);
204 P_aa.topLeftCorner(M / 2, M / 2) = P.matrix().topLeftCorner(M / 2, M / 2);
205
206 SquareMatrix<Scalar> P_ab = SquareMatrix<Scalar>::Zero(M);
207 P_ab.topRightCorner(M / 2, M / 2) = P.matrix().topRightCorner(M / 2, M / 2);
208
209 SquareMatrix<Scalar> P_ba = SquareMatrix<Scalar>::Zero(M);
210 P_ba.bottomLeftCorner(M / 2, M / 2) = P.matrix().bottomLeftCorner(M / 2, M / 2);
211
212 SquareMatrix<Scalar> P_bb = SquareMatrix<Scalar>::Zero(M);
213 P_bb.bottomRightCorner(M / 2, M / 2) = P.matrix().bottomRightCorner(M / 2, M / 2);
214
215
216 // Specify the contraction pairs for the exchange contractions:
217 // P(lambda rho) (mu rho|lambda nu).
218 const auto& g = sq_hamiltonian.twoElectron().parameters();
219 const auto K_aa = g.template einsum<2>("ijkl,kj->il", P_aa).asMatrix();
220 const auto K_ab = g.template einsum<2>("ijkl,kj->il", P_ba).asMatrix();
221 const auto K_ba = g.template einsum<2>("ijkl,kj->il", P_ab).asMatrix();
222 const auto K_bb = g.template einsum<2>("ijkl,kj->il", P_bb).asMatrix();
223
224 // Each of the spin-blocks are calculated separately (while the other blocks are zero), so the total exchange matrix can be calculated as the sum of each part.
225 return ScalarGSQOneElectronOperator<Scalar>(K_aa + K_ab + K_ba + K_bb);
226 }

237 static ScalarGSQOneElectronOperator<Scalar> calculateScalarBasisFockMatrix(const G1DM<Scalar>& P, const GSQHamiltonian<Scalar>& sq_hamiltonian) {
238
239 const auto& H_core = sq_hamiltonian.core();
240 const auto J = QCModel::GHF<Scalar>::calculateScalarBasisDirectMatrix(P, sq_hamiltonian);
241 const auto K = QCModel::GHF<Scalar>::calculateScalarBasisExchangeMatrix(P, sq_hamiltonian);
242
243 return H_core + J - K;
244 }

255 static G1DM<Scalar> calculateOrthonormalBasis1DM(const size_t M, const size_t N) {
256
257 // The 1-DM for GHF looks like (for M=5, N=3):
258 // 1 0 0 0 0
259 // 0 1 0 0 0
260 // 0 0 1 0 0
261 // 0 0 0 0 0
262 // 0 0 0 0 0
263
264 SquareMatrix<Scalar> D = SquareMatrix<Scalar>::Zero(M);
265 D.topLeftCorner(N, N) = SquareMatrix<Scalar>::Identity(N);
266
267 return G1DM<Scalar> {D};
268 }

279 static G2DM<Scalar> calculateOrthonormalBasis2DM(const size_t M, const size_t N) {
280
281 // Create the orbital space to determine the loops.
282 const auto orbital_space = GHF<Scalar>::orbitalSpace(M, N);
283
284
285 // Implement a KISS formula for the GHF 2-DM.
286 SquareRankFourTensor<Scalar> d = SquareRankFourTensor<Scalar>::Zero(M);
287
288 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
289 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
290 for (const auto& k : orbital_space.indices(OccupationType::k_occupied)) {
291 for (const auto& l : orbital_space.indices(OccupationType::k_occupied)) {
292 if ((i == j) && (k == l)) {
293 d(i, j, k, l) += 1.0;
294 }
295
296 if ((i == l) && (j == k)) {
297 d(i, j, k, l) -= 1.0;
298 }
299 }
300 }
301 }
302 }
303
304 return G2DM<Scalar>(d);
305 }

330 Vector<complex, 3> calculateExpectationValueOf(const ElectronicSpinOperator& spin_op, const ScalarGSQOneElectronOperator<Scalar>& S) const {
331
332 // Prepare some variables.
333 const auto M = this->numberOfSpinors();
334 const auto C_alpha = this->expansion().alpha();
335 const auto C_beta = this->expansion().beta();
336 const SquareMatrix<complex> S_AO = S.parameters().topLeftCorner(M / 2, M / 2); // Assume equal for alpha and beta.
337
338 // Calculate overlaps between the alpha- and beta-spinors.
339 const MatrixX<complex> overlap_aa = C_alpha.adjoint() * S_AO * C_alpha;
340 const MatrixX<complex> overlap_ab = C_alpha.adjoint() * S_AO * C_beta;
341 const MatrixX<complex> overlap_ba = overlap_ab.adjoint();
342 const MatrixX<complex> overlap_bb = C_beta.adjoint() * S_AO * C_beta;
343
344 // Create the orbital space for the GHF wavefunction model.
345 const GQCP::OrbitalSpace orbital_space = this->orbitalSpace(this->numberOfSpinors(), this->numberOfElectrons());
346
347 // A KISS implementation of the expectation value of S, from knowdes. (https://gqcg-res.github.io/knowdes/spin-expectation-values-for-ghf.html)
348 complex s_x {0.0};
349 for (const auto& I : orbital_space.indices(OccupationType::k_occupied)) { // Loop over the occupied spinors.
350 s_x += overlap_ab(I, I).real();
351 }
352
353 complex s_y {0.0};
354 for (const auto& I : orbital_space.indices(OccupationType::k_occupied)) { // Loop over the occupied spinors.
355 s_y += overlap_ba(I, I).imag();
356 }
357
358 complex s_z {0.0};
359 for (const auto& I : orbital_space.indices(OccupationType::k_occupied)) { // Loop over the occupied spinors.
360 s_z += 0.5 * (overlap_aa(I, I) - overlap_bb(I, I));
361 }
362
363 Vector<complex, 3> s_expectation_value = Vector<complex, 3>::Zero();
364 s_expectation_value << s_x, s_y, s_z;
365
366 return s_expectation_value;
367 }

376 complex calculateExpectationValueOf(const ElectronicSpinSquaredOperator& spin_op, const ScalarGSQOneElectronOperator<Scalar>& S) const {
377
378 // Prepare some variables.
379 const auto M = this->numberOfSpinors();
380 const auto C_alpha = this->expansion().alpha();
381 const auto C_beta = this->expansion().beta();
382 const SquareMatrix<complex> S_AO = S.parameters().topLeftCorner(M / 2, M / 2); // Assume equal for alpha and beta.
383
384 // Calculate overlaps between the alpha- and beta-spinors.
385 const MatrixX<complex> overlap_aa = C_alpha.adjoint() * S_AO * C_alpha;
386 const MatrixX<complex> overlap_ab = C_alpha.adjoint() * S_AO * C_beta;
387 const MatrixX<complex> overlap_ba = overlap_ab.adjoint();
388 const MatrixX<complex> overlap_bb = C_beta.adjoint() * S_AO * C_beta;
389
390 // Create the orbital space for the GHF wavefunction model.
391 const GQCP::OrbitalSpace orbital_space = this->orbitalSpace(this->numberOfSpinors(), this->numberOfElectrons());
392
393 // A KISS implementation of the expectation value of S, from knowdes. (https://gqcg-res.github.io/knowdes/spin-expectation-values-for-ghf.html)
394 complex s_z {0.0};
395 for (const auto& I : orbital_space.indices(OccupationType::k_occupied)) { // Loop over the occupied spinors.
396 s_z += 0.5 * (overlap_aa(I, I) - overlap_bb(I, I));
397 }
398
399
400 complex s2 = s_z * (s_z + 1.0);
401 for (const auto& I : orbital_space.indices(OccupationType::k_occupied)) {
402 // Add the contribution from the second term.
403 s2 += 0.25 * (overlap_aa(I, I) + overlap_bb(I, I));
404
405 // Add the contribution from the fourth term.
406 s2 += overlap_bb(I, I);
407
408 for (const auto& J : orbital_space.indices(OccupationType::k_occupied)) {
409 // Add the contribution from the third term.
410 s2 -= 0.25 * std::norm(overlap_aa(I, J) - overlap_bb(I, J)); // `std::norm` is actually the squared norm.
411
412 // Add the contribution from the fifth term.
413 s2 += overlap_ba(I, I) * overlap_ab(J, J) - overlap_ba(I, J) * overlap_ab(J, I);
414 }
415 }
416
417 return s2;
418 }

426 GQCP::MatrixX<Scalar> excitationEnergies() const {
427
428 // Create the orbital space to determine the loops.
429 const auto orbital_space = this->orbitalSpace();
430
431 // Determine the number of occupied and virtual orbitals.
432 const auto& n_occ = orbital_space.numberOfOrbitals(OccupationType::k_occupied);
433 const auto& n_virt = orbital_space.numberOfOrbitals(OccupationType::k_virtual);
434
435 // Calculate the occupied and virtual orbital energies.
436 const auto occupied_energies = this->occupiedOrbitalEnergies();
437 const auto virtual_energies = this->virtualOrbitalEnergies();
438
439 // Create the F matrix.
440 GQCP::MatrixX<Scalar> F_values {n_virt, n_occ};
441 for (int a = 0; a < n_virt; a++) {
442 for (int i = 0; i < n_occ; i++) {
443 F_values(a, i) = virtual_energies[a] - occupied_energies[i];
444 }
445 }
446 return F_values;
447 }

482 GQCP::MatrixX<Scalar> calculatePartialStabilityMatrixA(const GSQHamiltonian<Scalar>& gsq_hamiltonian) const {
483
484 // Create the orbital space to determine the loops.
485 const auto orbital_space = this->orbitalSpace();
486
487 // Determine the number of occupied and virtual orbitals.
488 const auto& n_occ = orbital_space.numberOfOrbitals(OccupationType::k_occupied);
489 const auto& n_virt = orbital_space.numberOfOrbitals(OccupationType::k_virtual);
490
491 // We need the anti-symmetrized tensor: (AI||JB) = (AI|JB) - (AB|JI). This is obtained by the `.antisymmetrized()` method.
492 const auto g = gsq_hamiltonian.twoElectron().antisymmetrized().parameters();
493
494 // The elements F_BA and F_IJ are the eigenvalues of the one-electron Fock operator.
495 // The excitationEnergies API can be used to find these values
496 const auto F_values = this->excitationEnergies();
497
498 // The next step is to create the needed tensor slice.
499 // Zero-initialize an occupied-virtual-occupied-virtual object.
500 auto A_iajb_slice = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_occupied, OccupationType::k_virtual);
501 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
502 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
503 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
504 for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
505 A_iajb_slice(i, a, j, b) = g(a, i, j, b);
506 }
507 }
508 }
509 }
510
511 // Turn the ImplicitRankFourTensorSlice in an actual Tensor.
512 auto A_iajb = A_iajb_slice.asTensor();
513
514 // Add the previously calculated F values on the correct positions.
515 for (int a = 0; a < n_virt; a++) {
516 for (int i = 0; i < n_occ; i++) {
517 A_iajb(i, a, i, a) += F_values(a, i);
518 }
519 }
520
521 // Finally, reshape the tensor to a matrix.
522 const auto A_matrix = A_iajb.reshape(n_occ * n_virt, n_occ * n_virt);
523
524 return A_matrix;
525 }

538 GQCP::MatrixX<Scalar> calculatePartialStabilityMatrixB(const GSQHamiltonian<Scalar>& gsq_hamiltonian) const {
539
540 // Create the orbital space to determine the loops.
541 const auto orbital_space = this->orbitalSpace();
542
543 // Determine the number of occupied and virtual orbitals.
544 const auto& n_occ = orbital_space.numberOfOrbitals(OccupationType::k_occupied);
545 const auto& n_virt = orbital_space.numberOfOrbitals(OccupationType::k_virtual);
546
547 // We need the anti-symmetrized tensor: (AI||BJ) = (AI|BJ) - (AJ|BI). This is obtained by the `.antisymmetrized()` method.
548 const auto g = gsq_hamiltonian.twoElectron().antisymmetrized().parameters();
549
550 // The next step is to create the needed tensor slice.
551 // Zero-initialize an occupied-virtual-occupied-virtual object.
552 auto B_iajb_slice = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_occupied, OccupationType::k_virtual);
553 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
554 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
555 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
556 for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
557 B_iajb_slice(i, a, j, b) = g(a, i, b, j);

567 const auto B_matrix = B_iajb.reshape(n_occ * n_virt, n_occ * n_virt);

623 std::vector<Scalar> occupiedOrbitalEnergies() const {
624

626 const auto n_occ = this->orbitalSpace().numberOfOrbitals(OccupationType::k_occupied);
627

630 mo_energies.push_back(this->orbitalEnergy(i));
631 }
632
633 // Add the values with indices smaller than the occupied orbital indices, to the new vector.
634 std::vector<Scalar> mo_energies_occupied;
635 std::copy(mo_energies.begin(), mo_energies.begin() + n_occ, std::back_inserter(mo_energies_occupied));
636 return mo_energies_occupied;
