126  const McMurchieDavidsonCoefficient E_x {K_x, a, L_x, b};
127  const McMurchieDavidsonCoefficient E_y {K_y, a, L_y, b};

134 
135  // Prepare the Hermite Coulomb integral.
136  const double p = a + b;
137  const double q = c + d;
138  const double alpha = p * q / (p + q);
139 
140  const Vector<double, 3> P {E_x.centerOfMass(), E_y.centerOfMass(), E_z.centerOfMass()};
141  const Vector<double, 3> Q {E_x_.centerOfMass(), E_y_.centerOfMass(), E_z_.centerOfMass()};
142 
143  const HermiteCoulombIntegral R {alpha, P, Q};
144 
145 
146  // Calculate the Coulomb repulsion integrals over the primitives.
147  double integral {};
148  for (int t = 0; t <= i + j; t++) {
149  for (int u = 0; u <= k + l; u++) {
150  for (int v = 0; v <= m + n; v++) {
151  for (int tau = 0; tau <= i_ + j_; tau++) {
152  for (int mu = 0; mu <= k_ + l_; mu++) {
153  for (int nu = 0; nu <= m_ + n_; nu++) {
154  // Add the contribution to the integral. The prefactor will be applied at the end.
155  integral += E_x(i, j, t) * E_y(k, l, u) * E_z(m, n, v) *
156  E_x_(i_, j_, tau) * E_y_(k_, l_, mu) * E_z_(m_, n_, nu) *
157  std::pow(-1, tau + mu + nu) * R(0, t + tau, u + mu, v + nu);
158  }
159  }
160  }
161  }
162  }
163  }
164 
165  return 2 * std::pow(boost::math::constants::pi<double>(), 2.5) / (p * q * std::sqrt(p + q)) * integral;
166  }
