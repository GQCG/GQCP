163  static enable_if_t<std::is_same<Z, SpinResolvedSelectedONVBasis>::value, LinearExpansion<Z>> FromGAMESSUS(const std::string& GAMESSUS_filename) {
164 
165  // If the filename isn't properly converted into an input file stream, we assume the user supplied a wrong file.
166  std::ifstream input_file_stream {GAMESSUS_filename};
167  if (!input_file_stream.good()) {
168  throw std::runtime_error("LinearExpansionReader(std::string): The provided GAMESS file is illegible. Maybe you specified a wrong path?");
169  }
170 
171  std::ifstream input_file_stream_count {GAMESSUS_filename}; // made to count the expansion size
172 
173 
174  // Do the actual parsing.
175 
176  // Read in dummy lines up until we actually get to the ONVs and coefficients.
177  std::string line;
178  std::string buffer; // dummy for the counting stream
179  while (std::getline(input_file_stream, line)) {
180  std::getline(input_file_stream_count, buffer);
181 
182 
183  if (line.find("ALPHA") != std::string::npos && line.find("BETA") != std::string::npos && line.find("COEFFICIENT") != std::string::npos) { // if find() returns an index that's different from the 'not-found' index
184 
185  // This line should have dashes and we skip it
186  std::getline(input_file_stream, line);
187  std::getline(input_file_stream_count, buffer);
188  break;
189  }
190  }
191 
192  size_t space_size = 0;
193  // Count the number of configurations.
194  while (std::getline(input_file_stream_count, buffer)) {
195  if (buffer.length() != 0 | buffer[0] != '\n') {
196  space_size++;
197  }
198  }
199 
200  VectorX<double> coefficients = VectorX<double>::Zero(space_size);
201 
202 
203  std::getline(input_file_stream, line);
204 
205  // Read the first line containing the configurations.
206  std::vector<std::string> splitted_line;
207  boost::split(splitted_line, line, boost::is_any_of("|")); // split on '|'
208 
209 
210  // Create an alpha ONV for the first field
211  std::string trimmed_alpha = boost::algorithm::trim_copy(splitted_line[0]);
212 
213  // Create a beta ONV for the second field
214  std::string trimmed_beta = boost::algorithm::trim_copy(splitted_line[1]);
215 
216 
217  size_t index_count = 0; // counts the number of configurations added
218  coefficients(index_count) = std::stod(splitted_line[2]);
219 
220 
221  // Parse the trimmed ONV strings into boost::dynamic_bitset to use its functionality.
222  std::string reversed_alpha {trimmed_alpha.rbegin(), trimmed_alpha.rend()};
223  std::string reversed_beta {trimmed_beta.rbegin(), trimmed_beta.rend()};
224 
225  boost::dynamic_bitset<> alpha_transfer {reversed_alpha};
226  boost::dynamic_bitset<> beta_transfer {reversed_beta};
227 
228  size_t K = alpha_transfer.size();
229  size_t N_alpha = alpha_transfer.count();
230  size_t N_beta = beta_transfer.count();
231 
232  SpinResolvedSelectedONVBasis onv_basis {K, N_alpha, N_beta};
233  onv_basis.expandWith(SpinResolvedONV::FromString(reversed_alpha, reversed_beta));
234 
235 
236  // Read in the ONVs and the coefficients by splitting the line on '|', and then trimming whitespace.
237  // In the GAMESS-US format, the bit strings are ordered in reverse.
238  while (std::getline(input_file_stream, line)) {
239 
240  index_count++;
241  boost::split(splitted_line, line, boost::is_any_of("|")); // split on '|'
242 
243 
244  // Create an alpha SpinUnresolvedONV for the first field
245  trimmed_alpha = boost::algorithm::trim_copy(splitted_line[0]);
246  if (trimmed_alpha.length() != K) {
247  throw std::invalid_argument("LinearExpansionReader(std::string): One of the provided alpha ONVs does not have the correct number of orbitals.");
248  }
249  reversed_alpha = std::string(trimmed_alpha.rbegin(), trimmed_alpha.rend());
250 
251  // Create a beta SpinUnresolvedONV for the second field
252  trimmed_beta = boost::algorithm::trim_copy(splitted_line[1]);
253  if (trimmed_beta.length() != K) {
254  throw std::invalid_argument("LinearExpansionReader(std::string): One of the provided beta ONVs does not have the correct number of orbitals.");
255  }
256  reversed_beta = std::string(trimmed_beta.rbegin(), trimmed_beta.rend());
257 
258 
259  // Create a double for the third field
260  coefficients(index_count) = std::stod(splitted_line[2]);
261  onv_basis.expandWith(SpinResolvedONV::FromString(reversed_alpha, reversed_beta));
262 
263  } // while getline
264 
265  return LinearExpansion<SpinResolvedSelectedONVBasis>(onv_basis, coefficients);
266  }

279  static enable_if_t<std::is_same<Z, SpinResolvedONVBasis>::value, LinearExpansion<Z>> FromONVProjection(const SpinResolvedONV& onv, const RSpinOrbitalBasis<double, GTOShell>& r_spinor_basis, const USpinOrbitalBasis<double, GTOShell>& u_spinor_basis) {
280 
281  // Determine the overlap matrices of the underlying scalar orbital bases, which is needed later on.
282  auto S_r = r_spinor_basis.overlap(); // the overlap matrix of the restricted MOs/spin-orbitals
283  S_r.transform(r_spinor_basis.expansion().inverse()); // now in AO basis
284 
285  auto S_u = u_spinor_basis.overlap(); // The overlap matrix of the unrestricted spin-orbitals.
286  S_u.transform(u_spinor_basis.expansion().inverse()); // Now in AO basis.
287 
288  if (!(S_r.parameters().isApprox(S_u.alpha().parameters(), 1.0e-08)) || !(S_r.parameters().isApprox(S_u.beta().parameters(), 1.0e-08))) {
289  throw std::invalid_argument("LinearExpansion::FromONVProjection(const SpinResolvedONV&, const RSpinOrbitalBasis<double, GTOShell>&, const USpinOrbitalBasis<double, GTOShell>&): The given spinor bases are not expressed using the same scalar orbital basis.");
290  }
291 
292 
293  // Prepare some parameters.
294  const auto& C_restricted = r_spinor_basis.expansion();
295 
296  const auto C_unrestricted = u_spinor_basis.expansion();
297 
298 
299  // Set up the required spin-resolved ONV basis.
300  const auto K = onv.numberOfSpatialOrbitals(Spin::alpha); // assume equal numbers of spin-orbitals for alpha- and beta-electrons
301  const auto N_alpha = onv.numberOfElectrons(Spin::alpha);
302  const auto N_beta = onv.numberOfElectrons(Spin::beta);
303  const SpinResolvedONVBasis onv_basis {K, N_alpha, N_beta};
304 
305 
306  // Determine the coefficients through calculating the overlap between two ONVs.
307  VectorX<double> coefficients = VectorX<double>::Zero(onv_basis.dimension());
308 
309  onv_basis.forEach([&onv, &C_unrestricted, &C_restricted, &S_r, &coefficients, &onv_basis](const SpinUnresolvedONV& alpha_onv, const size_t I_alpha, const SpinUnresolvedONV& beta_onv, const size_t I_beta) {
310  const SpinResolvedONV onv_on {alpha_onv, beta_onv}; // the spin-resolved ONV that should be projected 'on'
311 
312  const auto coefficient = onv.calculateProjection(onv_on, C_unrestricted, C_restricted, S_r.parameters());
313  const auto address = onv_basis.compoundAddress(I_alpha, I_beta);
314 
315  coefficients(address) = coefficient;
316  });
317 
318  return LinearExpansion<Z>(onv_basis, coefficients);
319  }

403  template <typename Z = ONVBasis>
404  enable_if_t<std::is_same<Z, SpinResolvedONVBasis>::value> basisTransform(const RTransformation<double>& T) {
405 
406  const auto K = onv_basis.numberOfOrbitals(); // number of spatial orbitals
407  if (K != T.numberOfOrbitals()) {
408  throw std::invalid_argument("LinearExpansion::basisTransform(const RTransformation<double>&): The number of spatial orbitals does not match the dimension of the transformation matrix.");
409  }
410 
411 
412  // LU-decompose the transformation matrix LU decomposition for T
413  const auto& lu_decomposition = T.matrix().noPivotLUDecompose();
414 
415  SquareMatrix<double> L = SquareMatrix<double>::Identity(K);
416  L.triangularView<Eigen::StrictlyLower>() = lu_decomposition[0];
417 
418  SquareMatrix<double> U = SquareMatrix<double>(lu_decomposition[1].triangularView<Eigen::Upper>());
419  SquareMatrix<double> U_inv = U.inverse();
420 
421 
422  // Calculate t (the operator which allows per-orbital transformation of the wave function)
423  SquareMatrix<double> t = SquareMatrix<double>::Identity(K) - L + U_inv;
424 
425 
426  // Set up spin-unresolved ONV basis variables for the loops over the ONVs
427  const SpinUnresolvedONVBasis& alpha_onv_basis = onv_basis.alpha();
428  const SpinUnresolvedONVBasis& beta_onv_basis = onv_basis.beta();
429 
430  auto dim_alpha = alpha_onv_basis.dimension();
431  auto dim_beta = beta_onv_basis.dimension();
432  auto N_alpha = alpha_onv_basis.numberOfElectrons();
433  auto N_beta = beta_onv_basis.numberOfElectrons();
434 
435 
441  VectorX<double> current_coefficients = this->m_coefficients; // coefficients will be updated after each orbital transform (C^(n-1)) in Helgaker
442  VectorX<double> correction_coefficients = VectorX<double>::Zero(onv_basis.dimension());
443 
444 
445  for (size_t m = 0; m < K; m++) { // iterate over all orbitals
446 
447  // Perform alpha and beta CI iterations.
448 
449  // 1) Alpha-branch
450  SpinUnresolvedONV alpha = alpha_onv_basis.constructONVFromAddress(0);
451  for (size_t I_alpha = 0; I_alpha < dim_alpha; I_alpha++) {
452  if (!alpha.isOccupied(m)) {
453  for (size_t e1 = 0; e1 < N_alpha; e1++) { // e1 (electron 1) loops over the (number of) electrons
454  size_t p = alpha.occupationIndexOf(e1); // retrieve the index of a given electron
455 
456  if (p < m) {
457  size_t address = I_alpha - alpha_onv_basis.vertexWeight(p, e1 + 1);
458  size_t e2 = e1 + 1;
459  size_t q = p + 1;
460  int sign = 1;
461 
462  alpha_onv_basis.shiftUntilNextUnoccupiedOrbital<1>(alpha, address, q, e2, sign);
463  while (q != m) {
464  q++;
465  alpha_onv_basis.shiftUntilNextUnoccupiedOrbital<1>(alpha, address, q, e2, sign);
466  }
467 
468  address += alpha_onv_basis.vertexWeight(q, e2);
469 
470  for (size_t I_beta = 0; I_beta < dim_beta; I_beta++) {
471  correction_coefficients(I_alpha * dim_beta + I_beta) += sign * t(p, m) * current_coefficients(address * dim_beta + I_beta);
472  }
473  }
474 
475  if (p > m) {
476  size_t address = I_alpha - alpha_onv_basis.vertexWeight(p, e1 + 1);
477  size_t e2 = e1 - 1;
478  size_t q = p - 1;
479  int sign = 1;
480 
481  alpha_onv_basis.shiftUntilPreviousUnoccupiedOrbital<1>(alpha, address, q, e2, sign);
482  while (q != m) {
483  q--;
484  alpha_onv_basis.shiftUntilPreviousUnoccupiedOrbital<1>(alpha, address, q, e2, sign);
485  }
486 
487  address += alpha_onv_basis.vertexWeight(q, e2 + 2);
488  for (size_t I_beta = 0; I_beta < dim_beta; I_beta++) {
489  correction_coefficients(I_alpha * dim_beta + I_beta) += sign * t(p, m) * current_coefficients(address * dim_beta + I_beta);
490  }
491  }
492  }
493 
494  } else { // if orbital m is occupied we can perform an in-place operation
495  for (size_t I_beta = 0; I_beta < dim_beta; I_beta++) {
496  correction_coefficients(I_alpha * dim_beta + I_beta) += (t(m, m) - 1) * current_coefficients(I_alpha * dim_beta + I_beta);
497  }
498  }
499 
500 
501  if (I_alpha < dim_alpha - 1) { // prevent the last permutation from occurring
502  alpha_onv_basis.transformONVToNextPermutation(alpha);
503  }
504  }
505 
506  current_coefficients += correction_coefficients;
507  correction_coefficients.setZero();
508 
509  // 2) Beta-branch
510  SpinUnresolvedONV beta = beta_onv_basis.constructONVFromAddress(0);
511 
512  for (size_t I_beta = 0; I_beta < dim_beta; I_beta++) {
513  if (!beta.isOccupied(m)) {
514  for (size_t e1 = 0; e1 < N_beta; e1++) { // e1 (electron 1) loops over the (number of) electrons
515  size_t p = beta.occupationIndexOf(e1); // retrieve the index of a given electron
516 
517  if (p < m) {
518  size_t address = I_beta - beta_onv_basis.vertexWeight(p, e1 + 1);
519  size_t e2 = e1 + 1;
520  size_t q = p + 1;
521  int sign = 1;
522 
523  beta_onv_basis.shiftUntilNextUnoccupiedOrbital<1>(beta, address, q, e2, sign);
524  while (q != m) {
525  q++;
526  beta_onv_basis.shiftUntilNextUnoccupiedOrbital<1>(beta, address, q, e2, sign);
527  }
528 
529  address += beta_onv_basis.vertexWeight(q, e2);
530 
531  for (size_t I_alpha = 0; I_alpha < dim_alpha; I_alpha++) {
532  correction_coefficients(I_alpha * dim_beta + I_beta) += sign * t(p, m) * current_coefficients(I_alpha * dim_beta + address);
533  }
534  }
535 
536  if (p > m) {
537  size_t address = I_beta - beta_onv_basis.vertexWeight(p, e1 + 1);
538  size_t e2 = e1 - 1;
539  size_t q = p - 1;
540 
541  int sign = 1;
542 
543  beta_onv_basis.shiftUntilPreviousUnoccupiedOrbital<1>(beta, address, q, e2, sign);
544  while (q != m) {
545  q--;
546  beta_onv_basis.shiftUntilPreviousUnoccupiedOrbital<1>(beta, address, q, e2, sign);
547  }
548 
549  address += beta_onv_basis.vertexWeight(q, e2 + 2);
550 
551  for (size_t I_alpha = 0; I_alpha < dim_alpha; I_alpha++) {
552  correction_coefficients(I_alpha * dim_beta + I_beta) += sign * t(p, m) * current_coefficients(I_alpha * dim_beta + address);
553  }
554  }
555  }
556 
557  } else { // if orbital m is occupied we can perform an in-place operation
558  for (size_t I_alpha = 0; I_alpha < dim_alpha; I_alpha++) {
559  correction_coefficients(I_alpha * dim_beta + I_beta) += (t(m, m) - 1) * current_coefficients(I_alpha * dim_beta + I_beta);
560  }
561  }
562 
563  if (I_beta < dim_beta - 1) { // prevent the last permutation from occurring
564  beta_onv_basis.transformONVToNextPermutation(beta);
565  }
566  }
567 
568  current_coefficients += correction_coefficients;
569  correction_coefficients.setZero();
570  }
571  this->m_coefficients = current_coefficients;
572  }

579  /*
580  * Calculate general one-electron density matrix for a spin-unresolved wave function expansion.
581  *
582  * @return The generalized one-electron density matrix.
583  */
584  template <typename Z = ONVBasis>
585  enable_if_t<std::is_same<Z, SpinUnresolvedONVBasis>::value, G1DM<double>> calculate1DM() const {
586 
587  // Prepare some variables.
588  const auto M = this->onv_basis.numberOfOrbitals();
589  const auto N = this->onv_basis.numberOfElectrons();
590  const auto dim = onv_basis.dimension(); // Dimension of the SpinUnresolvedONVBasis = number of SpinUnresolvedONVs.
591 
592  GQCP::G1DM<double> D = GQCP::G1DM<double>::Zero(M);
593 
594  SpinUnresolvedONV onv = onv_basis.constructONVFromAddress(0); // Start with ONV with address 0.
595  for (size_t J = 0; J < dim; J++) { // Loops over all possible ONV indices.
596 
597  // Loop over electrons that can be annihilated in an ONV.
598  for (size_t e1 = 0; e1 < N; e1++) {
599 
600  // Create an ONVPath for each new ONV.
601  ONVPath<SpinUnresolvedONVBasis> onv_path {onv_basis, onv};
602  const auto c_J = this->coefficient(J);
603 
604  // Figure out the orbital index of the electron that will be annihilated.
605  const auto q = onv.occupationIndexOf(e1);
606 
607 
608  // The diagonal values are a result of annihilation-creation on the same orbital index and are thus the same as the initial ONV.
609  D(q, q) += c_J * c_J;
610 
611  // For the non-diagonal values, we will create all possible matrix elements of the density matrix in the routine below.
612  onv_path.annihilate(q, e1);
613 
614  // Stop the loop if 1) the path is finished, meaning that orbital index p is at M (the total number of orbitals) and 2) if the orbital index is out of bounds after left translation of a vertical arc.
615  while (!onv_path.isFinished() && onv_path.isOrbitalIndexValid()) {
616 
617  // Find the next unoccupied orbital, i.e. the next vertical arc in the path.
618  onv_path.leftTranslateDiagonalArcUntilVerticalArc();
619 
620  // Calculate the address of the path if we would close it right now.
621  const auto I = onv_path.addressAfterCreation();
622  const auto c_I = this->coefficient(I);
623 
624 
625  // Add the density matrix elements.
626  const auto p = onv_path.orbitalIndex();
627 
628  const double value = c_I * c_J;
629 
630  D(p, q) += onv_path.sign() * value;
631  D(q, p) += onv_path.sign() * value;
632 
633  // Move orbital index such that other unoccupied orbitals can be found within the loop.
634  onv_path.leftTranslateVerticalArc();
635  }
636  }
637 
638  // Prevent last ONV since there is no possibility for an electron to be annihilated anymore.
639  if (J < dim - 1) {
640  onv_basis.transformONVToNextPermutation(onv);
641  }
642  }
643 
644  return D;
645  }

659  enable_if_t<std::is_same<Z, SpinUnresolvedONVBasis>::value, double> calculateNDMElement(const std::vector<size_t>& bra_indices, const std::vector<size_t>& ket_indices) const {

661  // The ket indices should be reversed because the annihilators on the ket should be applied from right to left
662  std::vector<size_t> ket_indices_reversed = ket_indices;
663  std::reverse(ket_indices_reversed.begin(), ket_indices_reversed.end());
664 
665 
666  double value = 0.0;
667  int sign = 1;
668  const size_t dim = this->onv_basis.dimension();
669 
670 
671  SpinUnresolvedONV bra = this->onv_basis.constructONVFromAddress(0);
672  size_t I = 0;
673  while (I < dim) { // loop over all bra addresses
674 
675  // Annihilate the bra on the bra indices
676  if (!bra.annihilateAll(bra_indices, sign)) { // if we can't annihilate, the bra doesn't change
677 
678  // Go to the beginning of the outer while loop with the next bra
679  if (I < dim - 1) { // prevent the last permutation from occurring
680  this->onv_basis.transformONVToNextPermutation(bra);
681  I++;
682  sign = 1;
683  continue;
684  } else {
685  break; // we have to jump out if we have looped over the whole bra dimension
686  }
687  }
688 
689 
690  SpinUnresolvedONV ket = this->onv_basis.constructONVFromAddress(0);
691  size_t J = 0;
692  while (J < dim) { // loop over all ket indices
693 
694  // Annihilate the ket on the ket indices
695  if (!ket.annihilateAll(ket_indices_reversed, sign)) { // if we can't annihilate, the ket doesn't change
696  // Go to the beginning of this (the inner) while loop with the next bra
697  if (J < dim - 1) { // prevent the last permutation from occurring
698  this->onv_basis.transformONVToNextPermutation(ket);
699  J++;
700  sign = 1;
701  continue;
702  } else {
703  break; // we have to jump out if we have looped over the whole ket dimension
704  }
705  }
706 
707  if (bra == ket) {
708  value += sign * this->coefficient(I) * this->coefficient(J);
709  }
710 
711  // Reset the previous ket annihilations and move to the next ket
712  if (J == dim - 1) { // prevent the last permutation from occurring
713  break; // out of the J-loop
714  }
715  ket.createAll(ket_indices_reversed);
716  this->onv_basis.transformONVToNextPermutation(ket);
717  sign = 1;
718  J++;
719  } // while J loop
720 
721  // Reset the previous bra annihilations and move to the next bra
722  if (I == dim - 1) { // prevent the last permutation from occurring
723  break; // out of the I-loop
724  }
725  bra.createAll(bra_indices);
726  this->onv_basis.transformONVToNextPermutation(bra);
727  sign = 1;
728  I++;
729  } // while I loop
730 
731  return value;
732  }
