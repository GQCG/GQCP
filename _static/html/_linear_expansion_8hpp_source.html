169 
170  // If the filename isn't properly converted into an input file stream, we assume the user supplied a wrong file.
171  std::ifstream input_file_stream {GAMESSUS_filename};
172  if (!input_file_stream.good()) {
173  throw std::runtime_error("LinearExpansionReader(std::string): The provided GAMESS file is illegible. Maybe you specified a wrong path?");
174  }
175 
176  std::ifstream input_file_stream_count {GAMESSUS_filename}; // made to count the expansion size
177 
178 
179  // Do the actual parsing.
180 
181  // Read in dummy lines up until we actually get to the ONVs and coefficients.
182  std::string line;
183  std::string buffer; // dummy for the counting stream
184  while (std::getline(input_file_stream, line)) {
185  std::getline(input_file_stream_count, buffer);
186 
187 
188  if (line.find("ALPHA") != std::string::npos && line.find("BETA") != std::string::npos && line.find("COEFFICIENT") != std::string::npos) { // if find() returns an index that's different from the 'not-found' index
189 
190  // This line should have dashes and we skip it
191  std::getline(input_file_stream, line);
192  std::getline(input_file_stream_count, buffer);
193  break;
194  }
195  }
196 
197  size_t space_size = 0;
198  // Count the number of configurations.
199  while (std::getline(input_file_stream_count, buffer)) {
200  if (buffer.length() != 0 | buffer[0] != '\n') {
201  space_size++;
202  }
203  }
204 
205  VectorX<double> coefficients = VectorX<double>::Zero(space_size);
206 
207 
208  std::getline(input_file_stream, line);
209 
210  // Read the first line containing the configurations.
211  std::vector<std::string> splitted_line;
212  boost::split(splitted_line, line, boost::is_any_of("|")); // split on '|'
213 
214 
215  // Create an alpha ONV for the first field
216  std::string trimmed_alpha = boost::algorithm::trim_copy(splitted_line[0]);
217 
218  // Create a beta ONV for the second field
219  std::string trimmed_beta = boost::algorithm::trim_copy(splitted_line[1]);
220 
221 
222  size_t index_count = 0; // counts the number of configurations added
223  coefficients(index_count) = std::stod(splitted_line[2]);
224 
225 
226  // Parse the trimmed ONV strings into boost::dynamic_bitset to use its functionality.
227  std::string reversed_alpha {trimmed_alpha.rbegin(), trimmed_alpha.rend()};
228  std::string reversed_beta {trimmed_beta.rbegin(), trimmed_beta.rend()};
229 
230  boost::dynamic_bitset<> alpha_transfer {reversed_alpha};
231  boost::dynamic_bitset<> beta_transfer {reversed_beta};
232 
233  size_t K = alpha_transfer.size();
234  size_t N_alpha = alpha_transfer.count();
235  size_t N_beta = beta_transfer.count();
236 
237  SpinResolvedSelectedONVBasis onv_basis {K, N_alpha, N_beta};
238  onv_basis.expandWith(SpinResolvedONV::FromString(reversed_alpha, reversed_beta));
239 
240 
241  // Read in the ONVs and the coefficients by splitting the line on '|', and then trimming whitespace.
242  // In the GAMESS-US format, the bit strings are ordered in reverse.
243  while (std::getline(input_file_stream, line)) {
244 
245  index_count++;
246  boost::split(splitted_line, line, boost::is_any_of("|")); // split on '|'
247 
248 
249  // Create an alpha SpinUnresolvedONV for the first field
250  trimmed_alpha = boost::algorithm::trim_copy(splitted_line[0]);
251  if (trimmed_alpha.length() != K) {
252  throw std::invalid_argument("LinearExpansionReader(std::string): One of the provided alpha ONVs does not have the correct number of orbitals.");
253  }
254  reversed_alpha = std::string(trimmed_alpha.rbegin(), trimmed_alpha.rend());
255 
256  // Create a beta SpinUnresolvedONV for the second field
257  trimmed_beta = boost::algorithm::trim_copy(splitted_line[1]);
258  if (trimmed_beta.length() != K) {
259  throw std::invalid_argument("LinearExpansionReader(std::string): One of the provided beta ONVs does not have the correct number of orbitals.");
260  }
261  reversed_beta = std::string(trimmed_beta.rbegin(), trimmed_beta.rend());
262 
263 
264  // Create a double for the third field
265  coefficients(index_count) = std::stod(splitted_line[2]);
266  onv_basis.expandWith(SpinResolvedONV::FromString(reversed_alpha, reversed_beta));
267 
268  } // The `while getline` loop.
269 
270  return LinearExpansion<double, SpinResolvedSelectedONVBasis>(onv_basis, coefficients);
271  }

300

408  template <typename Z1 = Scalar, typename Z2 = ONVBasis>
409  enable_if_t<std::is_same<Z1, double>::value && std::is_same<Z2, SpinResolvedONVBasis>::value> basisTransform(const RTransformation<double>& T) {
410 
411  const auto K = onv_basis.numberOfOrbitals(); // number of spatial orbitals
412  if (K != T.numberOfOrbitals()) {
413  throw std::invalid_argument("LinearExpansion::basisTransform(const RTransformation<double>&): The number of spatial orbitals does not match the dimension of the transformation matrix.");
414  }
415 
416 
417  // LU-decompose the transformation matrix LU decomposition for T
418  const auto& lu_decomposition = T.matrix().noPivotLUDecompose();
419 
420  SquareMatrix<double> L = SquareMatrix<double>::Identity(K);
421  L.triangularView<Eigen::StrictlyLower>() = lu_decomposition[0];
422 
423  SquareMatrix<double> U = SquareMatrix<double>(lu_decomposition[1].triangularView<Eigen::Upper>());
424  SquareMatrix<double> U_inv = U.inverse();
425 
426 
427  // Calculate t (the operator which allows per-orbital transformation of the wave function)
428  SquareMatrix<double> t = SquareMatrix<double>::Identity(K) - L + U_inv;
429 
430 
431  // Set up spin-unresolved ONV basis variables for the loops over the ONVs
432  const SpinUnresolvedONVBasis& alpha_onv_basis = onv_basis.alpha();
433  const SpinUnresolvedONVBasis& beta_onv_basis = onv_basis.beta();
434 
435  auto dim_alpha = alpha_onv_basis.dimension();
436  auto dim_beta = beta_onv_basis.dimension();
437  auto N_alpha = alpha_onv_basis.numberOfElectrons();
438  auto N_beta = beta_onv_basis.numberOfElectrons();
439 
440 
446  VectorX<double> current_coefficients = this->m_coefficients; // coefficients will be updated after each orbital transform (C^(n-1)) in Helgaker
447  VectorX<double> correction_coefficients = VectorX<double>::Zero(onv_basis.dimension());
448 
449 
450  for (size_t m = 0; m < K; m++) { // iterate over all orbitals
451 
452  // Perform alpha and beta CI iterations.
453 
454  // 1) Alpha-branch
455  SpinUnresolvedONV alpha = alpha_onv_basis.constructONVFromAddress(0);
456  for (size_t I_alpha = 0; I_alpha < dim_alpha; I_alpha++) {
457  if (!alpha.isOccupied(m)) {
458  for (size_t e1 = 0; e1 < N_alpha; e1++) { // e1 (electron 1) loops over the (number of) electrons
459  size_t p = alpha.occupationIndexOf(e1); // retrieve the index of a given electron
460 
461  if (p < m) {
462  size_t address = I_alpha - alpha_onv_basis.vertexWeight(p, e1 + 1);
463  size_t e2 = e1 + 1;
464  size_t q = p + 1;
465  int sign = 1;
466 
467  alpha_onv_basis.shiftUntilNextUnoccupiedOrbital<1>(alpha, address, q, e2, sign);
468  while (q != m) {
469  q++;
470  alpha_onv_basis.shiftUntilNextUnoccupiedOrbital<1>(alpha, address, q, e2, sign);
471  }
472 
473  address += alpha_onv_basis.vertexWeight(q, e2);
474 
475  for (size_t I_beta = 0; I_beta < dim_beta; I_beta++) {
476  correction_coefficients(I_alpha * dim_beta + I_beta) += sign * t(p, m) * current_coefficients(address * dim_beta + I_beta);
477  }
478  }
479 
480  if (p > m) {
481  size_t address = I_alpha - alpha_onv_basis.vertexWeight(p, e1 + 1);
482  size_t e2 = e1 - 1;
483  size_t q = p - 1;
484  int sign = 1;
485 
486  alpha_onv_basis.shiftUntilPreviousUnoccupiedOrbital<1>(alpha, address, q, e2, sign);
487  while (q != m) {
488  q--;
489  alpha_onv_basis.shiftUntilPreviousUnoccupiedOrbital<1>(alpha, address, q, e2, sign);
490  }
491 
492  address += alpha_onv_basis.vertexWeight(q, e2 + 2);
493  for (size_t I_beta = 0; I_beta < dim_beta; I_beta++) {
494  correction_coefficients(I_alpha * dim_beta + I_beta) += sign * t(p, m) * current_coefficients(address * dim_beta + I_beta);
495  }
496  }
497  }
498 
499  } else { // if orbital m is occupied we can perform an in-place operation
500  for (size_t I_beta = 0; I_beta < dim_beta; I_beta++) {
501  correction_coefficients(I_alpha * dim_beta + I_beta) += (t(m, m) - 1) * current_coefficients(I_alpha * dim_beta + I_beta);
502  }
503  }
504 
505 
506  if (I_alpha < dim_alpha - 1) { // prevent the last permutation from occurring
507  alpha_onv_basis.transformONVToNextPermutation(alpha);
508  }
509  }
510 
511  current_coefficients += correction_coefficients;
512  correction_coefficients.setZero();
513 
514  // 2) Beta-branch
515  SpinUnresolvedONV beta = beta_onv_basis.constructONVFromAddress(0);
516 
517  for (size_t I_beta = 0; I_beta < dim_beta; I_beta++) {
518  if (!beta.isOccupied(m)) {
519  for (size_t e1 = 0; e1 < N_beta; e1++) { // e1 (electron 1) loops over the (number of) electrons
520  size_t p = beta.occupationIndexOf(e1); // retrieve the index of a given electron
521 
522  if (p < m) {
523  size_t address = I_beta - beta_onv_basis.vertexWeight(p, e1 + 1);
524  size_t e2 = e1 + 1;
525  size_t q = p + 1;
526  int sign = 1;
527 
528  beta_onv_basis.shiftUntilNextUnoccupiedOrbital<1>(beta, address, q, e2, sign);
529  while (q != m) {
530  q++;
531  beta_onv_basis.shiftUntilNextUnoccupiedOrbital<1>(beta, address, q, e2, sign);
532  }
533 
534  address += beta_onv_basis.vertexWeight(q, e2);
535 
536  for (size_t I_alpha = 0; I_alpha < dim_alpha; I_alpha++) {
537  correction_coefficients(I_alpha * dim_beta + I_beta) += sign * t(p, m) * current_coefficients(I_alpha * dim_beta + address);
538  }
539  }
540 
541  if (p > m) {
542  size_t address = I_beta - beta_onv_basis.vertexWeight(p, e1 + 1);
543  size_t e2 = e1 - 1;
544  size_t q = p - 1;
545 
546  int sign = 1;
547 
548  beta_onv_basis.shiftUntilPreviousUnoccupiedOrbital<1>(beta, address, q, e2, sign);
549  while (q != m) {
550  q--;
551  beta_onv_basis.shiftUntilPreviousUnoccupiedOrbital<1>(beta, address, q, e2, sign);
552  }
553 
554  address += beta_onv_basis.vertexWeight(q, e2 + 2);
555 
556  for (size_t I_alpha = 0; I_alpha < dim_alpha; I_alpha++) {
557  correction_coefficients(I_alpha * dim_beta + I_beta) += sign * t(p, m) * current_coefficients(I_alpha * dim_beta + address);
558  }
559  }
560  }
561 
562  } else { // if orbital m is occupied we can perform an in-place operation
563  for (size_t I_alpha = 0; I_alpha < dim_alpha; I_alpha++) {
564  correction_coefficients(I_alpha * dim_beta + I_beta) += (t(m, m) - 1) * current_coefficients(I_alpha * dim_beta + I_beta);
565  }
566  }
567 
568  if (I_beta < dim_beta - 1) { // prevent the last permutation from occurring
569  beta_onv_basis.transformONVToNextPermutation(beta);
570  }
571  }
572 
573  current_coefficients += correction_coefficients;
574  correction_coefficients.setZero();
575  }
576  this->m_coefficients = current_coefficients;
577  }

584  /*
585  * Calculate general one-electron density matrix for a spin-unresolved wave function expansion.
586  *
587  * @return The generalized one-electron density matrix.
588  */
589  template <typename Z1 = Scalar, typename Z2 = ONVBasis>
590  enable_if_t<std::is_same<Z1, double>::value && std::is_same<Z2, SpinUnresolvedONVBasis>::value, G1DM<double>> calculate1DM() const {
591 
592  // Prepare some variables.
593  const auto M = this->onv_basis.numberOfOrbitals();
594  const auto N = this->onv_basis.numberOfElectrons();
595  const auto dim = onv_basis.dimension(); // Dimension of the SpinUnresolvedONVBasis = number of SpinUnresolvedONVs.
596 
597  GQCP::SquareMatrix<double> D = GQCP::SquareMatrix<double>::Zero(M);
598 
599  SpinUnresolvedONV onv = onv_basis.constructONVFromAddress(0); // Start with ONV with address 0.
600  for (size_t J = 0; J < dim; J++) { // Loops over all possible ONV indices.
601 
602  // Loop over electrons that can be annihilated in an ONV.
603  for (size_t e1 = 0; e1 < N; e1++) {
604 
605  // Create an ONVPath for each new ONV.
606  ONVPath<SpinUnresolvedONVBasis> onv_path {onv_basis, onv};
607  const auto c_J = this->coefficient(J);
608 
609  // Figure out the orbital index of the electron that will be annihilated.
610  const auto q = onv.occupationIndexOf(e1);
611 
612 
613  // The diagonal values are a result of annihilation-creation on the same orbital index and are thus the same as the initial ONV.
614  D(q, q) += c_J * c_J;
615 
616  // For the non-diagonal values, we will create all possible matrix elements of the density matrix in the routine below.
617  onv_path.annihilate(q, e1);
618 
619  // Stop the loop if 1) the path is finished, meaning that orbital index p is at M (the total number of orbitals) and 2) if the orbital index is out of bounds after left translation of a vertical arc.
620  while (!onv_path.isFinished() && onv_path.isOrbitalIndexValid()) {
621 
622  // Find the next unoccupied orbital, i.e. the next vertical arc in the path.
623  onv_path.leftTranslateDiagonalArcUntilVerticalArc();
624 
625  // Calculate the address of the path if we would close it right now.
626  const auto I = onv_path.addressAfterCreation();
627  const auto c_I = this->coefficient(I);
628 
629 
630  // Add the density matrix elements.
631  const auto p = onv_path.orbitalIndex();
632 
633  const double value = c_I * c_J;
634 
635  D(p, q) += onv_path.sign() * value;
636  D(q, p) += onv_path.sign() * value;
637 
638  // Move orbital index such that other unoccupied orbitals can be found within the loop.
639  onv_path.leftTranslateVerticalArc();
