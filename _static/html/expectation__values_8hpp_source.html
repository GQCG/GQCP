39 template <typename Scalar>
40 double calculateSpinSquared(const SpinResolved1DM<Scalar>& one_DMs, const SpinResolved2DM<Scalar>& two_DMs) {
41 
42  double sz = calculateSpinZ(one_DMs);
43  double s_squared = -sz;
44  const size_t K = one_DMs.numberOfOrbitals();
45  for (size_t p = 0; p < K; p++) {
46  s_squared += one_DMs.alpha()(p, p); // One-electron partition of S+S_
47  s_squared += (one_DMs.alpha()(p, p) + one_DMs.beta()(p, p)) / 4; // One-electron partition of S^2
48  for (size_t q = 0; q < K; q++) {
49  s_squared += -two_DMs.alphaBeta()(p, q, q, p); // Two-electron partition S+S_
50  s_squared += (two_DMs.alphaAlpha()(p, p, q, q) + two_DMs.betaBeta()(p, p, q, q) - two_DMs.alphaBeta()(p, p, q, q) - two_DMs.betaAlpha()(p, p, q, q)) / 4; // Two-electron partition of S^2
51  }
52  }
53  return s_squared;
54 }
