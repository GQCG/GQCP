61 template <typename QCObjective, typename Solver>
62 QCStructure<QCModel::RHF<Scalar>, Scalar> optimize(const QCObjective& objective, Solver& solver, RHFSCFEnvironment<Scalar>& environment) const {
63
64 // The RHF method's responsibility is to try to optimize the parameters of its method, given a solver and associated environment.
65 solver.perform(environment);
66
67 // To make a QCStructure<QCModel::RHF<Scalar>>, we need the electronic energy, coefficient matrix, orbital energies and the number of electrons.
68 // Furthermore, the current RHF SCF solvers only find the ground state wave function parameters, so the QCStructure only needs to contain the parameters for one state.
69 const auto& E_electronic = environment.electronic_energies.back();
70 const auto& C = environment.coefficient_matrices.back();
71 const auto& orbital_energies = environment.orbital_energies.back();
72 const auto& N_P = environment.N / 2;
73
74 const QCModel::RHF<Scalar> rhf_parameters {N_P, orbital_energies, C};
75
76 // Now that we have constructed an instance of the QCModel, we should check if the objective is fulfilled.
77 if (!objective.isSatisfiedWith(rhf_parameters)) {
78 throw std::runtime_error("QCModel::RHF::optimize(const QCObjective&, Solver&, RHFSCFEnvironment<Scalar>&): The solver produced a solution that does not fulfill the objective.");
79 }
80 return QCStructure<QCModel::RHF<Scalar>, Scalar>({E_electronic}, {rhf_parameters});
81 }
82};
