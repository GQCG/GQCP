104  StorageArray<Scalar, Vectorizer> calculateExpectationValue(const Derived2DM& d) const {
105 
106  if (this->numberOfOrbitals() != d.numberOfOrbitals()) {
107  throw std::invalid_argument("SimpleSQTwoElectronOperator::calculateExpectationValue(const Derived2DM&): The given 2-DM's dimension is not compatible with the two-electron operator.");
108  }
109 
110 
111  // Calculate the expectation value for every component of the operator.
112  const auto& parameters = this->allParameters();
113  std::vector<Scalar> expectation_values(this->numberOfComponents()); // Zero-initialize the vector with a number of elements.
114 
115  for (size_t i = 0; i < this->numberOfComponents(); i++) {
116 
117  // Perform the actual contraction (with prefactor 0.5):
118  // 0.5 g(p q r s) d(p q r s)
119  Eigen::Tensor<Scalar, 0> contraction = 0.5 * parameters[i].template einsum<4>("pqrs, pqrs->", d);
120 
121  // As the contraction is a scalar (a tensor of rank 0), we should access using `operator(0)`.
122  expectation_values[i] = contraction(0);
123  }
124 
125  return StorageArray<Scalar, Vectorizer> {expectation_values, this->array.vectorizer()}; // convert std::array to Vector
126  }

140  enable_if_t<std::is_same<Z, double>::value, StorageArray<SquareMatrix<double>, Vectorizer>> calculateFockianMatrix(const Derived1DM& D, const Derived2DM& d) const {
141 
142  if (D.numberOfOrbitals() != this->numberOfOrbitals()) {
143  throw std::invalid_argument("SimpleSQTwoElectronOperator::calculateFockianMatrix(const Derived1DM&, const Derived2DM&): The 1-DM's dimensions are not compatible with this two-electron operator.");
144  }
145 
146  if (d.numberOfOrbitals() != this->numberOfOrbitals()) {
147  throw std::invalid_argument("SimpleSQTwoElectronOperator::calculateFockianMatrix(const Derived1DM&, const Derived2DM&): The 2-DM's dimensions are not compatible with this two-electron operator.");
148  }
149 
150  const auto& parameters = this->allParameters(); // The parameters of this two-electron operator, as a vector.
151  std::vector<SquareMatrix<double>> F_vector {this->numberOfComponents()}; // The resulting vector of Fockian matrices.
152 
153  // A KISS implementation of the calculation of the Fockian matrix.
154  for (size_t i = 0; i < this->numberOfComponents(); i++) {
155 
156  const auto& g_i = parameters[i]; // The matrix representation of the parameters of the i-th component.
157 
158  // Calculate the Fockian matrix for every component and add it to the vector.
159  SquareMatrix<Scalar> F_i = SquareMatrix<double>::Zero(this->numberOfOrbitals());
160  for (size_t p = 0; p < this->numberOfOrbitals(); p++) {
161  for (size_t q = 0; q < this->numberOfOrbitals(); q++) {
162 
163  for (size_t r = 0; r < this->numberOfOrbitals(); r++) {
164  for (size_t s = 0; s < this->numberOfOrbitals(); s++) {
165  for (size_t t = 0; t < this->numberOfOrbitals(); t++) {
166  F_i(p, q) += 0.5 * g_i(q, r, s, t) * (d(p, r, s, t) + d(r, p, s, t)); // Include a factor 1/2 to accommodate for response density matrices.
167  }
168  }
169  }
170  }
171  } // F_i elements loop
172  F_vector[i] = F_i;
173  }
174 
175  return StorageArray<SquareMatrix<double>, Vectorizer> {F_vector, this->array.vectorizer()};
176  }

190  enable_if_t<std::is_same<Z, double>::value, StorageArray<SquareRankFourTensor<double>, Vectorizer>> calculateSuperFockianMatrix(const Derived1DM& D, const Derived2DM& d) const {
191 
192  if (D.numberOfOrbitals() != this->numberOfOrbitals()) {
193  throw std::invalid_argument("SimpleSQTwoElectronOperator::calculateFockianMatrix(const Derived1DM&, const Derived2DM&): The given 1-DM's dimensions are not compatible with this two-electron operator.");
194  }
195 
196  if (d.numberOfOrbitals() != this->numberOfOrbitals()) {
197  throw std::invalid_argument("SimpleSQTwoElectronOperator::calculateFockianMatrix(const Derived1DM&, const Derived2DM&): The given 2-DM's dimensions are not compatible with this two-electron operator.");
198  }
199 
200 
201  const auto& parameters = this->allParameters(); // The parameters of this two-electron operator, as a vector.
202  std::vector<SquareRankFourTensor<double>> G_vector {this->numberOfComponents()}; // The resulting vector of super-Fockian matrices.
203  const auto F_vector = this->calculateFockianMatrix(D, d).elements(); // The Fockian matrices are necessary in the calculation of the super-Fockian matrices.
204 
205 
206  // A KISS implementation of the calculation of the super-Fockian matrix
207  for (size_t i = 0; i < this->numberOfComponents(); i++) {
208 
209  const auto& g_i = parameters[i]; // The matrix representation of the parameters of the i-th component.
210  const auto& F_i = F_vector[i]; // The Fockian matrix of the i-th component.
211 
212  // Calculate the super-Fockian matrix for every component and add it to the array. Add factors 1/2 to accommodate for response density matrices.
213  SquareRankFourTensor<Scalar> G_i {this->numberOfOrbitals()};
214  G_i.setZero();
215  for (size_t p = 0; p < this->numberOfOrbitals(); p++) {
216  for (size_t q = 0; q < this->numberOfOrbitals(); q++) {
217  for (size_t r = 0; r < this->numberOfOrbitals(); r++) {
218  for (size_t s = 0; s < this->numberOfOrbitals(); s++) {
219 
220  if (q == r) {
221  G_i(p, q, r, s) += F_i(p, s);
222  }
223 
224  for (size_t t = 0; t < this->numberOfOrbitals(); t++) {
225  for (size_t u = 0; u < this->numberOfOrbitals(); u++) {
226  double value = g_i(s, t, q, u) * (d(r, t, p, u) + d(t, r, u, p));
227  value -= g_i(s, t, u, p) * (d(r, t, u, q) + d(t, r, q, u));
228  value -= g_i(s, p, t, u) * (d(r, q, t, u) + d(q, r, u, t));
229 
230  G_i(p, q, r, s) += 0.5 * value;
231  }
232  }
233  }
234  }
235  }
236  } // G_i elements loop
237  G_vector[i] = G_i;
238  }
239 
240  return StorageArray<SquareRankFourTensor<double>, Vectorizer> {G_vector, this->array.vectorizer()};
241  }

247  DerivedSQOneElectronOperator effectiveOneElectronPartition() const {
248 
249  // Prepare some variables.
250  const auto& g = this->allParameters(); // The parameters of this two-electron operator, as a vector.
251 
252  const auto K = this->numberOfOrbitals();
253  DerivedSQOneElectronOperator k_op = DerivedSQOneElectronOperator::Zero(K); // 'op' for operator.
254  auto& k = k_op.allParameters();
255 
256  // A KISS-implementation of the elements of the one-electron partition.
257  for (size_t i = 0; i < this->numberOfComponents(); i++) {
258 
259  for (size_t p = 0; p < K; p++) {
260  for (size_t q = 0; q < K; q++) {
261  for (size_t r = 0; r < K; r++) {
262  k[i](p, q) -= 0.5 * g[i](p, r, r, q);
263  }
264  }
265  }
266  }

283  DerivedOperator transformed(const Transformation& T) const override {
284 
285  // Since we're only getting T as a matrix, we should convert it to an appropriate tensor to perform contractions.
286  // Although not a necessity for the einsum implementation, it makes it a lot easier to follow the formulas.
287  const GQCP::Tensor<Scalar, 2> T_tensor = Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>>(T.matrix().data(), T.matrix().rows(), T.matrix().cols());
288 
289  // We calculate the conjugate as a tensor as well.
290  const GQCP::Tensor<Scalar, 2> T_conjugate = T_tensor.conjugate();
291 
292  // Calculate the basis transformation for every component of the operator.
293  const auto& parameters = this->allParameters();
294  auto result = this->allParameters();
295 
296  for (size_t i = 0; i < this->numberOfComponents(); i++) {
297 
298  // We will have to do four single contractions
299  // g(T U V W) T^*(V R) -> a(T U R W)
300  // a(T U R W) T(W S) -> b(T U R S)
301  const auto temp_1 = parameters[i].template einsum<1>("TUVW,VR->TURW", T_conjugate).template einsum<1>("TURW,WS->TURS", T_tensor);
302 
303  // T(U Q) b(T U R S) -> c(T Q R S)
304  const auto temp_2 = T_tensor.template einsum<1>("UQ,TURS->TQRS", temp_1);
305 
306  // T^*(T P) c(T Q R S) -> g'(P Q R S)
307  const auto transformed_component = T_conjugate.template einsum<1>("TP,TQRS->PQRS", temp_2);
308 
309  result[i] = transformed_component;
310  }
311 
312  return DerivedOperator {StorageArray<MatrixRepresentation, Vectorizer>(result, this->array.vectorizer())};
313  }

366  Self antisymmetrized() const {
367 
368  // Attempt to modify a copy of these integrals if they haven't been antisymmetrized already.
369  const auto& parameters = this->allParameters();
370  auto copy = *this;
371  auto& copy_parameters = copy.allParameters();
372 
373  if (!(this->isAntisymmetrized())) {
374 
375  // Determine the shuffle-indices for the matrix elements that will be subtracted.
376  Eigen::array<int, 4> shuffle_indices;
377  if (this->isExpressedUsingChemistsNotation()) {
378  shuffle_indices = Eigen::array<int, 4> {0, 3, 2, 1};
379  } else { // Expressed using physicist's notation.
380  shuffle_indices = Eigen::array<int, 4> {0, 1, 3, 2};
381  }
382 
383  auto& copy_parameters = copy.allParameters();
384  for (size_t i = 0; i < this->numberOfComponents(); i++) {
385  copy_parameters[i] -= parameters[i].shuffle(shuffle_indices);
386  }
387 
388  copy.is_antisymmetrized = true;
389  }
390 
391  return copy;
392  }
