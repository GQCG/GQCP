94  DoublySpinResolvedBase<PureUSQTwoElectronOperatorComponent<Scalar, Vectorizer>, MixedUSQTwoElectronOperatorComponent<Scalar, Vectorizer>, USQTwoElectronOperator<Scalar, Vectorizer>>(
95  PureUSQTwoElectronOperatorComponent<Scalar, Vectorizer> {StorageArray<SquareRankFourTensor<Scalar>, Vectorizer> {gs_aa, vectorizer}},
96  MixedUSQTwoElectronOperatorComponent<Scalar, Vectorizer> {StorageArray<SquareRankFourTensor<Scalar>, Vectorizer> {gs_ab, vectorizer}},
97  MixedUSQTwoElectronOperatorComponent<Scalar, Vectorizer> {StorageArray<SquareRankFourTensor<Scalar>, Vectorizer> {gs_ba, vectorizer}},
98  PureUSQTwoElectronOperatorComponent<Scalar, Vectorizer> {StorageArray<SquareRankFourTensor<Scalar>, Vectorizer> {gs_bb, vectorizer}}) {
99 
100  // Check if the given tensor representations have the same dimensions, for each spin part.
101  const auto dimension_of_first_aa = gs_aa[0].dimension();
102  const auto dimension_of_first_ab = gs_ab[0].dimension();
103  const auto dimension_of_first_ba = gs_ba[0].dimension();
104  const auto dimension_of_first_bb = gs_bb[0].dimension();
105 
106  for (size_t i = 1; i < N; i++) {
107 
108  const auto dimension_of_ith_aa = gs_aa[i].dimension();
109  const auto dimension_of_ith_ab = gs_ab[i].dimension();
110  const auto dimension_of_ith_ba = gs_ba[i].dimension();
111  const auto dimension_of_ith_bb = gs_bb[i].dimension();
112 
113  if ((dimension_of_first_aa != dimension_of_ith_aa) || (dimension_of_first_ab != dimension_of_ith_ab) || (dimension_of_first_ba != dimension_of_ith_ba) || (dimension_of_first_bb != dimension_of_ith_bb)) {
114  throw std::invalid_argument("USQTwoElectronOperator(const std::array<SquareRankFourTensor<Scalar>, N>&, const std::array<SquareRankFourTensor<Scalar>, N>&, const std::array<SquareRankFourTensor<Scalar>, N>&, const std::array<SquareRankFourTensor<Scalar>, N>&): The given tensor representations do not have the same dimensions for either the alpha, beta or one of the mixed components.");
115  }
116  }

183  StorageArray<Scalar, Vectorizer> calculateExpectationValue(const SpinResolved2DM<Scalar>& d) const {
184 
185  // Calculate the sum of all spin contributions.
186  // Unfortunately, we can't give `StorageArray` out-of-the-box vector-space arithmetic (since the default scalar type for scalar multiplication is unknown), so we'll have to do the summation ourselves.
187  const auto expectation_value_elements_aa = this->alphaAlpha().calculateExpectationValue(d.alphaAlpha()).elements();
188  const auto expectation_value_elements_ab = this->alphaBeta().calculateExpectationValue(d.alphaBeta()).elements();
189  const auto expectation_value_elements_ba = this->betaAlpha().calculateExpectationValue(d.betaAlpha()).elements();
190  const auto expectation_value_elements_bb = this->betaBeta().calculateExpectationValue(d.betaBeta()).elements();
191 
192  auto result_elements = expectation_value_elements_aa;
193  const auto vectorizer = this->alphaAlpha().vectorizer();
194  for (size_t i = 0; i < vectorizer.numberOfElements(); i++) {
195  result_elements[i] += expectation_value_elements_ab[i] + expectation_value_elements_ba[i] + expectation_value_elements_bb[i];
196  }
197 
198  return StorageArray<Scalar, Vectorizer> {result_elements, vectorizer};
199  }

212  size_t numberOfOrbitals(const Spin sigma, const Spin tau) const {
213 
214  if (sigma == Spin::alpha && tau == Spin::beta) {
215  return this->alphaAlpha().numberOfOrbitals();
216  } else if (sigma == Spin::alpha && tau == Spin::beta) {
217  return this->alphaBeta().numberOfOrbitals();
218  } else if (sigma == Spin::beta && tau == Spin::alpha) {
219  return this->betaAlpha().numberOfOrbitals();
220  } else {
221  return this->betaBeta().numberOfOrbitals();
222  }
223  }
