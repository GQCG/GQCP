88 EvaluatedOperator evaluate(const InputType& in) const {
89
90 // Evaluate the operator for every component.
91 std::vector<SquareMatrix<OutputType>> F_evaluated_vector {};
92 const auto& all_parameters = this->allParameters();
93 for (size_t i = 0; i < this->numberOfComponents(); i++) {
94 SquareMatrix<OutputType> F_evaluated = SquareMatrix<OutputType>::Zero(this->numberOfOrbitals());
95
96 // Evaluate the underlying functions for the given argument.
97 for (size_t p = 0; p < this->numberOfOrbitals(); p++) {
98 for (size_t q = 0; q < this->numberOfOrbitals(); q++) {
99 F_evaluated(p, q) = all_parameters[i](p, q).operator()(in); // Evaluate the function at the (p,q)-th element.
100 }
101 }
102
103 F_evaluated_vector.push_back(F_evaluated);
104 }
105
106 const StorageArray<SquareMatrix<OutputType>, Vectorizer> array {F_evaluated_vector, this->vectorizer()};
107 return EvaluatedOperator {array};
108 }

122 StorageArray<EvaluableLinearCombination<OutputType, FunctionType>, Vectorizer> calculateExpectationValue(const OneDM& D) const {
123
124 // The expectation value is a linear combination of this operator's component's 'matrix elements'.
125 const auto K = this->numberOfOrbitals();
126 std::vector<EvaluableLinearCombination<OutputType, FunctionType>> expectation_values;
127 const auto& parameters = this->allParameters();
128 for (size_t i = 0; i < this->numberOfComponents(); i++) {
129
130 EvaluableLinearCombination<OutputType, FunctionType> expectation_value;
131 for (size_t p = 0; p < K; p++) {
132 for (size_t q = 0; q < K; q++) {
133 const auto coefficient = D.matrix()(p, q);
134 const auto function = parameters[i](p, q);
135 expectation_value.append(coefficient, function);
136 }
137 }
138
139 expectation_values.push_back(expectation_value);
140 }
141
142 return StorageArray<EvaluableLinearCombination<OutputType, FunctionType>, Vectorizer> {expectation_values, this->vectorizer()};
143 }
