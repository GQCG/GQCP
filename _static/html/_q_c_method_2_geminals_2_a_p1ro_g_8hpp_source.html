71 template <typename Solver>
72 QCStructure<GQCP::QCModel::AP1roG> optimize(Solver& solver, NonLinearEquationEnvironment<double>& environment) const {
73
74 // The AP1roG method's responsibility is to try to optimize the parameters of its method, given a solver and associated environment.
75 solver.perform(environment);
76
77 // To make a QCStructure, we need the electronic energy, geminal coefficients and number of electrons.
78 // Furthermore, the solvers only find the ground state wave function parameters, so the QCStructure only needs to contain the parameters for one state.
79 const auto G_optimal = AP1roGGeminalCoefficients::FromColumnMajor(environment.variables.back(), this->N_P, this->K);
80 const GQCP::QCModel::AP1roG ap1rog_parameters {G_optimal};
81
82 const auto E_electronic = ap1rog_parameters.calculateEnergy(sq_hamiltonian);
83
84 return QCStructure<GQCP::QCModel::AP1roG>({E_electronic}, {ap1rog_parameters});
85 }
