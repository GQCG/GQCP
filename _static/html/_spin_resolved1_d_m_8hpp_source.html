96  G1DM<Scalar> generalized() const {
97 
98  // Determine the dimensions of the generalized, spin-blocked 1-DM.
99  const auto K_alpha = this->alpha().numberOfOrbitals();
100  const auto K_beta = this->beta().numberOfOrbitals();
101  const auto M = K_alpha + K_beta;
102 
103  // The generalized 1-DM contains the alpha part in the top-left corner, and the beta part in the bottom right corner.
104  G1DM<Scalar> D_generalized = G1DM<Scalar>::Zero(M);
105  D_generalized.topLeftCorner(K_alpha, K_alpha) = this->alpha();
106  D_generalized.bottomRightCorner(K_beta, K_beta) = this->beta();
107 
108  return D_generalized;
109  }
