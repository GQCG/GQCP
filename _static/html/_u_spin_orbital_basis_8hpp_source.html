306 USQOneElectronOperator<product_t<ElectronicSpin_zOperator::Scalar, ExpansionScalar>, ElectronicSpin_zOperator::Vectorizer> quantize(const ElectronicSpin_zOperator& fq_op) const {
307
308 using ResultScalar = product_t<ElectronicSpin_zOperator::Scalar, ExpansionScalar>;
309 using ResultOperator = USQOneElectronOperator<ResultScalar, ElectronicSpin_zOperator::Vectorizer>;
310
311 // We can use the quantization of the overlap operator for the quantization of S_z.
312 const auto S_a = this->alpha().overlap(); // In the current orbital basis.
313 const auto S_b = this->beta().overlap(); // In the current orbital basis.
314
315 const auto S_z_a = 0.5 * S_a;
316 const auto S_z_b = -0.5 * S_b;
317
318 return ResultOperator {S_z_a, S_z_b};
319 }

329 template <typename Z = Shell>
330 auto quantize(const SpinZeemanOperator& fq_op) const -> enable_if_t<std::is_same<Z, LondonGTOShell>::value, USQOneElectronOperator<product_t<SpinZeemanOperator::Scalar, ExpansionScalar>, SpinZeemanOperator::Vectorizer>> {
331
332 const auto& B = fq_op.magneticField().strength();
333 if ((std::abs(B(CartesianDirection::x)) > 1.0e-12) || (std::abs(B(CartesianDirection::y)) > 1.0e-12)) {
334 throw std::invalid_argument("USpinOrbitalBasis::quantize(const FQMolecularPauliHamiltonian&): Only the spin Zeeman operator for a magnetic field along the z-axis can be quantized in an USpinOrbitalBasis.");
335 }
336
337 return complex(B(CartesianDirection::z)) * this->quantize(ElectronicSpin_zOperator());
338 }

356 template <typename Z = Shell>
357 auto quantize(const CoulombRepulsionOperator& coulomb_op) const -> enable_if_t<std::is_same<Z, GTOShell>::value, ScalarUSQTwoElectronOperator<product_t<typename CoulombRepulsionOperator::Scalar, ExpansionScalar>>> {
358
359 using ResultScalar = product_t<typename CoulombRepulsionOperator::Scalar, ExpansionScalar>;
360 using ResultOperator = ScalarUSQTwoElectronOperator<ResultScalar>;
361
362 // Determine the matrix representation of the four spin-components of the second-quantized Coulomb operator.
363 const auto g_aa_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->alpha().scalarBasis(), this->alpha().scalarBasis()); // 'par' for 'parameters'
364 const auto g_ab_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->alpha().scalarBasis(), this->beta().scalarBasis()); // 'par' for 'parameters'
365 const auto g_ba_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->beta().scalarBasis(), this->alpha().scalarBasis()); // 'par' for 'parameters'
366 const auto g_bb_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->beta().scalarBasis(), this->beta().scalarBasis()); // 'par' for 'parameters'
367
368 auto g_aa = SquareRankFourTensor<ResultScalar>::Zero(g_aa_par.dimension(0));
369 auto g_ab = SquareRankFourTensor<ResultScalar>::Zero(g_ab_par.dimension(0));
370 auto g_ba = SquareRankFourTensor<ResultScalar>::Zero(g_ba_par.dimension(0));
371 auto g_bb = SquareRankFourTensor<ResultScalar>::Zero(g_bb_par.dimension(0));
372
373 for (size_t i = 0; i < g_aa_par.dimension(0); i++) {
374 for (size_t j = 0; j < g_aa_par.dimension(1); j++) {
375 for (size_t k = 0; k < g_aa_par.dimension(2); k++) {
376 for (size_t l = 0; l < g_aa_par.dimension(3); l++) {
377 g_aa(i, j, k, l) = g_aa_par(i, j, k, l);
378 g_ab(i, j, k, l) = g_ab_par(i, j, k, l);
379 g_ba(i, j, k, l) = g_ba_par(i, j, k, l);
380 g_bb(i, j, k, l) = g_bb_par(i, j, k, l);
381 }
382 }
383 }
384 }
385
386 // We have previously calculated the representations in the AO basis, so we'll still have to transform these representations to the current spin-orbitals.
387 ResultOperator g {g_aa, g_ab, g_ba, g_bb};
388 g.transform(this->expansion()); // Now, g is expressed in the current spin-orbital basis.
389
390 return g;
391 }

463
464 return USQHamiltonian<ExpansionScalar> {{T, OZ, D, V}, {g}};

506 auto quantize(const FQOneElectronOperator& fq_op) const -> USQOneElectronOperator<product_t<typename FQOneElectronOperator::Scalar, ExpansionScalar>, typename FQOneElectronOperator::Vectorizer> {
507
508 using ResultScalar = product_t<typename FQOneElectronOperator::Scalar, ExpansionScalar>;
509 using ResultOperator = USQOneElectronOperator<ResultScalar, typename FQOneElectronOperator::Vectorizer>;
510
511 // Quantize the one-electron operator in the alpha- and beta- bases and return the wrapped result.
512 const auto f_a = this->alpha().quantize(fq_op);
513 const auto f_b = this->beta().quantize(fq_op);
514
515 return ResultOperator {f_a, f_b};
