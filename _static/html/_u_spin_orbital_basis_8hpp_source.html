291  auto quantize(const FQOneElectronOperator& fq_op) const -> enable_if_t<std::is_same<Z, GTOShell>::value, USQOneElectronOperator<product_t<typename FQOneElectronOperator::Scalar, ExpansionScalar>, typename FQOneElectronOperator::Vectorizer>> {

313  template <typename Z = Shell>
314  auto quantize(const ElectronicSpin_zOperator& fq_op) const -> enable_if_t<std::is_same<Z, GTOShell>::value, USQOneElectronOperator<product_t<ElectronicSpin_zOperator::Scalar, ExpansionScalar>, ElectronicSpin_zOperator::Vectorizer>> {
315 
316  using ResultScalar = product_t<ElectronicSpin_zOperator::Scalar, ExpansionScalar>;
317  using ResultOperator = USQOneElectronOperator<ResultScalar, ElectronicSpin_zOperator::Vectorizer>;
318 
319  // We can use the quantization of the overlap operator for the quantization of S_z.
320  const auto S_a = this->alpha().overlap(); // In the current orbital basis.
321  const auto S_b = this->beta().overlap(); // In the current orbital basis.
322 
323  const auto S_z_a = 0.5 * S_a;
324  const auto S_z_b = -0.5 * S_b;
325 
326  return ResultOperator {S_z_a, S_z_b};
327  }

345  template <typename Z = Shell>
346  auto quantize(const CoulombRepulsionOperator& coulomb_op) const -> enable_if_t<std::is_same<Z, GTOShell>::value, ScalarUSQTwoElectronOperator<product_t<typename CoulombRepulsionOperator::Scalar, ExpansionScalar>>> {
347 
348  using ResultScalar = product_t<typename CoulombRepulsionOperator::Scalar, ExpansionScalar>;
349  using ResultOperator = ScalarUSQTwoElectronOperator<ResultScalar>;
350 
351  // Determine the matrix representation of the four spin-components of the second-quantized Coulomb operator.
352  const auto g_aa_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->alpha().scalarBasis(), this->alpha().scalarBasis()); // 'par' for 'parameters'
353  const auto g_ab_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->alpha().scalarBasis(), this->beta().scalarBasis()); // 'par' for 'parameters'
354  const auto g_ba_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->beta().scalarBasis(), this->alpha().scalarBasis()); // 'par' for 'parameters'
355  const auto g_bb_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->beta().scalarBasis(), this->beta().scalarBasis()); // 'par' for 'parameters'
356 
357  auto g_aa = SquareRankFourTensor<ResultScalar>::Zero(g_aa_par.dimension(0));
358  auto g_ab = SquareRankFourTensor<ResultScalar>::Zero(g_ab_par.dimension(0));
359  auto g_ba = SquareRankFourTensor<ResultScalar>::Zero(g_ba_par.dimension(0));
360  auto g_bb = SquareRankFourTensor<ResultScalar>::Zero(g_bb_par.dimension(0));
361 
362  for (size_t i = 0; i < g_aa_par.dimension(0); i++) {
363  for (size_t j = 0; j < g_aa_par.dimension(1); j++) {
364  for (size_t k = 0; k < g_aa_par.dimension(2); k++) {
365  for (size_t l = 0; l < g_aa_par.dimension(3); l++) {
366  g_aa(i, j, k, l) = g_aa_par(i, j, k, l);
367  g_ab(i, j, k, l) = g_ab_par(i, j, k, l);
368  g_ba(i, j, k, l) = g_ba_par(i, j, k, l);
369  g_bb(i, j, k, l) = g_bb_par(i, j, k, l);
370  }
371  }
372  }
373  }
374 
375  // We have previously calculated the representations in the AO basis, so we'll still have to transform these representations to the current spin-orbitals.
376  ResultOperator g {g_aa, g_ab, g_ba, g_bb};
377  g.transform(this->expansion()); // Now, g is expressed in the current spin-orbital basis.
378 
379  return g;
380  }
