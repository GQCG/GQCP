304  auto quantize(const ElectronicSpin_zOperator& fq_op) const -> USQOneElectronOperator<product_t<ElectronicSpin_zOperator::Scalar, ExpansionScalar>, ElectronicSpin_zOperator::Vectorizer> {
305 
306  using ResultScalar = product_t<ElectronicSpin_zOperator::Scalar, ExpansionScalar>;
307  using ResultOperator = USQOneElectronOperator<ResultScalar, ElectronicSpin_zOperator::Vectorizer>;
308 
309  // We can use the quantization of the overlap operator for the quantization of S_z.
310  const auto S_a = this->alpha().overlap(); // In the current orbital basis.
311  const auto S_b = this->beta().overlap(); // In the current orbital basis.
312 
313  const auto S_z_a = 0.5 * S_a;
314  const auto S_z_b = -0.5 * S_b;
315 
316  return ResultOperator {S_z_a, S_z_b};
317  }

335  auto quantize(const CoulombRepulsionOperator& coulomb_op) const -> ScalarUSQTwoElectronOperator<product_t<typename CoulombRepulsionOperator::Scalar, ExpansionScalar>> {
336 
337  using ResultScalar = product_t<typename CoulombRepulsionOperator::Scalar, ExpansionScalar>;
338  using ResultOperator = ScalarUSQTwoElectronOperator<ResultScalar>;
339 
340  // Determine the matrix representation of the four spin-components of the second-quantized Coulomb operator.
341  const auto g_aa_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->alpha().scalarBasis(), this->alpha().scalarBasis());
342  const auto g_ab_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->alpha().scalarBasis(), this->beta().scalarBasis());
343  const auto g_ba_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->beta().scalarBasis(), this->alpha().scalarBasis());
344  const auto g_bb_par = IntegralCalculator::calculateLibintIntegrals(coulomb_op, this->beta().scalarBasis(), this->beta().scalarBasis());
345 
346  // We have previously calculated the representations in the AO basis, so we'll still have to transform these representations to the current spin-orbitals.
347  ResultOperator g {g_aa_par, g_ab_par, g_ba_par, g_bb_par};
348  g.transform(this->expansion()); // Now, g is expressed in the current spin-orbital basis.
349 
350  return g;
351  }
