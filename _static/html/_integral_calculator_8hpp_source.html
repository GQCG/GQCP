61  template <typename Shell, size_t N, typename IntegralScalar>
62  static auto calculate(BaseOneElectronIntegralEngine<Shell, N, IntegralScalar>& engine, const ShellSet<Shell>& left_shell_set, const ShellSet<Shell>& right_shell_set) -> std::array<MatrixX<IntegralScalar>, N> {
63 
64  // Initialize the N components of the matrix representations of the operator.
65  const auto nbf_left = left_shell_set.numberOfBasisFunctions();
66  const auto nbf_right = right_shell_set.numberOfBasisFunctions();
67 
68  std::array<MatrixX<IntegralScalar>, N> components;
69  for (auto& component : components) {
70  component = MatrixX<IntegralScalar>::Zero(nbf_left, nbf_right);
71  }
72 
73 
74  // Loop over all left and right shells and let the engine calculate the integrals over the pairs of shells.
75  const auto nsh_left = left_shell_set.numberOfShells();
76  const auto left_shells = left_shell_set.asVector();
77  const auto nsh_right = right_shell_set.numberOfShells();
78  const auto right_shells = right_shell_set.asVector();
79 
80  for (size_t left_shell_index = 0; left_shell_index < nsh_left; left_shell_index++) {
81  const auto left_bf_index = left_shell_set.basisFunctionIndex(left_shell_index);
82  const auto left_shell = left_shells[left_shell_index];
83 
84  for (size_t right_shell_index = 0; right_shell_index < nsh_right; right_shell_index++) {
85  const auto right_bf_index = right_shell_set.basisFunctionIndex(right_shell_index);
86  const auto right_shell = right_shells[right_shell_index];
87 
88  const auto buffer = engine.calculate(left_shell, right_shell);
89 
90  // Only if the integrals are not all zero, place them inside the full matrices.
91  if (buffer->areIntegralsAllZero()) {
92  continue;
93  }
94  buffer->emplace(components, left_bf_index, right_bf_index);
95  } // right shells loop
96  } // left shells loop
97 
98  return components;
99  }

113  template <typename Shell, size_t N, typename IntegralScalar>
114  static auto calculate(BaseTwoElectronIntegralEngine<Shell, N, IntegralScalar>& engine, const ShellSet<Shell>& left_shell_set, const ShellSet<Shell>& right_shell_set) -> std::array<Tensor<IntegralScalar, 4>, N> {
115 
116  // Initialize the N components of the matrix representations of the operator.
117  const auto nbf_left = left_shell_set.numberOfBasisFunctions();
118  const auto nbf_right = right_shell_set.numberOfBasisFunctions();
119 
120  std::array<Tensor<IntegralScalar, 4>, N> components;
121  for (auto& component : components) {
122  component = Tensor<IntegralScalar, 4>(nbf_left, nbf_left, nbf_right, nbf_right);
123  component.setZero();
124  }
125 
126 
127  // Loop over all left and right shells and let the engine calculate the integrals over the 4-tuple of shells.
128  const auto nsh_left = left_shell_set.numberOfShells();
129  const auto left_shells = left_shell_set.asVector();
130  const auto nsh_right = right_shell_set.numberOfShells();
131  const auto right_shells = right_shell_set.asVector();
132 
133  for (size_t left_shell_index1 = 0; left_shell_index1 < nsh_left; left_shell_index1++) {
134  const auto left_bf1_index = left_shell_set.basisFunctionIndex(left_shell_index1);
135  const auto left_shell1 = left_shells[left_shell_index1];
136 
137  for (size_t left_shell_index2 = 0; left_shell_index2 < nsh_left; left_shell_index2++) {
138  const auto left_bf2_index = left_shell_set.basisFunctionIndex(left_shell_index2);
139  const auto left_shell2 = left_shells[left_shell_index2];
140 
141  for (size_t right_shell_index1 = 0; right_shell_index1 < nsh_right; right_shell_index1++) {
142  const auto right_bf1_index = right_shell_set.basisFunctionIndex(right_shell_index1);
143  const auto right_shell1 = right_shells[right_shell_index1];
144 
145  for (size_t right_shell_index2 = 0; right_shell_index2 < nsh_right; right_shell_index2++) {
146  const auto right_bf2_index = right_shell_set.basisFunctionIndex(right_shell_index2);
147  const auto right_shell2 = right_shells[right_shell_index2];
148 
149  const auto buffer = engine.calculate(left_shell1, left_shell2, right_shell1, right_shell2);
150 
151  // Only if the integrals are not all zero, place them inside the full matrices
152  if (buffer->areIntegralsAllZero()) {
153  continue;
154  }
155  buffer->emplace(components, left_bf1_index, left_bf2_index, right_bf1_index, right_bf2_index); // place the calculated integrals inside the full tensors
156 
157  } // sh4_index
158  } // right_shell_index1
159  } // left_shell_index2
160  } // left_shell_index1
161 
162  return components;
163  }
