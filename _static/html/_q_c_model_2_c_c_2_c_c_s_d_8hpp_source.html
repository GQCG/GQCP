77 static Scalar calculateCorrelationEnergy(const SquareMatrix<Scalar>& f, const SquareRankFourTensor<Scalar>& V_A, const T1Amplitudes<Scalar>& t1, const T2Amplitudes<Scalar>& t2) {
78 const auto orbital_space = t1.orbitalSpace(); // assume t1 and t2 have the same orbital space.
79
80 // A KISS implementation of the CCSD energy correction.
81 // The implementation is in line with Crawford2000 "Chapter 2: An Introduction to Coupled Cluster Theory for Computational Chemists", eq. [134].
82 Scalar E {0.0};
83 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
84 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
85 E += f(i, a) * t1(i, a);
86 }
87 }
88
89 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
90 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
91 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
92 for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
93 E += 0.25 * V_A(i, j, a, b) * t2(i, j, a, b);
94 E += 0.5 * V_A(i, j, a, b) * t1(i, a) * t1(j, b);
95 }
96 }
97 }
98 }
99
100 return E;
101 }

120 static Scalar calculateT1AmplitudeEquation(const size_t i, const size_t a, const SquareMatrix<Scalar>& f, const SquareRankFourTensor<Scalar>& V_A, const T1Amplitudes<Scalar>& t1, const T2Amplitudes<Scalar>& t2, const ImplicitMatrixSlice<Scalar>& F1, const ImplicitMatrixSlice<Scalar>& F2, const ImplicitMatrixSlice<Scalar>& F3) {
121
122 const auto orbital_space = t1.orbitalSpace(); // assume t1 and t2 have the same orbital space.
123
124 // We will use equation (1) in Stanton1991 by putting the left-hand term (with the energy denominator) to the right.
125 Scalar result {0.0}; // zero-initialize the scalar value for the result
126
127 // Calculate the contribution from the left-hand side.
128 const auto D_ia = f(i, i) - f(a, a);
129 result -= t1(i, a) * D_ia;
130
131 // Calculate the contribution from the first term.
132 result += f(i, a);
133
134 // Calculate the contribution from the second term.
135 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
136 result += t1(i, e) * F1(a, e);
137 }
138
139 // Calculate the contribution from the third term.
140 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
141 result -= t1(m, a) * F2(m, i);
142 }
143
144 // Calculate the contribution from the fourth term.
145 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
146 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
147 result += t2(i, m, a, e) * F3(m, e);
148 }
149 }
150
151 // Calculate the contribution from the fifth term.
152 for (const auto& n : orbital_space.indices(OccupationType::k_occupied)) {
153 for (const auto& f : orbital_space.indices(OccupationType::k_virtual)) {
154 result -= t1(n, f) * V_A(n, a, i, f);
155 }
156 }
157
158 // Calculate the contribution from the sixth term.
159 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
160 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
161 for (const auto& f : orbital_space.indices(OccupationType::k_virtual)) {
162 result -= 0.5 * t2(i, m, e, f) * V_A(m, a, e, f);
163 }
164 }
165 }
166
167 // Calculate the contribution from the seventh term.
168 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
169 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
170 for (const auto& n : orbital_space.indices(OccupationType::k_occupied)) {
171 result -= 0.5 * t2(m, n, a, e) * V_A(n, m, e, i);
172 }
173 }
174 }
175
176 return result;
177 }

202 static Scalar calculateT2AmplitudeEquation(const size_t i, const size_t j, const size_t a, const size_t b, const SquareMatrix<Scalar>& f, const SquareRankFourTensor<Scalar>& V_A, const T1Amplitudes<Scalar>& t1, const T2Amplitudes<Scalar>& t2, const ImplicitRankFourTensorSlice<Scalar>& tau2, const ImplicitMatrixSlice<Scalar>& F1, const ImplicitMatrixSlice<Scalar>& F2, const ImplicitMatrixSlice<Scalar>& F3, const ImplicitRankFourTensorSlice<Scalar>& W1, const ImplicitRankFourTensorSlice<Scalar>& W2, const ImplicitRankFourTensorSlice<Scalar>& W3) {
203
204 const auto orbital_space = t1.orbitalSpace(); // assume t1 and t2 have the same orbital space.
205
206 // We will use equation (2) in Stanton1991 by putting the left-hand term (with the energy denominator) to the right.
207 Scalar result {0.0}; // zero-initialize the scalar value for the result
208
209 // Calculate the contribution from the left-hand side.
210 const auto D_ijab = f(i, i) + f(j, j) - f(a, a) - f(b, b);
211 result -= t2(i, j, a, b) * D_ijab;
212
213 // Calculate the contribution from the first term.
214 result += V_A(i, j, a, b);
215
216 // Calculate the contribution from the second term.
217 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
218 result += t2(i, j, a, e) * F1(b, e);
219 result -= t2(i, j, b, e) * F1(a, e); // P(ab) applied
220
221 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
222 result -= 0.5 * t2(i, j, a, e) * t1(m, b) * F3(m, e);
223 result += 0.5 * t2(i, j, b, e) * t1(m, a) * F3(m, e); // P(ab) applied
224 }
225 }
226
227 // Calculate the contribution from the third term.
228 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
229 result -= t2(i, m, a, b) * F2(m, j);
230 result += t2(j, m, a, b) * F2(m, i); // P(ij) applied
231
232 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
233 result -= 0.5 * t2(i, m, a, b) * t1(j, e) * F3(m, e);
234 result += 0.5 * t2(j, m, a, b) * t1(i, e) * F3(m, e); // P(ij) applied
235 }
236 }
237
238 // Calculate the contribution from the fourth term.
239 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
240 for (const auto& n : orbital_space.indices(OccupationType::k_occupied)) {
241 result += 0.5 * tau2(m, n, a, b) * W1(m, n, i, j);
242 }
243 }
244
245 // Calculate the contribution from the fifth term.
246 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
247 for (const auto& f : orbital_space.indices(OccupationType::k_virtual)) {
248 result += 0.5 * tau2(i, j, e, f) * W2(a, b, e, f);
249 }
250 }
251
252 // Calculate the contribution from the sixth term.
253 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
254 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
255 result += t2(i, m, a, e) * W3(m, b, e, j) - t1(i, e) * t1(m, a) * V_A(m, b, e, j);
256 result -= t2(j, m, a, e) * W3(m, b, e, i) - t1(j, e) * t1(m, a) * V_A(m, b, e, i); // P(ij) applied
257 result -= t2(i, m, b, e) * W3(m, a, e, j) - t1(i, e) * t1(m, b) * V_A(m, a, e, j); // P(ab) applied
258 result += t2(j, m, b, e) * W3(m, a, e, i) - t1(j, e) * t1(m, b) * V_A(m, a, e, i); // P(ij) P(ab) applied
259 }
260 }
261
262 // Calculate the contribution from the seventh term.
263 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
264 result += t1(i, e) * V_A(a, b, e, j);
265 result -= t1(j, e) * V_A(a, b, e, i); // P(ij) applied
266 }
267
268 // Calculate the contribution from the eight term.
269 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
270 result -= t1(m, a) * V_A(m, b, i, j);
271 result += t1(m, b) * V_A(m, a, i, j); // P(ab) applied
272 }
273
274 return result;
275 }

288 static ImplicitMatrixSlice<Scalar> calculateF1(const SquareMatrix<Scalar>& f, const SquareRankFourTensor<Scalar>& V_A, const T1Amplitudes<Scalar>& t1, const ImplicitRankFourTensorSlice<Scalar>& tau2_tilde) {
289
290 const auto& orbital_space = t1.orbitalSpace();
291
292 // Implement the formula for the F1-intermediate: equation (3) in Stanton1993.
293 auto F1 = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_virtual, OccupationType::k_virtual); // zero-initialize a virtual-virtual object
294 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
295 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
296 Scalar value {0.0}; // zero-initialize the scalar value to be added
297
298 // Calculate the contribution from the first term.
299 if (a != e) { // (1 - delta_ae)
300 value += f(a, e);
301 }
302
303 // Calculate the contribution from the second term.
304 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
305 value -= 0.5 * f(m, e) * t1(m, a);
306 }
307
308 // Calculate the contribution from the third term.
309 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
310 for (const auto f : orbital_space.indices(OccupationType::k_virtual)) {
311 value += t1(m, f) * V_A(m, a, f, e);
312 }
313 }
314
315 // Calculate the contribution from the fourth term.
316 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
317 for (const auto& n : orbital_space.indices(OccupationType::k_occupied)) {
318 for (const auto& f : orbital_space.indices(OccupationType::k_virtual)) {
319 value -= 0.5 * tau2_tilde(m, n, a, f) * V_A(m, n, e, f);
320 }
321 }
322 }
323
324 F1(a, e) = value;
325 }
326 }
327
328 return F1;
329 }

342 static ImplicitMatrixSlice<Scalar> calculateF2(const SquareMatrix<Scalar>& f, const SquareRankFourTensor<Scalar>& V_A, const T1Amplitudes<Scalar>& t1, const ImplicitRankFourTensorSlice<Scalar>& tau2_tilde) {
343
344 const auto& orbital_space = t1.orbitalSpace();
345
346 // Implement the formula for F2 in equation (4) in Stanton1991.
347 auto F2 = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_occupied, OccupationType::k_occupied); // zero-initialize an occupied-occupied object
348 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
349 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
350 Scalar value {0.0}; // zero-initialize the scalar value to be added
351
352 // Calculate the contribution from the first term.
353 if (m != i) { // (1 - delta_mi)
354 value += f(m, i);
355 }
356
357 // Calculate the contribution from the second term.
358 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
359 value += 0.5 * t1(i, e) * f(m, e);
360 }
361
362 // Calculate the contribution from the third term.
363 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
364 for (const auto& n : orbital_space.indices(OccupationType::k_occupied)) {
365 value += t1(n, e) * V_A(m, n, i, e);
366 }
367 }
368
369 // Calculate the contribution from the fourth term.
370 for (const auto& n : orbital_space.indices(OccupationType::k_occupied)) {
371 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
372 for (const auto& f : orbital_space.indices(OccupationType::k_virtual)) {
373 value += 0.5 * tau2_tilde(i, n, e, f) * V_A(m, n, e, f);
374 }
375 }
376 }
377
378 F2(m, i) = value;
379 }
380 }

431 static ImplicitRankFourTensorSlice<Scalar> calculateTau2(const T1Amplitudes<Scalar>& t1, const T2Amplitudes<Scalar>& t2) {
432
433 const auto& orbital_space = t1.orbitalSpace(); // assume the orbital spaces for t1 and t2 are equal
434
435 // Implement the formula for tau2 (equation 10).
436 auto tau2 = t2.asImplicitRankFourTensorSlice(); // the contribution from the first term
437
438 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
439 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
440 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
441 for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
442 tau2(i, j, a, b) += t1(i, a) * t1(j, b) - t1(i, b) * t1(j, a);
443 }
444 }
445 }
446 }
447
448 return tau2;
449 }

461
462 const auto& orbital_space = t1.orbitalSpace(); // assume the orbital spaces for t1 and t2 are equal
463
464 // Implement the formula for tau2_tilde (equation 9).
465 auto tau2_tilde = t2.asImplicitRankFourTensorSlice(); // the contribution from the first term
466
467 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
468 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
469 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
470 for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
471 tau2_tilde(i, j, a, b) += 0.5 * (t1(i, a) * t1(j, b) - t1(i, b) * t1(j, a));
472 }
473 }
474 }
475 }

477 return tau2_tilde;

492 const auto& orbital_space = t1.orbitalSpace();
493
494 // Implement the formula for W1 (equation 6).
495 auto W1 = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_occupied, OccupationType::k_occupied, OccupationType::k_occupied, OccupationType::k_occupied); // zero-initialize an occupied-occupied-occupied-occupied object
496 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
497 for (const auto& n : orbital_space.indices(OccupationType::k_occupied)) {
498 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
499 for (const auto j : orbital_space.indices(OccupationType::k_occupied)) {
500 Scalar value {0.0}; // zero-initialize the scalar value to be added
501
502 // Calculate the contribution from the first term.
503 value += V_A(m, n, i, j);
504
505 // Calculate the contribution from the second term.
506 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
507 value += t1(j, e) * V_A(m, n, i, e);
508 value -= t1(i, e) * V_A(m, n, j, e); // P(ij) applied
509 }
510
511 // Calculate the contribution from the third term.
512 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
513 for (const auto& f : orbital_space.indices(OccupationType::k_virtual)) {
514 value += 0.25 * tau2(i, j, e, f) * V_A(m, n, e, f);

519 }
520 }
521 }

584 static ImplicitRankFourTensorSlice<Scalar> calculateW3(const SquareRankFourTensor<Scalar>& V_A, const T1Amplitudes<Scalar>& t1, const T2Amplitudes<Scalar>& t2) {
585
586 const auto& orbital_space = t1.orbitalSpace(); // assume the orbital spaces for t1 and t2 are equal
587
588 // Implement the formula for W3 (equation 8).
589 auto W3 = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_virtual, OccupationType::k_occupied); // zero-initialize an occupied-virtual-virtual-occupied object
590 for (const auto& m : orbital_space.indices(OccupationType::k_occupied)) {
591 for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
592 for (const auto& e : orbital_space.indices(OccupationType::k_virtual)) {
593 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
594 Scalar value {0.0}; // zero-initialize the scalar value to be added
595
596 // Calculate the contribution from the first term.
597 value += V_A(m, b, e, j);
598
599 // Calculate the contribution from the second term.
600 for (const auto& f : orbital_space.indices(OccupationType::k_virtual)) {
601 value += t1(j, f) * V_A(m, b, e, f);
602 }
603
604 // Calculate the contribution from the third term.
605 for (const auto& n : orbital_space.indices(OccupationType::k_occupied)) {
606 value -= t1(n, b) * V_A(m, n, e, j);
607 }
608
609 // Calculate the contribution from the fourth term.
610 for (const auto& n : orbital_space.indices(OccupationType::k_occupied)) {
611 for (const auto& f : orbital_space.indices(OccupationType::k_virtual)) {
612 value -= (0.5 * t2(j, n, f, b) + t1(j, f) * t1(n, b)) * V_A(m, n, e, f);
613 }
614 }
615
616 W3(m, b, e, j) = value;
617 }
618 }
619 }
620 }
