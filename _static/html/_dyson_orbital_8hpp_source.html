83  static DysonOrbital<Scalar> TransitionAmplitudes(const LinearExpansion<SpinResolvedONVBasis>& linear_expansion_J, const LinearExpansion<SpinResolvedONVBasis>& linear_expansion_I) {
84 
85  const auto onv_basis_J = linear_expansion_J.onvBasis();
86  const auto onv_basis_I = linear_expansion_I.onvBasis();
87 
88  if ((onv_basis_J.alpha().numberOfElectrons() - onv_basis_I.alpha().numberOfElectrons() != 0) && (onv_basis_J.beta().numberOfElectrons() - onv_basis_I.beta().numberOfElectrons() != 1)) {
89  if ((onv_basis_J.alpha().numberOfElectrons() - onv_basis_I.alpha().numberOfElectrons() != 1) && (onv_basis_J.beta().numberOfElectrons() - onv_basis_I.beta().numberOfElectrons() != 0)) {
90  throw std::runtime_error("DysonOrbital::TransitionAmplitudes(LinearExpansion, LinearExpansion): linear_expansion_I is not expressed in a spin-resolved ONV basis with one fewer electron than linear_expansion_J.");
91  }
92  }
93 
94  // Initialize environment variables.
95 
96  // The 'passive' ONV basis is the ONV basis that is equal for both wave functions.
97  // The 'target' ONV basis has an electron difference of one.
98  // We initialize the variables for the case in which they differ in one beta electron, if this isn't the case, we will update it later.
99  auto passive_onv_basis_J = onv_basis_J.alpha();
100  auto passive_onv_basis_I = onv_basis_I.alpha();
101  auto target_onv_basis_J = onv_basis_J.beta();
102  auto target_onv_basis_I = onv_basis_I.beta();
103 
104  // Mod variables relate to the modification of the address jump in coefficient index according to the ordering of the spin ONVs.
105  size_t passive_mod_J = target_onv_basis_J.dimension();
106  size_t passive_mod_I = target_onv_basis_I.dimension();
107  size_t target_mod = 1;
108 
109  // If instead the ONV bases differ by one alpha electron we re-assign the variables to match the algorithm.
110  if ((onv_basis_J.alpha().numberOfElectrons() - onv_basis_I.alpha().numberOfElectrons() == 1) && (onv_basis_J.beta().numberOfElectrons() - onv_basis_I.beta().numberOfElectrons() == 0)) {
111  passive_onv_basis_J = target_onv_basis_J;
112  passive_onv_basis_I = target_onv_basis_I;
113  target_onv_basis_J = onv_basis_J.alpha();
114  target_onv_basis_I = onv_basis_I.alpha();
115 
116  passive_mod_J = 1;
117  passive_mod_I = 1;
118  target_mod = passive_onv_basis_J.dimension();
119  }
120 
121  const auto& ci_coeffs_J = linear_expansion_J.coefficients();
122  const auto& ci_coeffs_I = linear_expansion_I.coefficients();
123 
124  VectorX<double> dyson_coeffs = VectorX<double>::Zero(onv_basis_J.alpha().numberOfOrbitals());
125 
126  // The actual algorithm to determine the Dyson amplitudes.
127 
128  // Since we want to calculate the overlap between two wave functions, the ONVs should have an equal number of electrons.
129  // We therefore iterate over the ONVs of the 'target' ONV basis, which all have an electron more, and annihilate in one of the orbitals (let the index of that orbital be called 'p').
130  // By calculating the overlap in the (N-1)-electron ONV basis, we can calculate the contributions to the 'p'-th coefficient (i.e. the Dyson amplitude) of the Dyson orbital.
131  SpinUnresolvedONV onv = target_onv_basis_J.constructONVFromAddress(0);
132 
133  for (size_t Jt = 0; Jt < target_onv_basis_J.dimension(); Jt++) { // Jt loops over addresses of the target ONV basis.
134  int sign = -1; // Total phase factor of all the annihilations that have occurred.
135  for (size_t e = 0; e < target_onv_basis_J.numberOfElectrons(); e++) { // Loop over electrons in the ONV.
136 
137  // Annihilate on the corresponding orbital, to make sure we can calculate overlaps in the (N-1)-'target' ONV basis.
138  sign *= -1;
139  size_t p = onv.occupationIndexOf(e);
140  onv.annihilate(p);
141 
142  // Now, we calculate the overlap in the (N-1)-electron 'target' ONV basis.
143  // In order to access the correct coefficients for the, we need the address of the resulting (annihilated) ONV inside the 'target' (N-1)-electron ONV basis.
144  size_t address = target_onv_basis_I.addressOf(onv.unsignedRepresentation());
145 
146  double coeff = 0;
147  for (size_t Jp = 0; Jp < passive_onv_basis_J.dimension(); Jp++) { // Jp loops over the addresses of the passive ONV basis.
148  coeff += sign * ci_coeffs_J(Jt * target_mod + Jp * passive_mod_J) * ci_coeffs_I(address * target_mod + Jp * passive_mod_I); // Access the indices of the coefficient vectors.
149  }
150  dyson_coeffs(p) += coeff;
151  onv.create(p); // Allow the iteration to continue with the original ONV.
152  }
153 
154  if (Jt < target_onv_basis_J.dimension() - 1) { // Prevent the last permutation from occurring.
155  target_onv_basis_J.transformONVToNextPermutation(onv);
156  }
157  } // Target address (Jt) loop.
158 
159  return DysonOrbital<Scalar>(dyson_coeffs);
160  }

187 
188  // Since we want to calculate the overlap between two wave functions, the ONVs should have an equal number of electrons.
189  // We therefore iterate over the ONVs of the N-electron ONV basis, which all have an electron more, and annihilate in one of the orbitals (let the index of that orbital be called 'p').

191  SpinUnresolvedONV onv = onv_basis_J.constructONVFromAddress(0);
192  double coeff = 0;
193 
194  for (size_t J = 0; J < onv_basis_J.dimension(); J++) { // J loops over addresses of the N-electron spin-unresolved ONV basis.
195  int sign = -1; // Total phase factor of all the annihilations that have occurred.
196  for (size_t e = 0; e < onv_basis_J.numberOfElectrons(); e++) { // Loop over electrons in the ONV.

199  sign *= -1;
