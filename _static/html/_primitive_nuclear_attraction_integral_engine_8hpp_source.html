95  template <typename Z = Shell>
96  enable_if_t<std::is_same<Z, GTOShell>::value, IntegralScalar> calculate(const CartesianGTO& left, const CartesianGTO& right) {
97 
98  // Prepare some variables.
99  const auto i = static_cast<int>(left.cartesianExponents().value(CartesianDirection::x));
100  const auto k = static_cast<int>(left.cartesianExponents().value(CartesianDirection::y));
101  const auto m = static_cast<int>(left.cartesianExponents().value(CartesianDirection::z));
102 
103  const auto j = static_cast<int>(right.cartesianExponents().value(CartesianDirection::x));
104  const auto l = static_cast<int>(right.cartesianExponents().value(CartesianDirection::y));
105  const auto n = static_cast<int>(right.cartesianExponents().value(CartesianDirection::z));
106 
107  const auto a = left.gaussianExponent();
108  const auto b = right.gaussianExponent();
109 
110  const auto K_x = left.center()(CartesianDirection::x);
111  const auto K_y = left.center()(CartesianDirection::y);
112  const auto K_z = left.center()(CartesianDirection::z);
113 
114  const auto L_x = right.center()(CartesianDirection::x);
115  const auto L_y = right.center()(CartesianDirection::y);
116  const auto L_z = right.center()(CartesianDirection::z);
117 
118 
119  // Prepare the McMurchie-Davidson coefficients.
120  const McMurchieDavidsonCoefficient E_x {K_x, a, L_x, b};
121  const McMurchieDavidsonCoefficient E_y {K_y, a, L_y, b};
122  const McMurchieDavidsonCoefficient E_z {K_z, a, L_z, b};
123 
124  const double p = a + b;
125  const Vector<double, 3> P {E_x.centerOfMass(), E_y.centerOfMass(), E_z.centerOfMass()};
126 
127 
128  // Calculate the contributions from every nuclear center.
129  double total_integral {0.0};
130 
131  const auto& nuclei = this->nuclear_attraction_operator.nuclearFramework().nucleiAsVector();
132  for (const auto& nucleus : nuclei) {
133  double integral {0.0};
134 
135  const auto& C = nucleus.position();
136  const HermiteCoulombIntegral R {p, P, C};
137 
138  for (int t = 0; t <= i + j; t++) {
139  for (int u = 0; u <= k + l; u++) {
140  for (int v = 0; v <= m + n; v++) {
141  // Add the contribution to the integral. The prefactor will be applied at the end.
142  integral += E_x(i, j, t) * E_y(k, l, u) * E_z(m, n, v) * R(0, t, u, v);
143  }
144  }
145  }
146  const auto charge = static_cast<double>(nucleus.charge());
147  total_integral += (-charge) * integral;
148  }
149 
150  return 2 * boost::math::constants::pi<double>() / p * total_integral;
151  }
152 };
