161  template <typename Z = Shell>
162  auto quantize(const CoulombRepulsionOperator& fq_op) const -> enable_if_t<std::is_same<Z, GTOShell>::value, RSQTwoElectronOperator<product_t<CoulombRepulsionOperator::Scalar, ExpansionScalar>, CoulombRepulsionOperator::Vectorizer>> {
163

165  using ResultOperator = RSQTwoElectronOperator<ResultScalar, CoulombRepulsionOperator::Vectorizer>;
166 
167  const auto one_op_par = IntegralCalculator::calculateLibintIntegrals(fq_op, this->scalarBasis()); // in AO/scalar basis
168 
169  auto g_par = SquareRankFourTensor<ResultScalar>::Zero(one_op_par.dimension(0)); // 'par' for 'parameters'
170 
171  for (size_t i = 0; i < one_op_par.dimension(0); i++) {
172  for (size_t j = 0; j < one_op_par.dimension(1); j++) {
173  for (size_t k = 0; k < one_op_par.dimension(2); k++) {
174  for (size_t l = 0; l < one_op_par.dimension(3); l++) {
175  g_par(i, j, k, l) = one_op_par(i, j, k, l);
176  }
177  }
178  }
179  }
180 
181  ResultOperator op {g_par}; // op for 'operator'
182  op.transform(this->expansion()); // now in spatial/spin-orbital basis
183  return op;

252  ScalarEvaluableRSQOneElectronOperator<DensityDistribution> quantize(const ElectronicDensityOperator& fq_density_op) const {
253 
254  // There aren't any 'integrals' to be calculated for the density operator: we can just multiply every pair of spatial orbitals.
255  const auto phi = this->spatialOrbitals();
256  const auto K = this->numberOfSpatialOrbitals();
257 
258  SquareMatrix<DensityDistribution> rho_par {K};
259  for (size_t p = 0; p < K; p++) {
260  for (size_t q = 0; q < K; q++) {
261  rho_par(p, q) = phi[p] * phi[q];
262  }
263  }
264 
265  return ScalarEvaluableRSQOneElectronOperator<DensityDistribution> {rho_par};
266  }

276  VectorEvaluableRSQOneElectronOperator<CurrentDensityMatrixElement<ExpansionScalar, CartesianGTO>> quantize(const CurrentDensityOperator& fq_current_density_op) const {
277 
278  using namespace GQCP::literals;
279 
280  // There aren't any 'integrals' to be calculated for the current density operator: we can just multiply every pair of (spatial orbital, spatial orbital derivative).
281  const auto K = this->numberOfSpatialOrbitals();
282  const auto phi = this->spatialOrbitals();
283  const auto dphi = this->spatialOrbitalGradients();
284 
285  std::vector<SquareMatrix<CurrentDensityMatrixElement<ExpansionScalar, CartesianGTO>>> j_par_vector {};
286  for (size_t i = 0; i < 3; i++) {
287 
288  SquareMatrix<CurrentDensityMatrixElement<ExpansionScalar, CartesianGTO>> j_i {K};
289  for (size_t p = 0; p < K; p++) {
290  for (size_t q = 0; q < K; q++) {
291  j_i(p, q) = CurrentDensityMatrixElement<ExpansionScalar, CartesianGTO> {phi[p], phi[q], dphi[p](i), dphi[q](i)};
292  }
293  }
294 
295  j_par_vector.push_back(j_i);
296  }
297 
298  return VectorEvaluableRSQOneElectronOperator<CurrentDensityMatrixElement<ExpansionScalar, CartesianGTO>> {j_par_vector};
299  }

327  std::vector<SpatialOrbital> spatialOrbitals() const {
328 
329  // The spatial orbitals are a linear combination of the basis functions, where every column of the coefficient matrix describes one expansion of a spatial orbital in terms of the basis functions.
330  const auto basis_functions = this->scalar_basis.basisFunctions();
331  const auto& C = this->C;
332 
333 
334  // For all spatial orbitals, proceed to calculate the contraction between the associated coefficient matrix column and the basis functions.
335  std::vector<SpatialOrbital> spatial_orbitals;
336  spatial_orbitals.reserve(this->numberOfSpatialOrbitals());
337  for (size_t p = 0; p < this->numberOfSpatialOrbitals(); p++) {
338 
339  // Calculate the spatial orbitals as a contraction between a column of the coefficient matrix and the basis functions.
340  SpatialOrbital spatial_orbital {};
341  for (size_t mu = 0; mu < basis_functions.size(); mu++) {
342  const auto coefficient = this->expansion().matrix().col(p)(mu);
343  const auto& function = basis_functions[mu];
344  spatial_orbital.append(coefficient, function);
345  }
346 
347  spatial_orbitals.push_back(spatial_orbital);
348  }
349 
350  return spatial_orbitals;
351  }

357  std::vector<Vector<SpatialOrbitalDerivative, 3>> spatialOrbitalGradients() const {
358 
359  const auto K = this->numberOfSpatialOrbitals();
360  const auto spatial_orbitals = this->spatialOrbitals();
361 
362  std::vector<GQCP::Vector<SpatialOrbitalDerivative, 3>> spatial_orbital_gradients {K};
363  for (size_t m = 0; m < 3; m++) {
364  for (size_t p = 0; p < K; p++) {
365  const auto& spatial_orbital = spatial_orbitals[p];
366 
367  // A spatial orbital is a linear combination of basis functions (which are contracted GTOs).
368  const auto& expansion_coefficients = spatial_orbital.coefficients();
369  const auto& basis_functions = spatial_orbital.functions();
370 
371  std::vector<Vector<BasisFunctionDerivative, 3>> basis_function_gradients {K};
372  for (size_t mu = 0; mu < K; mu++) {
373  const auto& expansion_coefficient = expansion_coefficients[mu];
374  const auto& basis_function = basis_functions[mu];
375  const auto contraction_length = basis_function.length();
376 
377  // A basis function (a.k.a. contracted GTO) is a contraction (i.e. a linear combination) of Cartesian GTOs.
378  const auto& contraction_coefficients = basis_function.coefficients();
379  const auto& primitives = basis_function.functions();
380 
381  for (size_t d = 0; d < contraction_length; d++) {
382  const auto& contraction_coefficient = contraction_coefficients[d];
383  const auto primitive_gradient = primitives[d].calculatePositionGradient();
384  basis_function_gradients[mu](m).append(contraction_coefficient, primitive_gradient(m));
385  }
386 
387  spatial_orbital_gradients[p](m).append(expansion_coefficient, basis_function_gradients[mu](m));
388  }
389  }
390  }
391 
392  return spatial_orbital_gradients;
393  }

399  std::vector<Spinor<ExpansionScalar, BasisFunction>> spinOrbitals() const {
400 
401  // The spin-orbitals for a restricted spin-orbital basis can be easily constructed from the spatial orbitals, by assigning a zero component once for the beta component of the spin-orbital and once for the alpha component of the spin-orbital.
402  const auto spatial_orbitals = this->spatialOrbitals();
403 
404  std::vector<Spinor<ExpansionScalar, BasisFunction>> spin_orbitals;
405  spin_orbitals.reserve(this->numberOfSpinors());
406  for (const auto& spatial_orbital : spatial_orbitals) {
407 
408  // Add the alpha- and beta-spin-orbitals accordingly.
409  const Spinor<ExpansionScalar, BasisFunction> alpha_spin_orbital {spatial_orbital, 0}; // the '0' int literal can be converted to a zero EvaluableLinearCombination
410  spin_orbitals.push_back(alpha_spin_orbital);
411 
412  const Spinor<ExpansionScalar, BasisFunction> beta_spin_orbital {0, spatial_orbital};
413  spin_orbitals.push_back(beta_spin_orbital);
414  }
415 
416  return spin_orbitals;
417  }
