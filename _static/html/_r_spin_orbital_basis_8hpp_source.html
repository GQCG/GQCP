154  EvaluatableScalarRSQOneElectronOperator<ScalarFunctionProduct<LinearCombination<double, BasisFunction>>> quantize(const ElectronicDensityOperator& fq_density_op) const {
155 
156  using Evaluatable = ScalarFunctionProduct<LinearCombination<double, BasisFunction>>; // the evaluatable type for the density operator
157  using ResultOperator = EvaluatableScalarRSQOneElectronOperator<Evaluatable>;
158 
159  // There aren't any 'integrals' to be calculated for the density operator: we can just multiply every pair of spatial orbitals.
160  const auto phi = this->spatialOrbitals();
161  const auto K = this->numberOfSpatialOrbitals();
162 
163  SquareMatrix<Evaluatable> rho_par {K}; // the matrix representation ('par' for 'parameters') of the second-quantized (one-electron) density operator
164  for (size_t p = 0; p < K; p++) {
165  for (size_t q = 0; q < K; q++) {
166  rho_par(p, q) = phi[p] * phi[q];
167  }
168  }
169 
170  return ResultOperator(rho_par);
171  }

181  std::vector<SpatialOrbital> spatialOrbitals() const {
182 
183  // The spatial orbitals are a linear combination of the basis functions, where every column of the coefficient matrix describes one expansion of a spatial orbital in terms of the basis functions.
184  const auto basis_functions = this->scalar_basis.basisFunctions();
185  const auto& C = this->C;
186 
187 
188  // For all spatial orbitals, proceed to calculate the contraction between the associated coefficient matrix column and the basis functions.
189  std::vector<SpatialOrbital> spatial_orbitals;
190  spatial_orbitals.reserve(this->numberOfSpatialOrbitals());
191  for (size_t p = 0; p < this->numberOfSpatialOrbitals(); p++) {
192 
193  // Calculate the spatial orbitals as a contraction between a column of the coefficient matrix and the basis functions.
194  SpatialOrbital spatial_orbital {};
195  for (size_t mu = 0; mu < basis_functions.size(); mu++) {
196  const auto coefficient = this->expansion().matrix().col(p)(mu);
197  const auto& function = basis_functions[mu];
198  spatial_orbital.append({coefficient}, {function});
199  }
200 
201  spatial_orbitals.push_back(spatial_orbital);
202  }
203 
204  return spatial_orbitals;
205  }

211  std::vector<Spinor<ExpansionScalar, BasisFunction>> spinOrbitals() const {
212 
213  // The spin-orbitals for a restricted spin-orbital basis can be easily constructed from the spatial orbitals, by assigning a zero component once for the beta component of the spin-orbital and once for the alpha component of the spin-orbital.
214  const auto spatial_orbitals = this->spatialOrbitals();
215 
216  std::vector<Spinor<ExpansionScalar, BasisFunction>> spin_orbitals;
217  spin_orbitals.reserve(this->numberOfSpinors());
218  for (const auto& spatial_orbital : spatial_orbitals) {
219 
220  // Add the alpha- and beta-spin-orbitals accordingly.
221  const Spinor<ExpansionScalar, BasisFunction> alpha_spin_orbital {spatial_orbital, 0}; // the '0' int literal can be converted to a zero LinearCombination
222  spin_orbitals.push_back(alpha_spin_orbital);
223 
224  const Spinor<ExpansionScalar, BasisFunction> beta_spin_orbital {0, spatial_orbital};
225  spin_orbitals.push_back(beta_spin_orbital);
226  }
227 
228  return spin_orbitals;
229  }
