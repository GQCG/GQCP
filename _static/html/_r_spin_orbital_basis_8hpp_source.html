142  template <typename Z = Shell>
143  auto quantize(const CoulombRepulsionOperator& fq_op) const -> enable_if_t<std::is_same<Z, GTOShell>::value, RSQTwoElectronOperator<product_t<CoulombRepulsionOperator::Scalar, ExpansionScalar>, CoulombRepulsionOperator::Vectorizer>> {
144 
145  using ResultScalar = product_t<CoulombRepulsionOperator::Scalar, ExpansionScalar>;
146  using ResultOperator = RSQTwoElectronOperator<ResultScalar, CoulombRepulsionOperator::Vectorizer>;
147 
148  const auto one_op_par = IntegralCalculator::calculateLibintIntegrals(fq_op, this->scalarBasis()); // in AO/scalar basis
149 
150  auto g_par = SquareRankFourTensor<ResultScalar>::Zero(one_op_par.dimension(0)); // 'par' for 'parameters'
151 
152  for (size_t i = 0; i < one_op_par.dimension(0); i++) {
153  for (size_t j = 0; j < one_op_par.dimension(1); j++) {
154  for (size_t k = 0; k < one_op_par.dimension(2); k++) {
155  for (size_t l = 0; l < one_op_par.dimension(3); l++) {
156  g_par(i, j, k, l) = one_op_par(i, j, k, l);
157  }
158  }
159  }
160  }
161 
162  ResultOperator op {g_par}; // op for 'operator'
163  op.transform(this->expansion()); // now in spatial/spin-orbital basis
164  return op;
165  }

179  EvaluatableScalarRSQOneElectronOperator<ScalarFunctionProduct<LinearCombination<double, BasisFunction>>> quantize(const ElectronicDensityOperator& fq_density_op) const {
180 
181  using Evaluatable = ScalarFunctionProduct<LinearCombination<double, BasisFunction>>; // the evaluatable type for the density operator
182  using ResultOperator = EvaluatableScalarRSQOneElectronOperator<Evaluatable>;
183 
184  // There aren't any 'integrals' to be calculated for the density operator: we can just multiply every pair of spatial orbitals.
185  const auto phi = this->spatialOrbitals();
186  const auto K = this->numberOfSpatialOrbitals();
187 
188  SquareMatrix<Evaluatable> rho_par {K}; // the matrix representation ('par' for 'parameters') of the second-quantized (one-electron) density operator
189  for (size_t p = 0; p < K; p++) {
190  for (size_t q = 0; q < K; q++) {
191  rho_par(p, q) = phi[p] * phi[q];
192  }
193  }
194 
195  return ResultOperator(rho_par);
196  }

224  std::vector<SpatialOrbital> spatialOrbitals() const {
225 
226  // The spatial orbitals are a linear combination of the basis functions, where every column of the coefficient matrix describes one expansion of a spatial orbital in terms of the basis functions.
227  const auto basis_functions = this->scalar_basis.basisFunctions();
228  const auto& C = this->C;
229 
230 
231  // For all spatial orbitals, proceed to calculate the contraction between the associated coefficient matrix column and the basis functions.
232  std::vector<SpatialOrbital> spatial_orbitals;
233  spatial_orbitals.reserve(this->numberOfSpatialOrbitals());
234  for (size_t p = 0; p < this->numberOfSpatialOrbitals(); p++) {
235 
236  // Calculate the spatial orbitals as a contraction between a column of the coefficient matrix and the basis functions.
237  SpatialOrbital spatial_orbital {};
238  for (size_t mu = 0; mu < basis_functions.size(); mu++) {
239  const auto coefficient = this->expansion().matrix().col(p)(mu);
240  const auto& function = basis_functions[mu];
241  spatial_orbital.append({coefficient}, {function});
242  }
243 
244  spatial_orbitals.push_back(spatial_orbital);
245  }
246 
247  return spatial_orbitals;
248  }

254  std::vector<Spinor<ExpansionScalar, BasisFunction>> spinOrbitals() const {
255 
256  // The spin-orbitals for a restricted spin-orbital basis can be easily constructed from the spatial orbitals, by assigning a zero component once for the beta component of the spin-orbital and once for the alpha component of the spin-orbital.
257  const auto spatial_orbitals = this->spatialOrbitals();
258 
259  std::vector<Spinor<ExpansionScalar, BasisFunction>> spin_orbitals;
260  spin_orbitals.reserve(this->numberOfSpinors());
261  for (const auto& spatial_orbital : spatial_orbitals) {
262 
263  // Add the alpha- and beta-spin-orbitals accordingly.
264  const Spinor<ExpansionScalar, BasisFunction> alpha_spin_orbital {spatial_orbital, 0}; // the '0' int literal can be converted to a zero LinearCombination
265  spin_orbitals.push_back(alpha_spin_orbital);
266 
267  const Spinor<ExpansionScalar, BasisFunction> beta_spin_orbital {0, spatial_orbital};
268  spin_orbitals.push_back(beta_spin_orbital);
269  }
270 
271  return spin_orbitals;
272  }
