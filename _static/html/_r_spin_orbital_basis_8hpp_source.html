254  ScalarEvaluableRSQOneElectronOperator<DensityDistribution> quantize(const ElectronicDensityOperator& fq_density_op) const {
255 
256  // There aren't any 'integrals' to be calculated for the density operator: we can just multiply every pair of spatial orbitals.
257  const auto phi = this->spatialOrbitals();
258  const auto K = this->numberOfSpatialOrbitals();
259 
260  SquareMatrix<DensityDistribution> rho_par {K};
261  for (size_t p = 0; p < K; p++) {
262  for (size_t q = 0; q < K; q++) {
263  rho_par(p, q) = phi[p] * phi[q];
264  }
265  }
266 
267  return ScalarEvaluableRSQOneElectronOperator<DensityDistribution> {rho_par};
268  }

278  VectorEvaluableRSQOneElectronOperator<CurrentDensityMatrixElement<ExpansionScalar, CartesianGTO>> quantize(const CurrentDensityOperator& fq_current_density_op) const {
279 
280  using namespace GQCP::literals;
281 
282  // There aren't any 'integrals' to be calculated for the current density operator: we can just multiply every pair of (spatial orbital, spatial orbital derivative).
283  const auto K = this->numberOfSpatialOrbitals();
284  const auto phi = this->spatialOrbitals();
285  const auto dphi = this->spatialOrbitalGradients();
286 
287  std::vector<SquareMatrix<CurrentDensityMatrixElement<ExpansionScalar, CartesianGTO>>> j_par_vector {};

291  for (size_t p = 0; p < K; p++) {
292  for (size_t q = 0; q < K; q++) {
293  j_i(p, q) = CurrentDensityMatrixElement<ExpansionScalar, CartesianGTO> {phi[p], phi[q], dphi[p](i), dphi[q](i)};
294  }
295  }
296

315  template <typename FQOneElectronOperator, typename Z = Shell>
316  auto quantize(const FQOneElectronOperator& fq_one_op) const -> enable_if_t<std::is_same<Z, LondonGTOShell>::value, RSQOneElectronOperator<product_t<typename FQOneElectronOperator::Scalar, ExpansionScalar>, typename FQOneElectronOperator::Vectorizer>> {
317 
318  using ResultScalar = product_t<typename FQOneElectronOperator::Scalar, ExpansionScalar>;
319  using ResultOperator = RSQOneElectronOperator<ResultScalar, typename FQOneElectronOperator::Vectorizer>;
320  using Vectorizer = typename FQOneElectronOperator::Vectorizer;
321 
322  const auto N = FQOneElectronOperator::NumberOfComponents;
323  const auto& vectorizer = FQOneElectronOperator::vectorizer;
324 
325 
326  // The strategy for calculating the matrix representation of the one-electron operator in this spin-orbital basis is to:
327  // 1. Express the operator in the underlying scalar bases and;
328  // 2. Afterwards transform them using the current coefficient matrix.
329  const auto K = this->numberOfSpatialOrbitals();
330 
331 
332  // 1. Express the operator in the underlying scalar basis.
333  auto engine = GQCP::IntegralEngine::InHouse<GQCP::LondonGTOShell>(fq_one_op);
334  const auto F = GQCP::IntegralCalculator::calculate(engine, this->scalarBasis().shellSet(), this->scalarBasis().shellSet());
335 
336 
337  // For each of the components of the operator, place the scalar basis representations into the spinor basis representation.
338  std::array<SquareMatrix<ResultScalar>, N> fs;
339  for (size_t i = 0; i < N; i++) {
340  fs[i] = SquareMatrix<ResultScalar>(F[i]);
341  }
342 
343 
344  // 2. Transform using the current coefficient matrix.
345  StorageArray<SquareMatrix<ResultScalar>, Vectorizer> array {fs, vectorizer};
346  ResultOperator op {array}; // 'op' for 'operator'.
347  op.transform(this->expansion());
348  return op;
349  }

376  template <typename Z = Shell>
377  auto quantize(const DiamagneticOperator& op) const -> enable_if_t<std::is_same<Z, LondonGTOShell>::value, RSQOneElectronOperator<product_t<DiamagneticOperator::Scalar, ExpansionScalar>, DiamagneticOperator::Vectorizer>> {
378 
379  using ResultScalar = product_t<DiamagneticOperator::Scalar, ExpansionScalar>;
380  using ResultOperator = RSQOneElectronOperator<ResultScalar, DiamagneticOperator::Vectorizer>;
381 
382 
383  // Return the diamagnetic operator as a contraction beween the magnetic field and the electronic quadrupole operator.
384  const auto Q = this->quantize(op.electronicQuadrupole()).allParameters();
385  const auto& B = op.magneticField().strength();
386 
387  // Prepare some variables.
388  const auto& B_x = B(CartesianDirection::x);
389  const auto& B_y = B(CartesianDirection::y);
390  const auto& B_z = B(CartesianDirection::z);
391 
392  const auto& Q_xx = Q[DyadicCartesianDirection::xx];
393  const auto& Q_xy = Q[DyadicCartesianDirection::xy];
394  const auto& Q_xz = Q[DyadicCartesianDirection::xz];
395  const auto& Q_yy = Q[DyadicCartesianDirection::yy];
396  const auto& Q_yz = Q[DyadicCartesianDirection::yz];
397  const auto& Q_zz = Q[DyadicCartesianDirection::zz];
398 
399 
400  SquareMatrix<ResultScalar> D_par = 0.125 * ((std::pow(B_y, 2) + std::pow(B_z, 2)) * Q_xx +
401  (std::pow(B_x, 2) + std::pow(B_z, 2)) * Q_yy +
402  (std::pow(B_x, 2) + std::pow(B_y, 2)) * Q_zz -
403  2 * B_x * B_y * Q_xy -
404  2 * B_x * B_z * Q_xz -
405  2 * B_y * B_z * Q_yz);
406 
407  return ResultOperator {D_par};
408  }

497  std::vector<SpatialOrbital> spatialOrbitals() const {
498 
499  // The spatial orbitals are a linear combination of the basis functions, where every column of the coefficient matrix describes one expansion of a spatial orbital in terms of the basis functions.
500  const auto basis_functions = this->scalar_basis.basisFunctions();
501  const auto& C = this->C;
502 
503 
504  // For all spatial orbitals, proceed to calculate the contraction between the associated coefficient matrix column and the basis functions.
505  std::vector<SpatialOrbital> spatial_orbitals;
506  spatial_orbitals.reserve(this->numberOfSpatialOrbitals());
507  for (size_t p = 0; p < this->numberOfSpatialOrbitals(); p++) {
508 
509  // Calculate the spatial orbitals as a contraction between a column of the coefficient matrix and the basis functions.
510  SpatialOrbital spatial_orbital {};
511  for (size_t mu = 0; mu < basis_functions.size(); mu++) {
512  const auto coefficient = this->expansion().matrix().col(p)(mu);
513  const auto& function = basis_functions[mu];
514  spatial_orbital.append(coefficient, function);
515  }
516 
517  spatial_orbitals.push_back(spatial_orbital);
518  }
519 
520  return spatial_orbitals;
521  }

527  std::vector<Vector<SpatialOrbitalDerivative, 3>> spatialOrbitalGradients() const {
528 
529  const auto K = this->numberOfSpatialOrbitals();
530  const auto spatial_orbitals = this->spatialOrbitals();
531 
532  std::vector<GQCP::Vector<SpatialOrbitalDerivative, 3>> spatial_orbital_gradients {K};
533  for (size_t m = 0; m < 3; m++) {
534  for (size_t p = 0; p < K; p++) {
535  const auto& spatial_orbital = spatial_orbitals[p];
536 
537  // A spatial orbital is a linear combination of basis functions (which are contracted GTOs).
538  const auto& expansion_coefficients = spatial_orbital.coefficients();
539  const auto& basis_functions = spatial_orbital.functions();
540 
541  std::vector<Vector<BasisFunctionDerivative, 3>> basis_function_gradients {K};
542  for (size_t mu = 0; mu < K; mu++) {
543  const auto& expansion_coefficient = expansion_coefficients[mu];
544  const auto& basis_function = basis_functions[mu];
545  const auto contraction_length = basis_function.length();
546 
547  // A basis function (a.k.a. contracted GTO) is a contraction (i.e. a linear combination) of Cartesian GTOs.
548  const auto& contraction_coefficients = basis_function.coefficients();
549  const auto& primitives = basis_function.functions();
550 
551  for (size_t d = 0; d < contraction_length; d++) {
552  const auto& contraction_coefficient = contraction_coefficients[d];
553  const auto primitive_gradient = primitives[d].calculatePositionGradient();
554  basis_function_gradients[mu](m).append(contraction_coefficient, primitive_gradient(m));
555  }
556 
557  spatial_orbital_gradients[p](m).append(expansion_coefficient, basis_function_gradients[mu](m));
558  }
559  }
560  }
561 
562  return spatial_orbital_gradients;
563  }

569  std::vector<Spinor<ExpansionScalar, BasisFunction>> spinOrbitals() const {
570 
571  // The spin-orbitals for a restricted spin-orbital basis can be easily constructed from the spatial orbitals, by assigning a zero component once for the beta component of the spin-orbital and once for the alpha component of the spin-orbital.
572  const auto spatial_orbitals = this->spatialOrbitals();
573 
574  std::vector<Spinor<ExpansionScalar, BasisFunction>> spin_orbitals;
575  spin_orbitals.reserve(this->numberOfSpinors());
576  for (const auto& spatial_orbital : spatial_orbitals) {
577 
578  // Add the alpha- and beta-spin-orbitals accordingly.
579  const Spinor<ExpansionScalar, BasisFunction> alpha_spin_orbital {spatial_orbital, 0}; // the '0' int literal can be converted to a zero EvaluableLinearCombination
580  spin_orbitals.push_back(alpha_spin_orbital);
581 
582  const Spinor<ExpansionScalar, BasisFunction> beta_spin_orbital {0, spatial_orbital};
583  spin_orbitals.push_back(beta_spin_orbital);
584  }
585
