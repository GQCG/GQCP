72 StorageArray<Scalar, Vectorizer> calculateExpectationValue(const MixedSpinResolved2DMComponent<Scalar>& d) const {
73
74 // FIXME: This is duplicate code from `SimpleSQTwoElectronOperator`. It would be double work to introduce a temporary intermediate class before resolving issue #559 (https://github.com/GQCG/GQCP/issues/559), which is why we chose to just copy-paste the implementation.
75
76 if (this->numberOfOrbitals() != d.numberOfOrbitals()) {
77 throw std::invalid_argument("MixedUSQTwoElectronOperatorComponent::calculateExpectationValue(const MixedSpinResolved2DMComponent&): The given 2-DM's dimension is not compatible with the two-electron operator.");
78 }
79
80
81 // Calculate the expectation value for every component of the operator.
82 const auto& parameters = this->allParameters();
83 std::vector<Scalar> expectation_values(this->numberOfComponents()); // Zero-initialize the vector with a number of elements.
84
85 for (size_t i = 0; i < this->numberOfComponents(); i++) {
86
87 // Perform the actual contraction (with prefactor 0.5):
88 // 0.5 g(p q r s) d(p q r s)
89 Eigen::Tensor<Scalar, 0> contraction = 0.5 * parameters[i].template einsum<4>("pqrs, pqrs->", d.tensor());
90
91 // As the contraction is a scalar (a tensor of rank 0), we should access using `operator(0)`.
92 expectation_values[i] = contraction(0);
93 }
94
95 return StorageArray<Scalar, Vectorizer> {expectation_values, this->array.vectorizer()}; // convert std::array to Vector
96 }

113 Self transformed(const UTransformationComponent<Scalar>& T, const Spin sigma) const {
114
115 // Since we're only getting T as a matrix, we should convert it to an appropriate tensor to perform contractions.
116 // Although not a necessity for the einsum implementation, it makes it a lot easier to follow the formulas.
117 const GQCP::Tensor<Scalar, 2> T_tensor = Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>>(T.matrix().data(), T.matrix().rows(), T.matrix().cols());
118
119 // We calculate the conjugate as a tensor as well.
120 const GQCP::Tensor<Scalar, 2> T_conjugate = T_tensor.conjugate();
121
122
123 // Depending on the given spin-component, we should either transform the first two, or the second two axes.
124 const auto& parameters = this->allParameters();
125 auto result = this->allParameters();
126
127 for (size_t i = 0; i < this->numberOfComponents(); i++) {
128 switch (sigma) {
129 case Spin::alpha: {
130 const auto temp = T_tensor.template einsum<1>("UQ,TUVW->TQVW", parameters[i]);
131 const auto transformed = T_conjugate.template einsum<1>("TP,TQVW->PQVW", temp);
132 result[i] = transformed;
133 break;
134 }
135
136 case Spin::beta: {
137 const auto temp = parameters[i].template einsum<1>("PQVW, WS->PQVS", T_tensor);
138 const auto transformed = temp.template einsum<1>("PQVS, VR->PQRS", T_conjugate);
139 result[i] = transformed;
140 break;
141 }
142 }
143 }
144
145 return Self {StorageArray<SquareRankFourTensor<Scalar>, Vectorizer>(result, this->array.vectorizer())};
146 }
