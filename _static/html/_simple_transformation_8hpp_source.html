211  DerivedTransformation rotated(const JacobiRotationType& jacobi_rotation) const override {
212 
213  const auto p = jacobi_rotation.p();

215 
216  // Create the matrix representation of a Jacobi rotation using Eigen's APIs.
217  // We're applying the Jacobi rotation as J = I * jacobi_rotation (cfr. B' = B T).
218  const auto eigen_jacobi_rotation = jacobi_rotation.Eigen();
219  auto result = this->matrix();
220  result.applyOnTheRight(p, q, eigen_jacobi_rotation);
221 
222  return DerivedTransformation {result};
