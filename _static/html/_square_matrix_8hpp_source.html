115  static Self FromStrictTriangle(const VectorX<Scalar>& v) {
116 
117  size_t dim = strictTriangularRootOf(v.size());
118  Self A = Self::Zero(dim);
119 
120  size_t column_index = 0;
121  size_t row_index = column_index + 1; // fill the lower triangle
122  for (size_t vector_index = 0; vector_index < v.size(); vector_index++) {
123  A(row_index, column_index) = v(vector_index);
124 
125  if (row_index == dim - 1) { // -1 because of computers
126  column_index++;
127  row_index = column_index + 1;
128  } else {
129  row_index++;
130  }
131  }
132  return A;
133  }

296  std::array<Self, 2> noPivotLUDecompose() const {
297 
298  const auto M = this->dimension();
299 
300  Self L = Self::Zero(M);
301  Self U = Self::Zero(M);
302 
303  // Algorithm from "https://www.geeksforgeeks.org/doolittle-algorithm-lu-decomposition/"
304  for (size_t i = 0; i < M; i++) {
305 
306  // Systematically solve for the entries of the upper triangular matrix U:
307  // U_ik = A_ik - (LU)_ik
308  for (size_t k = i; k < M; k++) {
309 
310  // Summation of L(i, j) * U(j, k)
311  double sum = 0.0;
312  for (size_t j = 0; j < i; j++) {
313  sum += (L(i, j) * U(j, k));
314  }
315 
316  // Evaluating U(i, k)
317  U(i, k) = this->operator()(i, k) - sum;
318  }
319 
320  // Systematically solve for the entries of the lower triangular matrix L:
321  // L_ik = (A_ik - (LU)_ik) / U_kk
322  for (size_t k = i; k < M; k++) {
323  if (i == k) {
324  L(i, i) = 1; // diagonal as 1
325  } else {
326 
327  // Summation of L(k, j) * U(j, i)
328  double sum = 0.0;
329  for (size_t j = 0; j < i; j++) {
330  sum += (L(k, j) * U(j, i));
331  }
332 
333  // Evaluating L(k, i)
334  L(k, i) = (this->operator()(k, i) - sum) / U(i, i);
335  }
336  }
337  }
338 
339  // Test if the decomposition was stable
340  Self A = L * U;
341  if (A.isApprox(*this)) {
342  return std::array<Self, 2>({L, U});
343  } else {
344  throw std::runtime_error("SquareMatrix<Scalar>::noPivotLUDecompose(): The decomposition was not stable");
345  }
346  }

362  VectorX<Scalar> pairWiseStrictReduced() const {
363 
364  const auto dim = this->dimension();
365 
366  VectorX<Scalar> m = VectorX<Scalar>::Zero((dim * (dim - 1) / 2)); // strictly lower triangle has dim(dim-1)/2 parameters
367 
368  size_t vector_index = 0;
369  for (size_t q = 0; q < dim; q++) { // "column major" ordering for, so we do p first, then q
370  for (size_t p = q + 1; p < dim; p++) { // strict lower triangle means p > q
371  m(vector_index) = this->operator()(p, q);
372  vector_index++;
373  }
374  }

387  double calculatePermanentCombinatorial() const {
388 
389  // The recursion ends when the given 'matrix' is just a number
390  if ((this->rows() == 1) && (this->cols() == 1)) {
391  return this->operator()(0, 0);
392  }
393 
394  size_t j = 0; // develop by the first column
395  double value = 0.0;

398  }

409  double calculatePermanentRyser() const {
410 
411  size_t n = this->dimension();
412 
413  // Loop over all submatrices of A
414  double value = 0.0; // value of the permanent
415  size_t number_of_submatrices = boost::numeric::converter<size_t, double>::convert(std::pow(2, n));
416  for (size_t S = 1; S < number_of_submatrices; S++) { // there are no 'chosen columns' in S=0
417 
418  // Generate the current submatrix through the Gray code of S: if the bit is 1, the column is chosen
419  size_t gray_code_value = grayCodeOf(S);
420  size_t k = __builtin_popcountll(gray_code_value); // number of columns
421 
422  MatrixX<Scalar> X = MatrixX<Scalar>::Zero(n, k);
423  size_t j = 0; // the column index in X
424  while (gray_code_value != 0) { // loop over the set bits in the Gray code
425  size_t index = __builtin_ctzll(gray_code_value); // the index in the original matrix
426 
427  X.col(j) = this->col(index);
428 
429  gray_code_value ^= gray_code_value & -gray_code_value; // flip the first set bit
430  j++;
431  }
432 
433 
434  // Calculate the product of all the row sums and multiply by the sign
435  double product_of_rowsums = X.array().rowwise().sum().prod();
436 
437  size_t t = n - k; // number of deleted columns
438  int sign = std::pow(-1, t);
439  value += sign * product_of_rowsums;
440  }
441 
442  return value;
443  }
