115 static Self FromStrictTriangle(const VectorX<Scalar>& v) {
116
117 size_t dim = strictTriangularRootOf(v.size());
118 Self A = Self::Zero(dim);
119
120 size_t column_index = 0;
121 size_t row_index = column_index + 1; // fill the lower triangle
122 for (size_t vector_index = 0; vector_index < v.size(); vector_index++) {
123 A(row_index, column_index) = v(vector_index);
124
125 if (row_index == dim - 1) { // -1 because of computers
126 column_index++;
127 row_index = column_index + 1;
128 } else {
129 row_index++;
130 }
131 }
132 return A;
133 }

231 static Self SymmetricFromUpperTriangle(const VectorX<Scalar>& v) {
232
233 size_t dim = triangularRootOf(v.size());
234
235 Self A = Self::Zero(dim);
236
237 size_t k = 0; // vector index
238 for (size_t i = 0; i < dim; i++) { // row index
239 for (size_t j = i; j < dim; j++) { // column index
240 if (i != j) {
241 A(i, j) = v(k);
242 A(j, i) = v(k);
243 } else {
244 A(i, i) = v(k);
245 }
246
247 k++;
248 }
249 }
250
251 return A;
252 }

337 std::array<Self, 2> noPivotLUDecompose() const {
338
339 const auto M = this->dimension();
340
341 Self L = Self::Zero(M);
342 Self U = Self::Zero(M);
343
344 // Algorithm from "https://www.geeksforgeeks.org/doolittle-algorithm-lu-decomposition/"
345 for (size_t i = 0; i < M; i++) {
346
347 // Systematically solve for the entries of the upper triangular matrix U:
348 // U_ik = A_ik - (LU)_ik
349 for (size_t k = i; k < M; k++) {
350
351 // Summation of L(i, j) * U(j, k)
352 double sum = 0.0;
353 for (size_t j = 0; j < i; j++) {
354 sum += (L(i, j) * U(j, k));
355 }
356
357 // Evaluating U(i, k)
358 U(i, k) = this->operator()(i, k) - sum;
359 }
360
361 // Systematically solve for the entries of the lower triangular matrix L:
362 // L_ik = (A_ik - (LU)_ik) / U_kk
363 for (size_t k = i; k < M; k++) {
364 if (i == k) {
365 L(i, i) = 1; // diagonal as 1
366 } else {
367
368 // Summation of L(k, j) * U(j, i)
369 double sum = 0.0;
370 for (size_t j = 0; j < i; j++) {
371 sum += (L(k, j) * U(j, i));
372 }
373
374 // Evaluating L(k, i)
375 L(k, i) = (this->operator()(k, i) - sum) / U(i, i);
376 }
377 }
378 }
379
380 // Test if the decomposition was stable
381 Self A = L * U;
382 if (A.isApprox(*this)) {
383 return std::array<Self, 2>({L, U});
384 } else {
385 throw std::runtime_error("SquareMatrix<Scalar>::noPivotLUDecompose(): The decomposition was not stable");
386 }
387 }

403 VectorX<Scalar> pairWiseStrictReduced() const {
404
405 const auto dim = this->dimension();
406
407 VectorX<Scalar> m = VectorX<Scalar>::Zero((dim * (dim - 1) / 2)); // strictly lower triangle has dim(dim-1)/2 parameters
408
409 size_t vector_index = 0;
410 for (size_t q = 0; q < dim; q++) { // "column major" ordering for, so we do p first, then q
411 for (size_t p = q + 1; p < dim; p++) { // strict lower triangle means p > q
412 m(vector_index) = this->operator()(p, q);
413 vector_index++;
414 }
415 }
416
417 return m;
418 }

428 double calculatePermanentCombinatorial() const {
429
430 // The recursion ends when the given 'matrix' is just a number
431 if ((this->rows() == 1) && (this->cols() == 1)) {
432 return this->operator()(0, 0);
433 }
434
435 size_t j = 0; // develop by the first column
436 double value = 0.0;
437 for (size_t i = 0; i < this->rows(); i++) {
438 value += this->operator()(i, j) * SquareMatrix<Scalar>(this->calculateMinor(i, j)).calculatePermanentCombinatorial(); // need to convert because 'minor' is a base function and isn't guaranteed to be square
439 }
440
441 return value;
442 }

450 double calculatePermanentRyser() const {
451
452 size_t n = this->dimension();
453
454 // Loop over all submatrices of A
455 double value = 0.0; // value of the permanent
456 size_t number_of_submatrices = boost::numeric::converter<size_t, double>::convert(std::pow(2, n));
457 for (size_t S = 1; S < number_of_submatrices; S++) { // there are no 'chosen columns' in S=0
458
459 // Generate the current submatrix through the Gray code of S: if the bit is 1, the column is chosen
460 size_t gray_code_value = grayCodeOf(S);
461 size_t k = __builtin_popcountll(gray_code_value); // number of columns
462
463 MatrixX<Scalar> X = MatrixX<Scalar>::Zero(n, k);
464 size_t j = 0; // the column index in X
465 while (gray_code_value != 0) { // loop over the set bits in the Gray code
466 size_t index = __builtin_ctzll(gray_code_value); // the index in the original matrix
467
468 X.col(j) = this->col(index);
469
470 gray_code_value ^= gray_code_value & -gray_code_value; // flip the first set bit
471 j++;
472 }
473
474
475 // Calculate the product of all the row sums and multiply by the sign
476 double product_of_rowsums = X.array().rowwise().sum().prod();
477
478 size_t t = n - k; // number of deleted columns
479 int sign = std::pow(-1, t);
480 value += sign * product_of_rowsums;
481 }
482
483 return value;
484 }
