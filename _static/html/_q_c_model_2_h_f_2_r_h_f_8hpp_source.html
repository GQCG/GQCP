94  static Scalar calculateElectronicEnergy(const Orbital1DM<Scalar>& D, const ScalarRSQOneElectronOperator<Scalar>& H_core, const ScalarRSQOneElectronOperator<Scalar>& F) {
95 
96  // First, calculate the sum of H_core and F (this saves a contraction).
97  const auto Z = H_core + F;
98 
99  // Convert the matrix Z to an GQCP::Tensor<double, 2> Z_tensor.
100  // Einsum is only implemented for a tensor + a matrix, not for 2 matrices.
101  Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>> Z_t {Z.parameters().data(), D.rows(), D.cols()};
102  Tensor<Scalar, 2> Z_tensor = Tensor<Scalar, 2>(Z_t);
103 
104  // To calculate the electronic energy, we must perform a double contraction (with prefactor 0.5):
105  // 0.5 D(nu mu) Z(mu nu).
106  Tensor<Scalar, 0> contraction = 0.5 * Z_tensor.template einsum<2>("ij,ji->", D);
107 
108  // As the double contraction of two matrices is a scalar (a tensor of rank 0), we should access the value as (0).
109  return contraction(0);
110  }

135  static Scalar calculateOrbitalHessianElement(const RSQHamiltonian<Scalar>& sq_hamiltonian, const size_t N_P, const size_t a, const size_t i, const size_t b, const size_t j) {
136 
137  // Prepare some variables.
138  const auto& g_op = sq_hamiltonian.twoElectron();
139  const auto K = g_op.numberOfOrbitals(); // The number of spatial orbitals.
140  const auto& g = g_op.parameters();
141 
142  const auto orbital_space = RHF<Scalar>::orbitalSpace(K, N_P);
143 
144 
145  double value {0.0};
146 
147  // Inactive Fock matrix part.
148  const auto F = sq_hamiltonian.calculateInactiveFockian(orbital_space).parameters();
149  if (i == j) {
150  value += F(a, b);
151  }
152 
153  if (a == b) {
154  value -= F(i, j);
155  }
156 
157 
158  // The two-electron part.
159  value += 4 * g(a, i, b, j) - g(a, b, i, j) - g(a, j, b, i);
160 
161  return 4 * value;
162  }

171  static ImplicitRankFourTensorSlice<Scalar> calculateOrbitalHessianTensor(const RSQHamiltonian<Scalar>& sq_hamiltonian, const size_t N_P) {
172 
173  // Create an occupied-virtual orbital space.
174  const auto K = sq_hamiltonian.numberOfOrbitals();
175  const auto orbital_space = OrbitalSpace::Implicit({{OccupationType::k_occupied, N_P}, {OccupationType::k_virtual, K - N_P}}); // N_P occupied spatial orbitals, K-N_P virtual spatial orbitals
176 
177  auto hessian = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_virtual, OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_occupied); // zero-initialize an object suitable for the representation of a virtual-occupied,virtual-occupied object (ai,bj)
178 
179  // Loop over all indices (ai,bj) to construct the orbital hessian.
180  for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
181  for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
182 
183  for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
184  for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
185  hessian(a, i, b, j) = RHF<Scalar>::calculateOrbitalHessianElement(sq_hamiltonian, N_P, a, i, b, j);
186  }
187  }
188  }
189  }
190 
191  return hessian;
192  }

203  static Orbital1DM<Scalar> calculateOrthonormalBasis1DM(const size_t K, const size_t N) {
204 
205  if (N % 2 != 0) {
206  throw std::invalid_argument("QCMethod::RHF::calculateOrthonormalBasis1DM(const size_t, const size_t): The number of given electrons cannot be odd for RHF.");
207  }
208 
209  // The 1-DM for RHF looks like (for K=5, N=6):
210  // 2 0 0 0 0
211  // 0 2 0 0 0
212  // 0 0 2 0 0
213  // 0 0 0 0 0
214  // 0 0 0 0 0
215 
216  Orbital1DM<double> D = Orbital1DM<double>::Zero(K);
217  D.topLeftCorner(N / 2, N / 2) = 2 * SquareMatrix<double>::Identity(N / 2);
218 
219  return D;
220  }

231  static Orbital2DM<Scalar> calculateOrthonormalBasis2DM(const size_t K, const size_t N) {
232 
233  if (N % 2 != 0) {
234  throw std::invalid_argument("QCMethod::RHF::calculateOrthonormalBasis1DM(const size_t, const size_t): The number of given electrons cannot be odd for RHF.");
235  }
236 
237  const size_t N_P = N / 2; // The number of electron pairs.
238 
239 
240  // Create the orbital space to determine the loops.
241  const auto orbital_space = RHF<Scalar>::orbitalSpace(K, N_P);
242 
243 
244  // Implement a KISS formula for the RHF 2-DM.
245  Orbital2DM<Scalar> d = Orbital2DM<Scalar>::Zero(K);
246 
247  for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
248  for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
249  for (const auto& k : orbital_space.indices(OccupationType::k_occupied)) {
250  for (const auto& l : orbital_space.indices(OccupationType::k_occupied)) {
251  if ((i == j) && (k == l)) {
252  d(i, j, k, l) += 4.0;
253  }
254 
255  if ((i == l) && (j == k)) {
256  d(i, j, k, l) -= 2.0;
257  }
258  }
259  }
260  }
261  }
262 
263  return d;
264  }

291  static ScalarRSQOneElectronOperator<Scalar> calculateScalarBasisFockMatrix(const Orbital1DM<Scalar>& D, const RSQHamiltonian<Scalar>& sq_hamiltonian) {
292 
293  // Get the two-electron parameters.
294  const auto& g = sq_hamiltonian.twoElectron().parameters();
295 
296  // To calculate G, we must perform two double contractions:
297  // 1. (mu nu|rho lambda) P(lambda rho),
298  const Tensor<Scalar, 2> direct_contraction = g.template einsum<2>("ijkl,lk->ij", D);
299  // 2. -0.5 (mu lambda|rho nu) P(lambda rho).
300  const Tensor<Scalar, 2> exchange_contraction = -0.5 * g.template einsum<2>("ilkj,lk->ij", D);
301 
302  // The previous contractions are Tensor<Scalar, 2> instances. In order to calculate the total G matrix, we will convert them back into GQCP::Matrix<Scalar>.
303  auto G1 = direct_contraction.asMatrix();
304  auto G2 = exchange_contraction.asMatrix();
305 
306  return ScalarRSQOneElectronOperator<Scalar> {sq_hamiltonian.core().parameters() + G1 + G2};
307  }

384  GQCP::MatrixX<Scalar> excitationEnergies() const {
385 
386  // Create the orbital space to determine the loops.
387  const auto orbital_space = this->orbitalSpace();
388 
389  // Determine the number of occupied and virtual orbitals.
390  const auto& n_occ = orbital_space.numberOfOrbitals(OccupationType::k_occupied);
391  const auto& n_virt = orbital_space.numberOfOrbitals(OccupationType::k_virtual);
392 
393  // Calculate the occupied and virtual orbital energies.
394  const auto occupied_energies = this->occupiedOrbitalEnergies();
395  const auto virtual_energies = this->virtualOrbitalEnergies();
396 
397  // Create the F matrix.
398  GQCP::MatrixX<Scalar> F_values(n_virt, n_occ);
399  for (int a = 0; a < n_virt; a++) {
400  for (int i = 0; i < n_occ; i++) {
401  F_values(a, i) = virtual_energies[a] - occupied_energies[i];
402  }
403  }
404  return F_values;
405  }

428  GQCP::MatrixX<Scalar> calculateSingletAStabilityMatrix(const RSQHamiltonian<Scalar>& rsq_hamiltonian) const {
429 
430  // Create the orbital space.
431  const auto orbital_space = this->orbitalSpace();
432 
433  // Create the number of occupied and virtual orbitals.
434  const auto& n_occ = orbital_space.numberOfOrbitals(OccupationType::k_occupied);
435  const auto& n_virt = orbital_space.numberOfOrbitals(OccupationType::k_virtual);
436 
437  // The two electron integrals are extracted from the Hamiltonian.
438  const auto& g = rsq_hamiltonian.twoElectron().parameters();
439 
440  // The elements (F_R)_AA and (F_R)_IJ are the eigenvalues of the one-electron Fock operator.
441  // The excitationEnergies API can be used to find these values.
442  const auto F_values = this->excitationEnergies();
443 
444  // The next step is to create the needed tensor slice.
445  // Zero-initialize an occupied-virtual-occupied-virtual object.
446  auto singlet_A_slice = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_occupied, OccupationType::k_virtual);
447  for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
448  for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
449  for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
450  for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
451  singlet_A_slice(i, a, j, b) = 2.0 * g(a, i, j, b) - g(a, b, j, i);
452  }
453  }
454  }
455  }
456 
457  // Turn the ImplicitRankFourTensorSlice in an actual Tensor.
458  auto singlet_A_iajb = singlet_A_slice.asTensor();
459 
460  // Add the previously calculated F values on the correct positions.
461  for (int a = 0; a < n_virt; a++) {
462  for (int i = 0; i < n_occ; i++) {
463  singlet_A_iajb(i, a, i, a) += F_values(a, i);
464  }
465  }
466 
467  // Finally, reshape the tensor to a matrix.
468  const auto singlet_A_matrix = singlet_A_iajb.reshape(n_occ * n_virt, n_occ * n_virt);
469 
470  return singlet_A_matrix;
471  }
