102 static Scalar calculateElectronicEnergy(const Orbital1DM<Scalar>& D, const ScalarRSQOneElectronOperator<Scalar>& H_core, const ScalarRSQOneElectronOperator<Scalar>& F) {
103
104 // First, calculate the sum of H_core and F (this saves a contraction).
105 const auto Z = H_core + F;
106
107 // Convert the matrix Z to an GQCP::Tensor<double, 2> Z_tensor.
108 // Einsum is only implemented for a tensor + a matrix, not for 2 matrices.
109 Eigen::TensorMap<Eigen::Tensor<const Scalar, 2>> Z_t {Z.parameters().data(), D.matrix().rows(), D.matrix().cols()};
110 Tensor<Scalar, 2> Z_tensor = Tensor<Scalar, 2>(Z_t);
111
112 // To calculate the electronic energy, we must perform a double contraction (with prefactor 0.5):
113 // 0.5 D(nu mu) Z(mu nu).
114 Tensor<Scalar, 0> contraction = 0.5 * Z_tensor.template einsum<2>("ij,ij->", D.matrix());
115
116 // As the double contraction of two matrices is a scalar (a tensor of rank 0), we should access the value as (0).
117 return contraction(0);
118 }

126 MatrixX<Scalar> excitationEnergies() const {
127
128 // Create the orbital space to determine the loops.
129 const auto orbital_space = this->orbitalSpace();
130
131 // Determine the number of occupied and virtual orbitals.
132 const auto& n_occ = orbital_space.numberOfOrbitals(OccupationType::k_occupied);
133 const auto& n_virt = orbital_space.numberOfOrbitals(OccupationType::k_virtual);
134
135 // Calculate the occupied and virtual orbital energies.
136 const auto occupied_energies = this->occupiedOrbitalEnergies();
137 const auto virtual_energies = this->virtualOrbitalEnergies();
138
139 // Create the F matrix.
140 MatrixX<Scalar> F_values(n_virt, n_occ);
141 for (int a = 0; a < n_virt; a++) {
142 for (int i = 0; i < n_occ; i++) {
143 F_values(a, i) = virtual_energies[a] - occupied_energies[i];
144 }
145 }
146 return F_values;
147 }

185 std::vector<Scalar> virtualOrbitalEnergies() const {
186
187 // Determine the number of occupied orbitals.
188 const auto n_occ = this->orbitalSpace().numberOfOrbitals(OccupationType::k_occupied);
189
190 std::vector<Scalar> mo_energies; // We use a std::vector in order to be able to slice the vector later on.
191 for (int i = 0; i < this->numberOfSpatialOrbitals(); i++) {
192 mo_energies.push_back(this->orbitalEnergy(i));
193 }
194
195 // Add the values with indices greater than the occupied orbital indices, i.e. the virtual orbital indices, to the new vector.
196 std::vector<Scalar> mo_energies_virtual;
197 std::copy(mo_energies.begin() + n_occ, mo_energies.end(), std::back_inserter(mo_energies_virtual));
198 return mo_energies_virtual;
199 }

265 static Scalar calculateOrbitalHessianElement(const RSQHamiltonian<Scalar>& sq_hamiltonian, const size_t N_P, const size_t a, const size_t i, const size_t b, const size_t j) {
266
267 // Prepare some variables.
268 const auto& g_op = sq_hamiltonian.twoElectron();
269 const auto K = g_op.numberOfOrbitals(); // The number of spatial orbitals.
270 const auto& g = g_op.parameters();
271
272 const auto orbital_space = RHF<Scalar>::orbitalSpace(K, N_P);
273
274
275 double value {0.0};
276
277 // Inactive Fock matrix part.
278 const auto F = sq_hamiltonian.calculateInactiveFockian(orbital_space).parameters();
279 if (i == j) {
280 value += F(a, b);
281 }
282
283 if (a == b) {
284 value -= F(i, j);
285 }
286
287
288 // The two-electron part.
289 value += 4 * g(a, i, b, j) - g(a, b, i, j) - g(a, j, b, i);
290
291 return 4 * value;
292 }

301 static ImplicitRankFourTensorSlice<Scalar> calculateOrbitalHessianTensor(const RSQHamiltonian<Scalar>& sq_hamiltonian, const size_t N_P) {
302
303 // Create an occupied-virtual orbital space.
304 const auto K = sq_hamiltonian.numberOfOrbitals();
305 const auto orbital_space = OrbitalSpace::Implicit({{OccupationType::k_occupied, N_P}, {OccupationType::k_virtual, K - N_P}}); // N_P occupied spatial orbitals, K-N_P virtual spatial orbitals
306
307 auto hessian = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_virtual, OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_occupied); // zero-initialize an object suitable for the representation of a virtual-occupied,virtual-occupied object (ai,bj)
308
309 // Loop over all indices (ai,bj) to construct the orbital hessian.
310 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
311 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
312
313 for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
314 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
315 hessian(a, i, b, j) = RHF<Scalar>::calculateOrbitalHessianElement(sq_hamiltonian, N_P, a, i, b, j);
316 }
317 }
318 }
319 }
320
321 return hessian;
322 }

333 template <typename Z = Scalar>
334 static enable_if_t<std::is_same<Z, double>::value, ImplicitRankFourTensorSlice<complex>> calculateOrbitalHessianForImaginaryResponse(const RSQHamiltonian<double>& sq_hamiltonian, const OrbitalSpace& orbital_space) {
335
336 using namespace GQCP::literals;
337
338 // Prepare some variables.
339 const auto& g = sq_hamiltonian.twoElectron().parameters();
340 const auto F = sq_hamiltonian.calculateInactiveFockian(orbital_space).parameters();
341
342
343 // Zero-initialize a virtual-occupied-virtual-occupied object.
344 auto H = orbital_space.initializeRepresentableObjectFor<complex>(OccupationType::k_virtual, OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_occupied);
345
346
347 // Calculate the elements of the specific orbital Hessian (ai,bj)
348 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
349 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
350 for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
351 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
352
353 complex value {}; // The term in parentheses.
354
355 // Add the contribution from the orbital energies, i.e. the diagonal term.
356 if ((a == b) && (i == j)) {
357 value += F(a, a) - F(i, i);
358 }
359
360 // Add the contributions from the other terms.
361 value += g(i, b, j, a) - g(i, j, b, a);
362
363 H(a, i, b, j) = -4_ii * value;
364 }
365 }
366 }
367 }
368
369 return H;
370 }

387 MatrixX<Scalar> calculateSingletAStabilityMatrix(const RSQHamiltonian<Scalar>& rsq_hamiltonian) const {
388
389 // Create the orbital space.
390 const auto orbital_space = this->orbitalSpace();
391
392 // Create the number of occupied and virtual orbitals.
393 const auto& n_occ = orbital_space.numberOfOrbitals(OccupationType::k_occupied);
394 const auto& n_virt = orbital_space.numberOfOrbitals(OccupationType::k_virtual);
395
396 // The two electron integrals are extracted from the Hamiltonian.
397 const auto& g = rsq_hamiltonian.twoElectron().parameters();
398
399 // The elements (F_R)_AA and (F_R)_IJ are the eigenvalues of the one-electron Fock operator.
400 // The excitationEnergies API can be used to find these values.
401 const auto F_values = this->excitationEnergies();
402
403 // The next step is to create the needed tensor slice.
404 // Zero-initialize an occupied-virtual-occupied-virtual object.
405 auto singlet_A_slice = orbital_space.template initializeRepresentableObjectFor<Scalar>(OccupationType::k_occupied, OccupationType::k_virtual, OccupationType::k_occupied, OccupationType::k_virtual);
406 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
407 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
408 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
409 for (const auto& b : orbital_space.indices(OccupationType::k_virtual)) {
410 singlet_A_slice(i, a, j, b) = 2.0 * g(a, i, j, b) - g(a, b, j, i);
411 }
412 }
413 }
414 }
415
416 // Turn the ImplicitRankFourTensorSlice in an actual Tensor.
417 auto singlet_A_iajb = singlet_A_slice.asTensor();
418
419 // Add the previously calculated F values on the correct positions.
420 for (int a = 0; a < n_virt; a++) {
421 for (int i = 0; i < n_occ; i++) {
422 singlet_A_iajb(i, a, i, a) += F_values(a, i);
423 }
424 }
425
426 // Finally, reshape the tensor to a matrix.
427 const auto singlet_A_matrix = singlet_A_iajb.reshape(n_occ * n_virt, n_occ * n_virt);
428
429 return singlet_A_matrix;
430 }

604 static Orbital1DM<Scalar> calculateOrthonormalBasis1DM(const size_t K, const size_t N) {
605
606 if (N % 2 != 0) {
607 throw std::invalid_argument("QCMethod::RHF::calculateOrthonormalBasis1DM(const size_t, const size_t): The number of given electrons cannot be odd for RHF.");
608 }
609
610 // The 1-DM for RHF looks like (for K=5, N=6):
611 // 2 0 0 0 0
612 // 0 2 0 0 0
613 // 0 0 2 0 0
614 // 0 0 0 0 0
615 // 0 0 0 0 0
616
617 SquareMatrix<Scalar> D = SquareMatrix<Scalar>::Zero(K);
618 D.topLeftCorner(N / 2, N / 2) = 2 * SquareMatrix<Scalar>::Identity(N / 2);
619
620 return Orbital1DM<Scalar> {D};
621 }

643 static Orbital2DM<Scalar> calculateOrthonormalBasis2DM(const size_t K, const size_t N) {
644
645 if (N % 2 != 0) {
646 throw std::invalid_argument("QCMethod::RHF::calculateOrthonormalBasis2DM(const size_t, const size_t): The number of given electrons cannot be odd for RHF.");
647 }
648
649 const size_t N_P = N / 2; // The number of electron pairs.
650
651
652 // Create the orbital space to determine the loops.
653 const auto orbital_space = RHF<Scalar>::orbitalSpace(K, N_P);
654
655
656 // Implement a KISS formula for the RHF 2-DM.
657 SquareRankFourTensor<Scalar> d = SquareRankFourTensor<Scalar>::Zero(K);
658
659 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
660 for (const auto& j : orbital_space.indices(OccupationType::k_occupied)) {
661 for (const auto& k : orbital_space.indices(OccupationType::k_occupied)) {
662 for (const auto& l : orbital_space.indices(OccupationType::k_occupied)) {
663 if ((i == j) && (k == l)) {
664 d(i, j, k, l) += 4.0;
665 }
666
667 if ((i == l) && (j == k)) {
668 d(i, j, k, l) -= 2.0;
669 }
670 }
671 }
672 }
673 }
674
675 return Orbital2DM<Scalar>(d);
676 }

750 static ScalarRSQOneElectronOperator<Scalar> calculateScalarBasisFockMatrix(const Orbital1DM<Scalar>& D, const RSQHamiltonian<Scalar>& sq_hamiltonian) {
751
752 // Get the two-electron parameters.
753 const auto& g = sq_hamiltonian.twoElectron().parameters();
754
755 // To calculate G, we must perform two double contractions:
756 // 1. (mu nu|rho lambda) P(lambda rho),
757 const Tensor<Scalar, 2> direct_contraction = g.template einsum<2>("ijkl,kl->ij", D.matrix());
758 // 2. -0.5 (mu lambda|rho nu) P(lambda rho).
759 const Tensor<Scalar, 2> exchange_contraction = -0.5 * g.template einsum<2>("ijkl,kj->il", D.matrix());
760
761 // The previous contractions are Tensor<Scalar, 2> instances. In order to calculate the total G matrix, we will convert them back into GQCP::Matrix<Scalar>.
762 auto G1 = direct_contraction.asMatrix();
763 auto G2 = exchange_contraction.asMatrix();
764

883 template <typename Z = Scalar>
884 enable_if_t<std::is_same<Z, double>::value, Matrix<complex, Dynamic, 3>> calculateMagneticFieldResponseForce(const VectorRSQOneElectronOperator<complex>& L_op) const {
885
886 // Prepare some variables.
887 const auto orbital_space = this->orbitalSpace();
888 const auto& L = L_op.allParameters();
889
890 // Every column of the matrix `F_kappa_B` contains the response force along the given component: x, y, z.
891 const auto dim = orbital_space.numberOfExcitations(OccupationType::k_occupied, OccupationType::k_virtual);
892 Matrix<complex, Dynamic, 3> F_kappa_B = Matrix<complex, Dynamic, 3>::Zero(dim, 3);
893 for (size_t m = 0; m < 3; m++) { // `m` labels a Cartesian direction.
894
895 // Initialize a virtual-occupied object for every component.
896 auto F_kappa_B_m = orbital_space.template initializeRepresentableObjectFor<complex>(OccupationType::k_virtual, OccupationType::k_occupied);
897
898 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
899 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
900 F_kappa_B_m(a, i) = -2.0 * L[m](i, a);
901 }
902 }
903 F_kappa_B.col(m) = F_kappa_B_m.asVector();
904 }
905
906 return F_kappa_B;
907 }

917 template <typename Z = Scalar>
918 enable_if_t<std::is_same<Z, double>::value, Matrix<complex, Dynamic, 6>> calculateGaugeOriginTranslationResponseForce(const VectorRSQOneElectronOperator<complex>& p_op) const {
919
920 // Prepare some variables.
921 const LeviCivitaTensor<double> epsilon {};
922 const auto& p = p_op.allParameters();
923 const auto orbital_space = this->orbitalSpace();
924
925
926 // Every column of the matrix `F_kappa_G_mn` contains the response force along the given component: xy, xz, yx, yz, zx, zy.
927 const auto dim = orbital_space.numberOfExcitations(OccupationType::k_occupied, OccupationType::k_virtual);
928 Matrix<complex, Dynamic, 6> F_kappa_G = Matrix<complex, Dynamic, 6>::Zero(dim, 6);
929 size_t column_index = 0;
930 for (size_t m = 0; m < 3; m++) { // `m` labels a Cartesian direction.
931 for (size_t n = 0; n < 3; n++) { // `n` labels another Cartesian direction.
932 if (m == n) { // Skip the diagonal components: xx, yy, zz.
933 continue;
934 }
935
936 // Initialize a virtual-occupied object for every component.
937 auto F_kappa_G_mn = orbital_space.template initializeRepresentableObjectFor<complex>(OccupationType::k_virtual, OccupationType::k_occupied);
938
939 for (const auto& a : orbital_space.indices(GQCP::OccupationType::k_virtual)) {
940 for (const auto& i : orbital_space.indices(GQCP::OccupationType::k_occupied)) {
941 const auto f = epsilon.nonZeroIndex(m, n);
942
943 F_kappa_G_mn(a, i) = 2.0 * epsilon(m, n, f) * p[f](i, a);
944 }
945 }
946 F_kappa_G.col(column_index) = F_kappa_G_mn.asVector();
947 column_index++;
948 }
949 }
950
951 return F_kappa_G;
952 }

970 static Matrix<complex, 3, 3> calculateIpsocentricMagneticInducibility(const Vector<double, 3>& r, const OrbitalSpace& orbital_space, const Matrix<complex, Dynamic, 3>& x_B, const Matrix<complex, Dynamic, 6>& x_G, const VectorEvaluableRSQOneElectronOperator<CurrentDensityMatrixElement<complex, CartesianGTO>>& j_op) {
971
972 using namespace GQCP::literals;
973
974 // Prepare some variables.
975 const auto& j = j_op.allParameters();
976 Matrix<complex, 3, 3> J = Matrix<complex, 3, 3>::Zero();
977
978 // Loop over both components of the magnetic inducibility.
979 for (size_t u = 0; u < 3; u++) { // `u` loops over the component of the induced current.
980 for (size_t m = 0; m < 3; m++) { // `m` loops over the component of the applied external magnetic field.
981
982 // Initialize a more useful matrix representation of the linear response coefficients related to the magnetic field perturbation.
983 auto x_m_matrix = MatrixX<complex>::FromColumnMajorVector(x_B.col(m), orbital_space.numberOfOrbitals(OccupationType::k_virtual), orbital_space.numberOfOrbitals(OccupationType::k_occupied));
984 auto x_m = orbital_space.createRepresentableObjectFor<complex>(OccupationType::k_virtual, OccupationType::k_occupied, x_m_matrix);
985
986 for (const auto& a : orbital_space.indices(OccupationType::k_virtual)) {
987 for (const auto& i : orbital_space.indices(OccupationType::k_occupied)) {
988
989 // Determine the left factor, i.e. the contribution from the field-free current density matrix elements.
990 complex left_value = 2.0_ii * j[u](i, a)(r);
991
992 // Determine the right factor, i.e. the contribution from the linear response coefficients.
993
994 // 1. The linear response coefficient related to the magnetic field perturbation.
995 complex right_value = -1.0_ii * x_m(a, i);
996
997 // 2. The linear response coefficient related to the gauge origin translation perturbation.
998 for (size_t n = 0; n < 3; n++) {
999
1000 // 'm' cannot be equal to 'n': there is no such linear response.
1001 if (n == m) {
1002 continue;
1003 }
1004
1005 // Determine the compound index 'mn' of the dyadic Cartesian direction in the 6-column matrix representation.
1006 const auto row_major_index = 3 * m + n;
1007 const auto mn = row_major_index < 4 ? row_major_index - 1 : row_major_index - 2;
1008
1009 // Initialize a more useful matrix representation of the linear response coefficients related to the gauge origin translation perturbation.
1010 auto epsilon_mn_matrix = MatrixX<complex>::FromColumnMajorVector(x_G.col(mn), orbital_space.numberOfOrbitals(OccupationType::k_virtual), orbital_space.numberOfOrbitals(OccupationType::k_occupied));
1011 auto epsilon_mn = orbital_space.createRepresentableObjectFor<complex>(OccupationType::k_virtual, OccupationType::k_occupied, epsilon_mn_matrix);
1012
1013 // Use the ipsocentric CSGT step, i.e. d = r - G_0 with G_0 the gauge origin (i.e. the origin).
1014 right_value += -1.0_ii * epsilon_mn(a, i) * r(n);
1015 }
1016
1017 J(u, m) += 2_ii * left_value * right_value;
1018 assert(J(u, m).imag() < 1.0e-12); // The magnetic inducibility should be a real-valued quantity.
1019 }
1020 }
1021 }
1022 }
1023
1024 return J;
1025 }

1040

1042

1053 }
