248

361  template <typename Z = Matrix<Scalar, Rows, 1>>
362  enable_if_t<Self::is_matrix, Z> pairWiseReduced(const size_t start_i = 0, size_t start_j = 0) const {
363 
364  // Initialize the resulting vector
365  Matrix<Scalar, Rows, 1> v((this->rows() - start_i) * (this->cols() - start_j));
366 
367  // Calculate the compound indices and bring the elements from the matrix over into the vector
368  size_t vector_index = 0;
369  for (size_t j = start_j; j < this->cols(); j++) { // "column major" ordering for row_index<-i,j so we do j first, then i
370  for (size_t i = start_i; i < this->rows(); i++) { // in column major indices, columns are contiguous, so the first of two indices changes more rapidly
371 
372  v(vector_index) = this->operator()(i, j);
373  vector_index++;
374  }
375  }
376 
377  return v;
378  }
