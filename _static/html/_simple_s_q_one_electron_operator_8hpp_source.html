96  StorageArray<Scalar, Vectorizer> calculateExpectationValue(const Derived1DM& D) const {
97

99  throw std::invalid_argument("SimpleSQOneElectronOperator::calculateExpectationValue(const Derived1DM<Scalar>&): The given 1-DM's dimension is not compatible with the one-electron operator.");

103  const auto& parameters = this->allParameters();

123  template <typename Z = Scalar>
124  enable_if_t<std::is_same<Z, double>::value, StorageArray<SquareMatrix<double>, Vectorizer>> calculateFockianMatrix(const Derived1DM& D, const Derived2DM& d) const {
125 
126  if (D.numberOfOrbitals() != this->numberOfOrbitals()) {
127  throw std::invalid_argument("SimpleSQOneElectronOperator::calculateFockianMatrix(const Derived1DM&, const Derived2DM&): The 1-DM's dimensions are not compatible with this one-electron operator.");
128  }
129 
130  if (d.numberOfOrbitals() != this->numberOfOrbitals()) {
131  throw std::invalid_argument("SimpleSQOneElectronOperator::calculateFockianMatrix(const Derived1DM&, const Derived2DM&): The 2-DM's dimensions are not compatible with this one-electron operator.");
132  }
133 
134  const auto& parameters = this->allParameters(); // The parameters of the one-electron operator, as a vector.
135  std::vector<SquareMatrix<double>> F_vector {this->numberOfComponents()}; // The resulting vector of Fockian matrices.
136 
137  // A KISS implementation of the calculation of the Fockian matrix.
138  for (size_t i = 0; i < this->numberOfComponents(); i++) {
139  const auto& f_i = parameters[i]; // The matrix representation of the parameters of the i-th component.
140 
141  // Calculate the Fockian matrix for every component and add it to the vector.
142  SquareMatrix<double> F_i = SquareMatrix<double>::Zero(this->numberOfOrbitals()); // The Fockian matrix of the i-th component.
143  for (size_t p = 0; p < this->numberOfOrbitals(); p++) {
144  for (size_t q = 0; q < this->numberOfOrbitals(); q++) {
145 
146  for (size_t r = 0; r < this->numberOfOrbitals(); r++) {
147  F_i(p, q) += f_i(q, r) * 0.5 * (D(p, r) + D(r, p)); // Include a factor 1/2 to accommodate for response density matrices.
148  }
149  }
150  } // F_i elements loop
151  F_vector[i] = F_i;
152  }
153 
154  return StorageArray<SquareMatrix<double>, Vectorizer> {F_vector, this->array.vectorizer()};
155  }

169  enable_if_t<std::is_same<Z, double>::value, StorageArray<SquareRankFourTensor<double>, Vectorizer>> calculateSuperFockianMatrix(const Derived1DM& D, const Derived2DM& d) const {
170 
171  if (D.numberOfOrbitals() != this->numberOfOrbitals()) {
172  throw std::invalid_argument("SimpleSQOneElectronOperator::calculateFockianMatrix(const Derived1DM&, const Derived2DM&): The given 1-DM's dimensions are not compatible with this one-electron operator.");
173  }
174 
175  if (d.numberOfOrbitals() != this->numberOfOrbitals()) {
176  throw std::invalid_argument("SimpleSQOneElectronOperator::calculateFockianMatrix(const Derived1DM&, const Derived2DM&): The given 2-DM's dimensions are not compatible with this one-electron operator.");
177  }
178 
179 
180  const auto& parameters = this->allParameters(); // The parameters of the one-electron operator, as a vector.
181  std::vector<SquareRankFourTensor<double>> G_vector {this->numberOfComponents()}; // The resulting vector of super-Fockian matrices.
182  const auto F_vector = this->calculateFockianMatrix(D, d).elements(); // The Fockian matrices are necessary in the calculation of the super-Fockian matrices.
183 
184  // A KISS implementation of the calculation of the super-Fockian matrix.
185  for (size_t i = 0; i < this->numberOfComponents(); i++) {
186 
187  const auto& f_i = parameters[i]; // The matrix representation of the parameters of the i-th component.
188  const auto& F_i = F_vector[i]; // The Fockian matrix of the i-th component.
189 
190  // Calculate the super-Fockian matrix for every component and add it to the array. Add factors 1/2 to accommodate for response density matrices.
191  SquareRankFourTensor<double> G_i {this->numberOfOrbitals()};
192  G_i.setZero();
193  for (size_t p = 0; p < this->numberOfOrbitals(); p++) {
194  for (size_t q = 0; q < this->numberOfOrbitals(); q++) {
195  for (size_t r = 0; r < this->numberOfOrbitals(); r++) {
196  for (size_t s = 0; s < this->numberOfOrbitals(); s++) {
197 
198  if (q == r) {
199  G_i(p, q, r, s) += F_i(p, s);
200  }
201 
202  G_i(p, q, r, s) -= f_i(s, p) * 0.5 * (D(r, q) + D(q, r));
203  }
204  }
205  }
206  } // G_i elements loop
207  G_vector[i] = G_i;
208  }
209 
210  return StorageArray<SquareRankFourTensor<double>, Vectorizer> {G_vector, this->array.vectorizer()};
211  }

262  const auto p = jacobi_rotation.p();
263  const auto q = jacobi_rotation.q();
264  const auto jacobi_rotation_eigen = jacobi_rotation.Eigen();
265 
266  // Calculate the basis transformation for every component of the operator.
267  auto result = this->allParameters();
268  for (size_t i = 0; i < this->numberOfComponents(); i++) {
269  result[i].applyOnTheLeft(p, q, jacobi_rotation_eigen.adjoint());
270  result[i].applyOnTheRight(p, q, jacobi_rotation_eigen);
271  }
