70  DenseVectorizer(const std::array<size_t, A>& dimensions, const Ordering ordering = Ordering::ColumnMajor) :
71  m_ordering {ordering},
72  m_dimensions {dimensions} {
73 
74  // Initialize the strides for every dimension.
75  switch (this->m_ordering) {
76  case Ordering::ColumnMajor: { // the first axis is contiguous
77  for (size_t k = 0; k < A; k++) {
78 
79  size_t stride {1};

81  stride *= this->m_dimensions[l];

83 
84  this->m_strides[k] = stride;

94  stride *= this->m_dimensions[l];
