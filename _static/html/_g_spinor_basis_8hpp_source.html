295  template <typename FQOneElectronOperator, typename Z = Shell>
296  auto quantize(const FQOneElectronOperator& fq_one_op) const -> enable_if_t<std::is_same<Z, GTOShell>::value, GSQOneElectronOperator<product_t<typename FQOneElectronOperator::Scalar, ExpansionScalar>, typename FQOneElectronOperator::Vectorizer>> {
297 
298  using ResultScalar = product_t<typename FQOneElectronOperator::Scalar, ExpansionScalar>;
299  using ResultOperator = GSQOneElectronOperator<ResultScalar, typename FQOneElectronOperator::Vectorizer>;
300 
301 
302  // The strategy for calculating the matrix representation of the one-electron operator in this spinor basis is to:
303  // 1. Express the operator in the underlying scalar bases; and
304  // 2. Afterwards transform them using the current coefficient matrix.
305  const auto K_alpha = this->numberOfCoefficients(Spin::alpha);
306  const auto K_beta = this->numberOfCoefficients(Spin::beta);
307  const auto M = this->numberOfSpinors();
308  SquareMatrix<ResultScalar> f = SquareMatrix<ResultScalar>::Zero(M); // the total result
309 
310  // 1. Express the operator in the underlying scalar bases: spin-independent operators only have alpha-alpha and beta-beta blocks.
311  const auto F_alpha = IntegralCalculator::calculateLibintIntegrals(fq_one_op, this->scalarBases().alpha());
312  const auto F_beta = IntegralCalculator::calculateLibintIntegrals(fq_one_op, this->scalarBases().beta());
313 
314  f.topLeftCorner(K_alpha, K_alpha) = F_alpha;
315  f.bottomRightCorner(K_beta, K_beta) = F_beta;
316 
317  // 2. Transform using the current coefficient matrix.
318  ResultOperator op {{f}}; // op for 'operator'
319  op.transform(this->expansion());
320  return op;
321  }

331  template <typename Z = Shell>
332  auto quantize(const ElectronicSpinOperator& fq_one_op) const -> enable_if_t<std::is_same<Z, GTOShell>::value, GSQOneElectronOperator<product_t<ElectronicSpinOperator::Scalar, ExpansionScalar>, ElectronicSpinOperator::Vectorizer>> {
333 
334  using ResultScalar = product_t<ElectronicSpinOperator::Scalar, ExpansionScalar>;
335  using ResultOperator = GSQOneElectronOperator<ResultScalar, ElectronicSpinOperator::Vectorizer>;
336 
337  const auto K_alpha = this->numberOfCoefficients(Spin::alpha);
338  const auto K_beta = this->numberOfCoefficients(Spin::beta);
339  const auto M = this->numberOfSpinors();
340 
341  // The strategy to quantize the spin operator is as follows.
342  // 1. First, calculate the necessary overlap integrals over the scalar bases.
343  // 2. Then, construct the scalar basis representations of the components of the spin operator by placing the overlaps into the correct blocks.
344  // 3. Transform the components (in scalar basis) with the current coefficient matrix to yield the components in spinor basis.
345 
346  SquareMatrix<ResultScalar> S_x = SquareMatrix<ResultScalar>::Zero(M);
347  SquareMatrix<ResultScalar> S_y = SquareMatrix<ResultScalar>::Zero(M);
348  SquareMatrix<ResultScalar> S_z = SquareMatrix<ResultScalar>::Zero(M);
349 
350 
351  // 1. Calculate the necessary overlap integrals over the scalar bases.
352  const auto S_aa = IntegralCalculator::calculateLibintIntegrals(OverlapOperator(), this->scalarBases().alpha());
353  const auto S_ab = IntegralCalculator::calculateLibintIntegrals(OverlapOperator(), this->scalarBases().alpha(), this->scalarBases().beta());
354  const auto S_ba = IntegralCalculator::calculateLibintIntegrals(OverlapOperator(), this->scalarBases().beta(), this->scalarBases().alpha());
355  const auto S_bb = IntegralCalculator::calculateLibintIntegrals(OverlapOperator(), this->scalarBases().beta());
356 
357 
358  // 2. Place the overlaps into the correct blocks.
359  S_x.block(0, K_alpha, K_alpha, K_beta) = 0.5 * S_ab;
360  S_x.block(K_alpha, 0, K_beta, K_alpha) = 0.5 * S_ba;
361 
362  using namespace GQCP::literals;
363  S_y.block(0, K_alpha, K_alpha, K_beta) = -0.5 * 1_ii * S_ab;
364  S_y.block(K_alpha, 0, K_beta, K_alpha) = 0.5 * 1_ii * S_ba;
365 
366  S_z.topLeftCorner(K_alpha, K_alpha) = 0.5 * S_aa;
367  S_z.bottomRightCorner(K_beta, K_beta) = -0.5 * S_bb;
368 
369 
370  // 3. Transform using the coefficient matrix
371  ResultOperator spin_op {std::vector<SquareMatrix<ResultScalar>> {S_x, S_y, S_z}}; // 'op' for operator
372  spin_op.transform(this->expansion());
373  return spin_op;
374  }

384  template <typename Z = Shell>
385  auto quantize(const CoulombRepulsionOperator& coulomb_op) const -> enable_if_t<std::is_same<Z, GTOShell>::value, GSQTwoElectronOperator<product_t<CoulombRepulsionOperator::Scalar, ExpansionScalar>, CoulombRepulsionOperator::Vectorizer>> {
386 
387  using ResultScalar = product_t<CoulombRepulsionOperator::Scalar, ExpansionScalar>;
388  using ResultOperator = GSQTwoElectronOperator<ResultScalar, CoulombRepulsionOperator::Vectorizer>;
389 
390  // The strategy for calculating the matrix representation of the two-electron operator in this spinor basis is to:
391  // 1. Calculate the Coulomb integrals in the underlying scalar bases;
392  // 2. Place the calculated integrals as 'blocks' in the larger representation, so that we can;
393  // 3. Transform the operator using the current coefficient matrix.
394 
395  // 1. Calculate the Coulomb integrals in the underlying scalar bases.
396  const auto g_aaaa = IntegralCalculator::calculateLibintIntegrals(CoulombRepulsionOperator(), this->scalarBases().alpha());
397  const auto g_aabb = IntegralCalculator::calculateLibintIntegrals(CoulombRepulsionOperator(), this->scalarBases().alpha(), this->scalarBases().beta());
398  const auto g_bbaa = IntegralCalculator::calculateLibintIntegrals(CoulombRepulsionOperator(), this->scalarBases().beta(), this->scalarBases().alpha());
399  const auto g_bbbb = IntegralCalculator::calculateLibintIntegrals(CoulombRepulsionOperator(), this->scalarBases().beta());
400 
401 
402  // 2. Place the calculated integrals as 'blocks' in the larger representation
403  const auto K_alpha = this->numberOfCoefficients(Spin::alpha);
404  const auto K_beta = this->numberOfCoefficients(Spin::beta);
405 
406  const auto M = this->numberOfSpinors();
407  auto g_par = SquareRankFourTensor<ResultScalar>::Zero(M); // 'par' for 'parameters'
408 
409  // Primed indices are indices in the larger representation, normal ones are those in the smaller tensors.
410  for (size_t mu_ = 0; mu_ < M; mu_++) { // mu 'prime'
411  const size_t mu = mu_ % K_alpha;
412 
413  for (size_t nu_ = 0; nu_ < M; nu_++) { // nu 'prime'
414  const size_t nu = nu_ % K_alpha;
415 
416  for (size_t rho_ = 0; rho_ < M; rho_++) { // rho 'prime'
417  const size_t rho = rho_ % K_alpha;
418 
419  for (size_t lambda_ = 0; lambda_ < M; lambda_++) { // lambda 'prime'
420  const size_t lambda = lambda_ % K_alpha;
421 
422  if ((mu_ < K_alpha) && (nu_ < K_alpha) && (rho_ < K_alpha) && (lambda_ < K_alpha)) {
423  g_par(mu_, nu_, rho_, lambda_) = g_aaaa(mu, nu, rho, lambda);
424  } else if ((mu_ < K_alpha) && (nu_ < K_alpha) && (rho_ >= K_alpha) && (lambda_ >= K_alpha)) {
425  g_par(mu_, nu_, rho_, lambda_) = g_aabb(mu, nu, rho, lambda);
426  } else if ((mu_ >= K_alpha) && (nu_ >= K_alpha) && (rho_ < K_alpha) && (lambda_ < K_alpha)) {
427  g_par(mu_, nu_, rho_, lambda_) = g_bbaa(mu, nu, rho, lambda);
428  } else if ((mu_ >= K_alpha) && (nu_ >= K_alpha) && (rho_ >= K_alpha) && (lambda_ >= K_alpha)) {
429  g_par(mu_, nu_, rho_, lambda_) = g_bbbb(mu, nu, rho, lambda);
430  }
431  }
432  }
433  }
434  }
435 
436 
437  // 3. Transform the operator using the current coefficient matrix.
438  ResultOperator g_op {g_par}; // 'op' for 'operator'
439  g_op.transform(this->expansion());
440  return g_op;
441  }

455  template <typename FQOneElectronOperator, typename Z = Shell>
456  auto quantize(const FQOneElectronOperator& fq_one_op) const -> enable_if_t<std::is_same<Z, LondonGTOShell>::value, GSQOneElectronOperator<product_t<typename FQOneElectronOperator::Scalar, ExpansionScalar>, typename FQOneElectronOperator::Vectorizer>> {
457 
458  using ResultScalar = product_t<typename FQOneElectronOperator::Scalar, ExpansionScalar>;
459  using ResultOperator = GSQOneElectronOperator<ResultScalar, typename FQOneElectronOperator::Vectorizer>;
460  using Vectorizer = typename FQOneElectronOperator::Vectorizer;
461 
462  const auto N = FQOneElectronOperator::NumberOfComponents;
463  const auto& vectorizer = FQOneElectronOperator::vectorizer;
464 
465 
466  // The strategy for calculating the matrix representation of the one-electron operator in this spinor basis is to:
467  // 1. Express the operator in the underlying scalar bases and;
468  // 2. Afterwards transform them using the current coefficient matrix.
469  const auto K_alpha = this->numberOfCoefficients(Spin::alpha);
470  const auto K_beta = this->numberOfCoefficients(Spin::beta);
471  const auto M = this->numberOfSpinors();
472 
473 
474  // 1. Express the operator in the underlying scalar bases: spin-independent operators only have alpha-alpha and beta-beta blocks.
475  auto engine = GQCP::IntegralEngine::InHouse<GQCP::LondonGTOShell>(fq_one_op);
476  const auto F_aa = GQCP::IntegralCalculator::calculate(engine, this->scalarBases().alpha().shellSet(), this->scalarBases().alpha().shellSet());
477  const auto F_bb = GQCP::IntegralCalculator::calculate(engine, this->scalarBases().beta().shellSet(), this->scalarBases().beta().shellSet());
478 
479  // For each of the components of the operator, place the scalar basis representations into the spinor basis representation.
480  std::array<SquareMatrix<ResultScalar>, N> fs;
481  for (size_t i = 0; i < N; i++) {
482  SquareMatrix<ResultScalar> f = SquareMatrix<ResultScalar>::Zero(M);
483 
484  f.topLeftCorner(K_alpha, K_alpha) = F_aa[i];
485  f.bottomRightCorner(K_beta, K_beta) = F_bb[i];
486 
487  fs[i] = f;
488  }
489 
490 
491  // 2. Transform using the current coefficient matrix.
492  StorageArray<SquareMatrix<ResultScalar>, Vectorizer> array {fs, vectorizer};
493  ResultOperator op {array}; // 'op' for 'operator'.
494  op.transform(this->expansion());
495  return op;
496  }

577  template <typename Z = Shell>
578  auto quantize(const DiamagneticOperator& op) const -> enable_if_t<std::is_same<Z, LondonGTOShell>::value, GSQOneElectronOperator<product_t<DiamagneticOperator::Scalar, ExpansionScalar>, DiamagneticOperator::Vectorizer>> {

581  using ResultOperator = GSQOneElectronOperator<ResultScalar, DiamagneticOperator::Vectorizer>;
582 
583 
584  // Return the diamagnetic operator as a contraction beween the magnetic field and the electronic quadrupole operator.
585  const auto Q = this->quantize(op.electronicQuadrupole()).allParameters();
586  const auto& B = op.magneticField().strength();
587 
588  // Prepare some variables.
589  const auto& B_x = B(CartesianDirection::x);
590  const auto& B_y = B(CartesianDirection::y);
591  const auto& B_z = B(CartesianDirection::z);
592 
593  const auto& Q_xx = Q[DyadicCartesianDirection::xx];
594  const auto& Q_xy = Q[DyadicCartesianDirection::xy];
595  const auto& Q_xz = Q[DyadicCartesianDirection::xz];
596  const auto& Q_yy = Q[DyadicCartesianDirection::yy];
597  const auto& Q_yz = Q[DyadicCartesianDirection::yz];
598  const auto& Q_zz = Q[DyadicCartesianDirection::zz];
599 
600 
601  SquareMatrix<ResultScalar> D_par = 0.125 * ((std::pow(B_y, 2) + std::pow(B_z, 2)) * Q_xx +
602  (std::pow(B_x, 2) + std::pow(B_z, 2)) * Q_yy +
603  (std::pow(B_x, 2) + std::pow(B_y, 2)) * Q_zz -
604  2 * B_x * B_y * Q_xy -
605  2 * B_x * B_z * Q_xz -
606  2 * B_y * B_z * Q_yz);
607 
608  return ResultOperator {D_par};

647  // 3. Transform the operator using the current coefficient matrix.
648 
649  // 1. Calculate the Coulomb integrals in the underlying scalar bases.
650  auto coulomb_engine = GQCP::IntegralEngine::InHouse<GQCP::LondonGTOShell>(CoulombRepulsionOperator());
651  const auto g = GQCP::IntegralCalculator::calculate(coulomb_engine, this->scalarBases().alpha().shellSet(), this->scalarBases().alpha().shellSet())[0];
652 
653 
654  // 2. Place the calculated integrals as 'blocks' in the larger representation.
655  const auto K_alpha = this->numberOfCoefficients(Spin::alpha);
656  const auto K_beta = this->numberOfCoefficients(Spin::beta);
657 
658  const auto M = this->numberOfSpinors();
659  auto g_par = SquareRankFourTensor<ResultScalar>::Zero(M); // 'par' for 'parameters'.
660 
661  // Primed indices are indices in the larger representation, normal ones are those in the smaller tensors.
662  for (size_t mu_ = 0; mu_ < M; mu_++) { // Mu 'prime'.
663  const size_t mu = mu_ % K_alpha;
664 
665  for (size_t nu_ = 0; nu_ < M; nu_++) { // Nu 'prime'.
666  const size_t nu = nu_ % K_alpha;
667 
668  for (size_t rho_ = 0; rho_ < M; rho_++) { // Rho 'prime'.
669  const size_t rho = rho_ % K_alpha;
670 
671  for (size_t lambda_ = 0; lambda_ < M; lambda_++) { // Lambda 'prime'.
672  const size_t lambda = lambda_ % K_alpha;
673 
674  if ((mu_ < K_alpha) && (nu_ < K_alpha) && (rho_ < K_alpha) && (lambda_ < K_alpha)) {
675  g_par(mu_, nu_, rho_, lambda_) = g(mu, nu, rho, lambda);
676  } else if ((mu_ < K_alpha) && (nu_ < K_alpha) && (rho_ >= K_alpha) && (lambda_ >= K_alpha)) {
677  g_par(mu_, nu_, rho_, lambda_) = g(mu, nu, rho, lambda);
678  } else if ((mu_ >= K_alpha) && (nu_ >= K_alpha) && (rho_ < K_alpha) && (lambda_ < K_alpha)) {
679  g_par(mu_, nu_, rho_, lambda_) = g(mu, nu, rho, lambda);
680  } else if ((mu_ >= K_alpha) && (nu_ >= K_alpha) && (rho_ >= K_alpha) && (lambda_ >= K_alpha)) {
681  g_par(mu_, nu_, rho_, lambda_) = g(mu, nu, rho, lambda);
682  }
683  }
684  }
685  }
686  }
687 
688 
689  // 3. Transform the operator using the current coefficient matrix.
690  ResultOperator g_op {g_par}; // 'op' for 'operator'.
691  g_op.transform(this->expansion());
692  return g_op;
693  }
