265  template <typename FQOneElectronOperator>
266  auto quantize(const FQOneElectronOperator& fq_one_op) const -> GSQOneElectronOperator<product_t<typename FQOneElectronOperator::Scalar, ExpansionScalar>, typename FQOneElectronOperator::Vectorizer> {
267 
268  using ResultScalar = product_t<typename FQOneElectronOperator::Scalar, ExpansionScalar>;
269  using ResultOperator = GSQOneElectronOperator<ResultScalar, typename FQOneElectronOperator::Vectorizer>;
270 
271 
272  // The strategy for calculating the matrix representation of the one-electron operator in this spinor basis is to:
273  // 1. Express the operator in the underlying scalar bases; and
274  // 2. Afterwards transform them using the current coefficient matrix.
275  const auto K_alpha = this->numberOfCoefficients(Spin::alpha);
276  const auto K_beta = this->numberOfCoefficients(Spin::beta);
277  const auto M = this->numberOfSpinors();
278  SquareMatrix<ResultScalar> f = SquareMatrix<ResultScalar>::Zero(M); // the total result
279 
280  // 1. Express the operator in the underlying scalar bases: spin-independent operators only have alpha-alpha and beta-beta blocks.
281  const auto F_alpha = IntegralCalculator::calculateLibintIntegrals(fq_one_op, this->scalarBases().alpha());
282  const auto F_beta = IntegralCalculator::calculateLibintIntegrals(fq_one_op, this->scalarBases().beta());
283 
284  f.topLeftCorner(K_alpha, K_alpha) = F_alpha;
285  f.bottomRightCorner(K_beta, K_beta) = F_beta;
286 
287  // 2. Transform using the current coefficient matrix.
288  ResultOperator op {{f}}; // op for 'operator'
289  op.transform(this->expansion());
290  return op;
291  }

301  auto quantize(const ElectronicSpinOperator& fq_one_op) const -> GSQOneElectronOperator<product_t<ElectronicSpinOperator::Scalar, ExpansionScalar>, ElectronicSpinOperator::Vectorizer> {
302 
303  using ResultScalar = product_t<ElectronicSpinOperator::Scalar, ExpansionScalar>;
304  using ResultOperator = GSQOneElectronOperator<ResultScalar, ElectronicSpinOperator::Vectorizer>;
305 
306  const auto K_alpha = this->numberOfCoefficients(Spin::alpha);
307  const auto K_beta = this->numberOfCoefficients(Spin::beta);
308  const auto M = this->numberOfSpinors();
309 
310  // The strategy to quantize the spin operator is as follows.
311  // 1. First, calculate the necessary overlap integrals over the scalar bases.
312  // 2. Then, construct the scalar basis representations of the components of the spin operator by placing the overlaps into the correct blocks.
313  // 3. Transform the components (in scalar basis) with the current coefficient matrix to yield the components in spinor basis.
314 
315  SquareMatrix<ResultScalar> S_x = SquareMatrix<ResultScalar>::Zero(M);
316  SquareMatrix<ResultScalar> S_y = SquareMatrix<ResultScalar>::Zero(M);
317  SquareMatrix<ResultScalar> S_z = SquareMatrix<ResultScalar>::Zero(M);
318 
319 
320  // 1. Calculate the necessary overlap integrals over the scalar bases.
321  const auto S_aa = IntegralCalculator::calculateLibintIntegrals(Operator::Overlap(), this->scalarBases().alpha());
322  const auto S_ab = IntegralCalculator::calculateLibintIntegrals(Operator::Overlap(), this->scalarBases().alpha(), this->scalarBases().beta());
323  const auto S_ba = IntegralCalculator::calculateLibintIntegrals(Operator::Overlap(), this->scalarBases().beta(), this->scalarBases().alpha());
324  const auto S_bb = IntegralCalculator::calculateLibintIntegrals(Operator::Overlap(), this->scalarBases().beta());
325 
326 
327  // 2. Place the overlaps into the correct blocks.
328  S_x.block(0, K_alpha, K_alpha, K_beta) = 0.5 * S_ab;
329  S_x.block(K_alpha, 0, K_beta, K_alpha) = 0.5 * S_ba;
330 
331  using namespace GQCP::literals;
332  S_y.block(0, K_alpha, K_alpha, K_beta) = -0.5 * 1_ii * S_ab;
333  S_y.block(K_alpha, 0, K_beta, K_alpha) = 0.5 * 1_ii * S_ba;
334 
335  S_z.topLeftCorner(K_alpha, K_alpha) = 0.5 * S_aa;
336  S_z.bottomRightCorner(K_beta, K_beta) = -0.5 * S_bb;
337 
338 
339  // 3. Transform using the coefficient matrix
340  ResultOperator spin_op {std::vector<SquareMatrix<ResultScalar>> {S_x, S_y, S_z}}; // 'op' for operator
341  spin_op.transform(this->expansion());
342  return spin_op;
343  }

353  auto quantize(const CoulombRepulsionOperator& coulomb_op) const -> GSQTwoElectronOperator<product_t<CoulombRepulsionOperator::Scalar, ExpansionScalar>, CoulombRepulsionOperator::Vectorizer> {
354 
355  using ResultScalar = product_t<CoulombRepulsionOperator::Scalar, ExpansionScalar>;
356  using ResultOperator = GSQTwoElectronOperator<ResultScalar, CoulombRepulsionOperator::Vectorizer>;
357 
358  // The strategy for calculating the matrix representation of the two-electron operator in this spinor basis is to:
359  // 1. Calculate the Coulomb integrals in the underlying scalar bases;
360  // 2. Place the calculated integrals as 'blocks' in the larger representation, so that we can;
361  // 3. Transform the operator using the current coefficient matrix.
362 
363  // 1. Calculate the Coulomb integrals in the underlying scalar bases.
364  const auto g_aaaa = IntegralCalculator::calculateLibintIntegrals(Operator::Coulomb(), this->scalarBases().alpha());
365  const auto g_aabb = IntegralCalculator::calculateLibintIntegrals(Operator::Coulomb(), this->scalarBases().alpha(), this->scalarBases().beta());
366  const auto g_bbaa = IntegralCalculator::calculateLibintIntegrals(Operator::Coulomb(), this->scalarBases().beta(), this->scalarBases().alpha());
367  const auto g_bbbb = IntegralCalculator::calculateLibintIntegrals(Operator::Coulomb(), this->scalarBases().beta());
368 
369 
370  // 2. Place the calculated integrals as 'blocks' in the larger representation
371  const auto K_alpha = this->numberOfCoefficients(Spin::alpha);
372  const auto K_beta = this->numberOfCoefficients(Spin::beta);
373 
374  const auto M = this->numberOfSpinors();
375  auto g_par = SquareRankFourTensor<ResultScalar>::Zero(M); // 'par' for 'parameters'
376 
377  // Primed indices are indices in the larger representation, normal ones are those in the smaller tensors.
378  for (size_t mu_ = 0; mu_ < M; mu_++) { // mu 'prime'
379  const size_t mu = mu_ % K_alpha;
380 
381  for (size_t nu_ = 0; nu_ < M; nu_++) { // nu 'prime'
382  const size_t nu = nu_ % K_alpha;
383 
384  for (size_t rho_ = 0; rho_ < M; rho_++) { // rho 'prime'
385  const size_t rho = rho_ % K_alpha;
386 
387  for (size_t lambda_ = 0; lambda_ < M; lambda_++) { // lambda 'prime'
388  const size_t lambda = lambda_ % K_alpha;
389 
390  if ((mu_ < K_alpha) && (nu_ < K_alpha) && (rho_ < K_alpha) && (lambda_ < K_alpha)) {
391  g_par(mu_, nu_, rho_, lambda_) = g_aaaa(mu, nu, rho, lambda);
392  } else if ((mu_ < K_alpha) && (nu_ < K_alpha) && (rho_ >= K_alpha) && (lambda_ >= K_alpha)) {
393  g_par(mu_, nu_, rho_, lambda_) = g_aabb(mu, nu, rho, lambda);
394  } else if ((mu_ >= K_alpha) && (nu_ >= K_alpha) && (rho_ < K_alpha) && (lambda_ < K_alpha)) {
395  g_par(mu_, nu_, rho_, lambda_) = g_bbaa(mu, nu, rho, lambda);
396  } else if ((mu_ >= K_alpha) && (nu_ >= K_alpha) && (rho_ >= K_alpha) && (lambda_ >= K_alpha)) {
397  g_par(mu_, nu_, rho_, lambda_) = g_bbbb(mu, nu, rho, lambda);
398  }
399  }
400  }
401  }
402  }
403 
404 
405  // 3. Transform the operator using the current coefficient matrix.
406  ResultOperator g_op {g_par}; // 'op' for 'operator'
407  g_op.transform(this->expansion());
408  return g_op;
409  }
